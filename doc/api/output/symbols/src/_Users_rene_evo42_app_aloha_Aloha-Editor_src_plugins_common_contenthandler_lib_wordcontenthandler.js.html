  7 define(
  8 ['aloha', 'aloha/jquery', 'aloha/contenthandlermanager'],
  9 function( Aloha, jQuery, ContentHandlerManager ) {
 10 	"use strict";
 11 
 12 	var WordContentHandler = ContentHandlerManager.createHandler({
 13 		/**
 14 		 * Handle the pasting. Try to detect content pasted from word and transform to clean html
 15 		 * @param content
 16 		 */
 17 		handleContent: function( content ) {
 18 			if ( typeof content === 'string' ){
 19 				content = jQuery( '<div>' + content + '</div>' );
 20 			} else if ( content instanceof jQuery ) {
 21 				content = jQuery( '<div>' ).append(content);
 22 			}
 23 
 24 			if (this.detectWordContent(content)) {
 25 				this.transformWordContent(content);
 26 			}
 27 
 28 			return content.html();
 29 		},
 30 
 31 		/**
 32 		 * Check whether the content of the given jQuery object is assumed to be pasted from word.
 33 		 * @param content
 34 		 * @return true for content pasted from word, false for other content
 35 		 */
 36 		detectWordContent: function (content) {
 37 			var wordDetected = false;
 38 			// check every element which was pasted.
 39 
 40 			content.find('*').each(function() {
 41 				// get the element style
 42 				var style = jQuery(this).attr('style'),
 43 					clazz;
 44 
 45 				if (style) {
 46 					// if 'mso' is found somewhere in the style, we found word content
 47 					if (style.toLowerCase().indexOf('mso') >= 0) {
 48 						wordDetected = true;
 49 						return false;
 50 					}
 51 				}
 52 				// get the element class
 53 				clazz = jQuery(this).attr('class');
 54 				if (clazz) {
 55 					// if 'mso' is found somewhere in the class, we found word content
 56 					if (clazz.toLowerCase().indexOf('mso') >= 0) {
 57 						wordDetected = true;
 58 						return false;
 59 					}
 60 				}
 61 			});
 62 			// return the result
 63 			return wordDetected;
 64 		},
 65 
 66 		/**
 67 		 * Check whether the given list span (first span in a paragraph which shall be a list item) belongs to an ordered list
 68 		 * @param listSpan
 69 		 * @return true for ordered lists, false for unordered
 70 		 */
 71 		isOrderedList: function(listSpan) {
 72 			// when the span has fontFamily "Wingdings" it is an unordered list
 73 			if (listSpan.css('fontFamily') == 'Wingdings' || listSpan.css('fontFamily') == 'Symbol') {
 74 				return false;
 75 			}
 76 			// otherwise check for a number, letter or '(' as first character
 77 			return listSpan.text().match(/^([0-9]{1,3}\.)|([0-9]{1,3}\)|([a-zA-Z]{1,5}\.)|([a-zA-Z]{1,5}\)))$/) ? true : false;
 78 		},
 79 
 80 		/**
 81 		 * Transform lists pasted from word
 82 		 * @param content
 83 		 */
 84 		transformListsFromWord: function (content) {
 85 			var that = this,
 86 				negateDetectionFilter, detectionFilter, spans,
 87 				paragraphs, bulletClass, listElementClass;
 88 
 89 			// this will be the class to mark paragraphs that will be transformed to lists
 90 			listElementClass = 'aloha-list-element';
 91 			bulletClass = 'aloha-list-bullet';
 92 
 93 			// first step is to find all paragraphs which will be converted into list elements and mark them by adding the class 'aloha-list-element'
 94 			detectionFilter = 'p.MsoListParagraphCxSpFirst,p.MsoListParagraph,p span';
 95 			paragraphs = content.find(detectionFilter);
 96 			paragraphs.each(function() {
 97 				var jqElem = jQuery(this),
 98 					fontFamily = jqElem.css('font-family') || '',
 99 					msoList = jqElem.css('mso-list') || '',
100 					style = jqElem.attr('style') || '';
101 
102 				// detect special classes
103 				if (jqElem.hasClass('MsoListParagraphCxSpFirst') || jqElem.hasClass('MsoListParagraph')) {
104 					jqElem.addClass(listElementClass);
105 106 				} else if (fontFamily.indexOf('Symbol') >= 0) {
107 					jqElem.closest('p').addClass(listElementClass);
108 				} else if (fontFamily.indexOf('Wingdings') >= 0) {
109 					jqElem.closest('p').addClass(listElementClass);
110 				} else if (msoList !== '') {
111 					jqElem.closest('p').addClass(listElementClass);
112 				} else if (style.indexOf('mso-list') >= 0) {
113 					jqElem.closest('p').addClass(listElementClass);
114 				}
115 			});
116 
117 			// now we search for paragraphs with three levels of nested spans, where the innermost span contains nothing but  
118 			detectionFilter = 'p span span span';
119 			spans = content.find(detectionFilter);
120 			spans.each(function() {
121 				var jqElem = jQuery(this),
122 					innerText = jqElem.text().trim().replace(/ /g, ''),
123 					outerText;
124 
125 				if (innerText.length === 0) {
126 					// check whether the outermost of the three spans contains nothing more than numbering
127 					outerText = jqElem.parent().parent().text().trim().replace(/ /g, '');
128 
129 					// patterns for list numbering
130 					// 1.
131 					// 1)
132 					// (1)
133 					// a.
134 					// a)
135 					// I.
136 					// i.
137 					// o ยง (or any other single character)
138 					if (outerText.match(/^([0-9]{1,3}\.)|([0-9]{1,3}\))|([a-zA-Z]{1,5}\.)|([a-zA-Z]{1,5}\))|(.)$/)) {
139 						jqElem.closest('p').addClass(listElementClass);
140 						jqElem.parent().parent().addClass(bulletClass);
141 					}
142 				}
143 			});
144 
145 			// no detect all marked paragraphs and transform into lists
146 			detectionFilter = 'p.' + listElementClass;
147 			negateDetectionFilter = ':not('+detectionFilter+')';
148 			paragraphs = content.find(detectionFilter);
149 
150 			if (paragraphs.length > 0) {
151 				paragraphs.each(function() {
152 					var jqElem = jQuery(this),
153 						jqNewLi, jqList, ordered, firstSpan, following, lists, margin, nestLevel;
154 
155 					jqElem.removeClass(listElementClass);
156 					// first remove all font tags
157 					jqElem.find('font').each(function() {
158 						jQuery(this).contents().unwrap();
159 					});
160 
161 					// initialize the nestlevel and the margin (we will try to detect nested
162 					// lists by comparing the left margin)
163 					nestLevel = [];
164 					margin = parseFloat(jqElem.css('marginLeft'));
165 					// this array will hold all ul/ol elements
166 					lists = [];
167 					// get all following list elements
168 					following = jqElem.nextUntil(negateDetectionFilter);
169 
170 					// get the first span in the element
171 					firstSpan = jQuery(jqElem.find('span.' + bulletClass));
172 					if (firstSpan.length === 0) {
173 174 						firstSpan = jqElem.find('span').eq(0);
175 					}
176 					// use the span to detect whether the list shall be ordered or unordered
177 					ordered = that.isOrderedList(firstSpan);
178 					// finally remove the span (numbers, bullets are rendered by the browser)
179 					firstSpan.remove();
180 
181 					// create the list element
182 					jqList = jQuery(ordered ? '<ol></ol>' : '<ul></ul>');
183 184 					lists.push(jqList);
185 
186 					// add a new list item
187 					jqNewLi = jQuery('<li></li>');
188 					// add the li into the list
189 					jqList.append(jqNewLi);
190 					// append the contents of the old dom element to the li
191 					jqElem.contents().appendTo(jqNewLi);
192 					// replace the old dom element with the new list
193 					jqElem.replaceWith(jqList);
194 
195 					// now proceed all following list elements
196 					following.each(function() {
197 						var jqElem = jQuery(this),
198 							newMargin, jqNewList;
199 
200 						// remove all font tags
201 						jqElem.find('font').each(function() {
202 							jQuery(this).contents().unwrap();
203 						});
204 						// check the new margin
205 						newMargin = parseFloat(jqElem.css('marginLeft'));
206 
207 						// get the first span
208 						firstSpan = jQuery(jqElem.find('span.' + bulletClass));
209 						if (firstSpan.length === 0) {
210 							firstSpan = jqElem.find('span').eq(0);
211 						}
212 						// ... and use it to detect ordered/unordered list elements (this
213 						// information will only be used at the start of a new list anyway)
214 						ordered = that.isOrderedList(firstSpan);
215 						// remove the span
216 						firstSpan.remove();
217 
218 						// check for nested lists by comparing the margins
219 						if (newMargin > margin) {
220 							// create a new list
221 							jqNewList = jQuery(ordered ? '<ol></ol>' : '<ul></ul>');
222 							// append the new list to the last list item of the prior list
223 							jqList.children(':last').append(jqNewList);
224 
225 							// store the list and increase the nest level
226 							jqList = jqNewList;
227 							lists.push(jqList);
228 							nestLevel.push(newMargin);
229 							margin = newMargin;
230 						} else if (newMargin < margin && nestLevel.length > 0) {
231 							while(nestLevel.length > 0 && nestLevel[nestLevel.length - 1] > newMargin) {
232 								nestLevel.pop();
233 								lists.pop();
234 							}
235 							// end nested list and append element to outer list
236 							jqList = lists[lists.length - 1];
237 							margin = newMargin;
238 						}
239 
240 						// create a list item
241 						jqNewLi = jQuery('<li></li>');
242 						// add the li into the list
243 						jqList.append(jqNewLi);
244 						// append the contents of the old dom element to the li
245 						jqElem.contents().appendTo(jqNewLi);
246 						// remove the old dom element
247 						jqElem.remove();
248 					});
249 				});
250 			}
251 		},
252 
253 		/**
254 		 * Transform Title and Subtitle pasted from word
255 		 * @param content
256 		 */
257 		transformTitles: function(content) {
258 			content.find('p.MsoTitle').each(function() {
259 				// titles will be transformed to h1
260 				Aloha.Markup.transformDomObject(jQuery(this), 'h1');
261 			});
262 			content.find('p.MsoSubtitle').each(function() {
263 				// titles will be transformed to h1
264 				Aloha.Markup.transformDomObject(jQuery(this), 'h2');
265 			});
266 		},
267 
268 		/**
269 		 * This is the main transformation method
270 		 * @param content
271 		 */
272 		transformWordContent: function (content) {
273 			// transform lists
274 			this.transformListsFromWord(content);
275 
276 			// transform titles
277 			this.transformTitles(content);
278 		}
279 	});
280 	
281 	return WordContentHandler;
282 });
