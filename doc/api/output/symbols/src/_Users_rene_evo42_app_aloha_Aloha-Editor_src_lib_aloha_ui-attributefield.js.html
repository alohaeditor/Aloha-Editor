 30 var extTemplate = function ( tpl ) {
 31 	if ( tpl ) {
 32 		tpl = '<tpl for="."><div class="x-combo-list-item">' +
 33 			'<tpl if="this.hasRepositoryTemplate(values)">{[ this.renderRepositoryTemplate(values) ]}</tpl>' +
 34 			'<tpl if="!this.hasRepositoryTemplate(values)">' + tpl + '</tpl>' +
 35 			'</div></tpl>';
 36 	} else {
 37 		tpl = '<tpl for="."><div class="x-combo-list-item">' +
 38 			'<tpl if="this.hasRepositoryTemplate(values)">{[ this.renderRepositoryTemplate(values) ]}</tpl>' +
 39 			'<tpl if="!this.hasRepositoryTemplate(values)"><span><b>{name}</b></span></tpl>' +
 40 			'</div></tpl>';
 41 	}
 42 	return new Ext.XTemplate(
 43 		tpl,
 44 		{
 45  46 			hasRepositoryTemplate : function ( values ) {
 47 				var rep = RepositoryManager.getRepository( values.repositoryId );
 48 				return rep && rep.hasTemplate();
 49 			},
 50 			renderRepositoryTemplate : function ( values ) {
 51 				var rep = RepositoryManager.getRepository( values.repositoryId );
 52 				if ( rep && rep.hasTemplate() ) {
 53 				// create extTemplate if template changed
 54 				if ( !rep._ExtTPL || rep.template !== rep._ExtTPLcache ) {
 55 					rep._ExtTPL = new Ext.XTemplate( rep.template );
 56 					rep._ExtTPLcache = rep.template;
 57 				}
 58 				return rep._ExtTPL.apply( values );
 59 			}
 60 		}
 61 	} );
 62 };

 65 // This will store the last attribute value. We need to keep track of this value
 66 // due to decide whether to update the value on finish editing

 69 Ext.ux.AlohaAttributeField = Ext.extend( Ext.form.ComboBox, {
 70 	typeAhead     : false,
 71 	mode          : 'remote',
 72 	triggerAction : 'all',
 73 	width         : 300,
 74 	hideTrigger   : true,
 75 	minChars      : 3,
 76 	valueField    : 'id',
 77 	displayField  : 'name',
 78 	listEmptyText : i18n.t( 'repository.no_item_found' ),
 79 	loadingText   : i18n.t( 'repository.loading' ) + '...',
 80 	enableKeyEvents : true,
 81 	store: new Ext.data.Store( {
 82 		proxy: new Ext.data.AlohaProxy(),
 83 		reader: new Ext.data.AlohaObjectReader()
 84 	} ),
 85     clickAttached: false, // remember that the click event has been attached to the innerList, as this is not implemented in the combobox
 86     tpl      : extTemplate(),
 87     onSelect : function ( item ) {
 88 		this.setItem( item.data );
 89 		if ( typeof this.alohaButton.onSelect == 'function' ) {
 90 			this.alohaButton.onSelect.call( this.alohaButton, item.data );
 91 		}
 92 		this.collapse();
 93 	},
 94 	finishEditing : function () {
 95 		var target = jQuery( this.getTargetObject() ), color;
 96 		
 97 		// Remove the highlighting and restore original color if was set before
 98 		if ( target ) {
 99 			if ( color = target.attr( 'data-original-background-color' ) ) {
100 				jQuery( target ).css( 'background-color', color );
101 			} else {
102 				jQuery( target ).css( 'background-color', '' );
103 			}
104 			jQuery( target ).removeAttr( 'data-original-background-color' );
105 		}
106 		
107 		// Check whether the attribute was changed since the last focus event. Return early when the attribute was not changed.
108 		if ( lastAttributeValue === target.attr( this.targetAttribute ) ) {
109 			return;
110 		}
111 
112 		// when no resource item was selected, remove any marking of the target object
113 		if ( !this.resourceItem ) {
114 			RepositoryManager.markObject( this.targetObject );
115 		}
116 
117 		if ( this.getValue() === '' ) {
118 			if ( this.wrap ) {
119 				jQuery( this.wrap.dom.children[ 0 ] ).css( 'color', '#AAA' );
120 			}
121 			this.setValue( this.placeholder );
122 		}
123 	},
124     listeners: {
125 		// repository object types could have changed
126 		'beforequery': function ( event ) {
127 			if ( this.noQuery ) {
128 				event.cancel = true;
129 				return;
130 			}
131 			if ( this.store !== null && this.store.proxy !== null ) {
132 				this.store.proxy.setParams( {
133 					objectTypeFilter: this.getObjectTypeFilter(),
134 					queryString: event.query
135 				} );
136 			}
137 		},
138 		'afterrender': function ( obj, event ) {
139 			jQuery( this.wrap.dom.children[ 0 ] ).css( 'color', '#AAA' );
140 			this.setValue( this.placeholder );
141 		},
142 		'keydown': function ( obj, event ) {
143 			// on ENTER or ESC leave the editing
144 			// just remember here the status and remove cursor on keyup event
145 			// Otherwise cursor moves to content and no more blur event happens!!??
146 			if ( event.keyCode == 13 || event.keyCode == 27 ) {
147 				if ( this.isExpanded() ) {
148 					this.ALOHAwasExpanded = true;
149 				} else {
150 					this.ALOHAwasExpanded = false;
151 				}
152 				event.preventDefault();
153 			}
154 		},
155 		'keyup': function ( obj, event ) {
156 			if ( ( event.keyCode == 13 || event.keyCode == 27 ) &&
157 					!this.ALOHAwasExpanded ) {
158 				// work around stupid behavior when moving focus
159 				setTimeout( function () {
160 					// Set focus to link element and select the object
161 					Selection.getRangeObject().select();
162 				}, 0 );
163 			}
164 
165 			// when a resource item was (initially) set, but the current value
166 167 			// is different from the reference value, we unset the resource item
168 			if ( this.resourceItem &&
169 					this.resourceValue !== this.wrap.dom.children[ 0 ].value ) {
170 				this.resourceItem = null;
171 				this.resourceValue = null;
172 			}
173 
174 			// update attribute, but only if no resource item was selected
175 			if ( !this.resourceItem ) {
176 				var v = this.wrap.dom.children[ 0 ].value;
177 				this.setAttribute( this.targetAttribute, v );
178 			}
179 		},
180 		'focus': function ( obj, event ) {
181 			// set background color to give visual feedback which link is modified
182 			var	target = jQuery( this.getTargetObject() ),
183 				s = target.css( 'background-color' );
184 			
185 			if ( this.getValue() === this.placeholder ) {
186 				this.setValue( '' );
187 				jQuery( this.wrap.dom.children[ 0 ] ).css( 'color', 'black' );
188 			}
189 			if ( target && target.context && target.context.style &&
190 					target.context.style[ 'background-color' ] ) {
191 				target.attr( 'data-original-background-color',
192 193 					target.context.style[ 'background-color' ] );
194 			}
195 			target.css( 'background-color', '#80B5F2' );
196 		},
197 		'blur': function ( obj, event ) {
198 			this.finishEditing();
199 		},
200 		'hide': function ( obj, event ) {
201 			this.finishEditing();
202 		},
203 		'expand': function ( combo ) {
204 			if ( this.noQuery ) {
205 				this.collapse();
206 			}
207 			if ( !this.clickAttached ) {
208 				var that = this;
209 				// attach the mousedown event to set the event handled,
210 				// so that the editable will not get deactivated
211 				this.mon( this.innerList, 'mousedown', function ( event ) {
212 					Aloha.eventHandled = true;
213 				}, this );
214 				// in the mouseup event, the flag will be reset
215 				this.mon( this.innerList, 'mouseup', function ( event ) {
216 					Aloha.eventHandled = false;
217 				}, this );
218 				this.clickAttached = true;
219 			}
220 		}
221 	},
222 	setItem: function ( item, displayField ) {
223 		this.resourceItem = item;
224 		
225 		if ( item ) {
226 			displayField = ( displayField ) ? displayField : this.displayField;
227 			// TODO split display field by '.' and get corresponding attribute, because it could be a properties attribute.
228 			var v = item[ displayField ];
229 			// set the value into the field
230 			this.setValue( v );
231 			// store the value to be the "reference" value for the currently selected resource item
232 			this.resourceValue = v;
233 			// set the attribute to the target object
234 			this.setAttribute( this.targetAttribute, item[ this.valueField ] );
235 			// call the repository marker
236 			RepositoryManager.markObject( this.targetObject, item );
237 		} else {
238 			// unset the reference value, since no resource item is selected
239 			this.resourceValue = null;
240 		}
241 	},
242 	getItem: function () {
243 		return this.resourceItem;
244 	},
245 	// Private hack to allow attribute setting by regex
246 	setAttribute: function ( attr, value, regex, reference ) {
247 		var setAttr = true, regxp;
248 		if ( this.targetObject) {
249 			// check if a reference value is submitted to check against with a regex
250 			if ( typeof reference != 'undefined' ) {
251 				regxp = new RegExp( regex );
252 				if ( !reference.match( regxp ) ) {
253 					setAttr = false;
254 				}
255 256 			}
257 
258 			// if no regex was successful or no reference value
259 			// was submitted remove the attribute
260 			if ( setAttr ) {
261 				jQuery( this.targetObject ).attr( attr, value );
262 			} else {
263 				jQuery( this.targetObject ).removeAttr( attr );
264 			}
265 		}
266 	},
267 	setTargetObject : function ( obj, attr ) {
268 	    var that = this;
269 		this.targetObject = obj;
270 	    this.targetAttribute = attr;
271 	    this.setItem( null );
272 	    
273 	    if ( obj && attr ) {
274 	    	lastAttributeValue = jQuery( obj ).attr( attr );
275 	    }
276 
277 		if ( this.targetObject && this.targetAttribute ) {
278 			this.setValue( jQuery( this.targetObject ).attr( this.targetAttribute ) );
279 		} else {
280 			this.setValue( '' );
281 		}
282 
283 		// check whether a repository item is linked to the object
284 		var that = this;
285 		RepositoryManager.getObject( obj, function ( items ) {
286 			if ( items && items.length > 0 ) {
287 				that.setItem( items[ 0 ] );
288 			}
289 		} );
290 	},
291 	getTargetObject : function () {
292 	    return this.targetObject;
293 	},
294 	setObjectTypeFilter : function ( otFilter ) {
295 		this.objectTypeFilter = otFilter;
296 	},
297 	getObjectTypeFilter : function () {
298 		return this.objectTypeFilter;
299 	},
300 	noQuery: true
301 });

314 Ui.AttributeField = Ui.Button.extend( {
315 	_constructor: function ( properties ) {
316 		/**
317 		 * @cfg Function called when an element is selected
318 		 */
319 		this.onSelect = null;
320 321 		this.listenerQueue = [];
322 		this.objectTypeFilter = null;
323 		this.tpl = null;
324 		this.displayField = null;
325 		this.valueField = null;
326 
327 		this.init( properties );
328 	},
329 
330 	/**
331 	 * Create a extjs alohaattributefield
332 	 * @hide
333 	 */
334 	getExtConfigProperties: function () {
335 		var props = {
336 		    alohaButton : this,
337 		    xtype       : 'alohaattributefield',
338 		    rowspan     : this.rowspan || undefined,
339 		    width       : this.width || undefined,
340 		    placeholder : this.placeholder || undefined,
341 		    id          : this.id,
342 		    cls         : this.cls || undefined
343 		};
344 		if ( this.valueField ) {
345 			props.valueField = this.valueField;
346 		}
347 		if ( this.displayField ) {
348 			props.displayField = this.displayField;
349 		}
350 		if ( this.minChars ) {
351 			props.minChars = this.minChars;
352 		}
353 		return props;
354 	},
355 
356 	/**
357 	 * Sets the target Object of which the Attribute should be modified
358 	 * @param {jQuery} obj the target object
359 	 * @param {String} attr Attribute to be modified ex. "href" of a link
360 	 * @void
361 	 */
362 	setTargetObject: function ( obj, attr ) {
363 		if ( this.extButton ) {
364 			this.extButton.setTargetObject( obj, attr );
365 		}
366 	},
367 
368 	/**
369 	 * @return {jQuery} object Returns the current target Object
370 	 */
371 	getTargetObject: function () {
372 		return this.extButton ? this.extButton.getTargetObject() : null;
373 	},
374 
375 	/**
376 	 * Focus to this field
377 	 * @void
378 	 */
379 	focus: function () {
380 		if ( this.extButton ) {
381 			this.extButton.focus();
382 			if ( this.extButton.getValue().length > 0 ) {
383 				this.extButton.selectText( 0, this.extButton.getValue().length );
384 			}
385 		}
386 	},
387 
388 	/**
389 	 * Adding a listener to the field
390 	 * @param {String} eventname The name of the event. Ex. 'keyup'
391 	 * @param {function} handler The function that should be called when the event happens.
392 	 * @param {Object} scope The scope object which the event should be attached
393 	 */
394 	addListener: function ( eventName, handler, scope ) {
395 		var listener;
396 
397 		if ( this.extButton ) {
398 			this.extButton.addListener( eventName, handler, null );
399 		} else {
400 			// if extButton not yet initialized adding listeners could be a problem
401 			// so all events are collected in a queue and added on initalizing
402 			listener = {
403 				'eventName' : eventName,
404 				'handler'   : handler,
405 				'scope'     : scope,
406 				'options'   : null
407 			};
408 			this.listenerQueue.push( listener );
409 		}
410 	},
411 
412 	/**
413 	 * Sets an attribute optionally based on a regex on reference
414 	 * @param {String} attr The Attribute name which should be set. Ex. "lang"
415 	 * @param {String} value The value to set. Ex. "de-AT"
416 	 * @param {String} regex The regex when the attribute should be set. The regex is applied to the value of refernece.
417 	 * @param {String} reference The value for the regex.
418 	 */
419 	setAttribute: function ( attr, value, regex, reference ) {
420 		if ( this.extButton ) {
421 			this.extButton.setAttribute( attr, value, regex, reference );
422 		}
423 	},
424 
425 	/**
426 	 * When at least on objectType is set the value in the Attribute field does a query to all registered repositories.
427 	 * @param {Array} objectTypeFilter The array of objectTypeFilter to be searched for.
428 	 * @void
429 	 */
430 	setObjectTypeFilter: function ( objectTypeFilter ) {
431 		if ( this.extButton ) {
432 			this.noQuery = false;
433 			this.extButton.setObjectType( objectTypeFilter );
434 		} else {
435 			if ( !objectTypeFilter ) {
436 				objectTypeFilter = 'all';
437 			}
438 			this.objectTypeFilter = objectTypeFilter;
439 		}
440 	},
441 
442 	/**
443 	 * Sets an item to the link tag.
444 	 * @param {resourceItem} item
445 	 */
446 	setItem: function ( item , displayField ) {
447 		if ( this.extButton ) {
448 			this.extButton.setItem( item, displayField );
449 		}
450 	},
451 
452 	/**
453 	 * Gets current item set.
454 	 * @return {resourceItem} item
455 	 */
456 	getItem: function () {
457 		if ( this.extButton ) {
458 			return this.extButton.getItem();
459 		}
460 		return null;
461 	},
462 
463 	/**
464 	 * Returns the current value
465 	 * @return {String} attributeValue
466 	 */
467 	getValue: function () {
468 		if ( this.extButton ) {
469 			return this.extButton.getValue();
470 		}
471 		return null;
472 	},
473 
474 	/**
475 	 * Sets the current value
476 	 * @param {String} va attributeValue
477 	 */
478 	setValue: function ( v ) {
479 		if ( this.extButton ) {
480 			this.extButton.setValue( v );
481 		}
482 	},
483 
484 	/**
485 	 * Returns the current query value.
486 	 * @return {String} queryValue
487 	 */
488 	getQueryValue: function () {
489 		if ( this.extButton ) {
490 			//return this.extButton.getValue();
491 			return this.extButton.wrap.dom.children[ 0 ].value;
492 		}
493 		return null;
494 	},
495 
496 	/**
497 	 * Set the display field, which is displayed in the combobox
498 	 * @param {String} displayField name of the field to be displayed
499 	 * @return display field name on success, null otherwise
500 	 */
501 	setDisplayField: function ( displayField ) {
502 		var result;
503 		if ( this.extButton ) {
504 			result = this.extButton.displayField = displayField;
505 		} else {
506 			result = this.displayField = displayField;
507 		}
508 		return result;
509 	},
510 
511 	/**
512 	 * Set the row template for autocomplete hints. The default template is:
513 	 * <span><b>{name}</b><br />{url}</span>
514 	 * @param {String} tpl template to be rendered for each row
515 	 * @return template on success or null otherwise
516 	 */
517 	setTemplate: function ( tpl ) {
518 		var result;
519 		if ( this.extButton ) {
520 			result = this.extButton.tpl = extTemplate( tpl );
521 		} else {
522 			result = this.tpl = extTemplate( tpl );
523 		}
524 		return result;
525 	}
