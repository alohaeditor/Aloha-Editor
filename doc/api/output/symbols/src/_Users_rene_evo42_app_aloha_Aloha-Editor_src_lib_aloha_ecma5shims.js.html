202   // http://www.w3.org/TR/DOM-Level-3-Core/core.html#Node3-compareDocumentPosition
203   // FIXME: Check if the DOMNode prototype can be set.
204   $_.compareDocumentPosition = function(node1, node2) {
205       
206     if ('compareDocumentPosition' in document.documentElement ) {
207       return node1.compareDocumentPosition(node2);
208     } 
209     
210     if (!("contains" in document.documentElement)) {
211       throw 'compareDocumentPosition nor contains is not supported by this browser.';
212     }
213     
214     if (node1 == node2) return 0;
215     
216     //if they don't have the same parent, there's a disconnect
217     if (getRootParent(node1) != getRootParent(node2)) return 1;
218     
219     //use this if both nodes have a sourceIndex (text nodes don't)
220     if ("sourceIndex" in node1 && "sourceIndex" in node2) {
221       return comparePosition(node1, node2);
222     }
223     
224     //document will definitely contain the other node
225     if (node1 == document) return 20;
226     else if (node2 == document) return 10;
227     
228     //get sourceIndexes to use for both nodes
229     var useNode1 = getUseNode(node1), useNode2 = getUseNode(node2);
230     
231     //call this function again to get the result
232     var result = comparePosition(useNode1, useNode2);
233     
234     //clean up if needed
235     if (node1 != useNode1) useNode1.parentNode.removeChild(useNode1);
236     if (node2 != useNode2) useNode2.parentNode.removeChild(useNode2);
237     return result;

240     //node.ownerDocument gives the document object, which isn't the right info for a disconnect
241     function getRootParent(node) {
242       do { var parent = node; }
243       while (node = node.parentNode);
244       return parent;
245     }

247     //Compare Position - MIT Licensed, John Resig; http://ejohn.org/blog/comparing-document-position/
248     //Already checked for equality and disconnect
249     function comparePosition(node1, node2) {
250       return (node1.contains(node2) && 16) +
251         (node2.contains(node1) && 8) +
252           (node1.sourceIndex >= 0 && node2.sourceIndex >= 0 ?
253             (node1.sourceIndex < node2.sourceIndex && 4) +
254               (node1.sourceIndex > node2.sourceIndex && 2) :
255             1);
256     }

258     //get a node with a sourceIndex to use
259     function getUseNode(node) {
260       //if the node already has a sourceIndex, use that node
261       if ("sourceIndex" in node) return node;
262       //otherwise, insert a comment (which has a sourceIndex but minimal DOM impact) before the node and use that
263       return node.parentNode.insertBefore(document.createComment(""), node);
