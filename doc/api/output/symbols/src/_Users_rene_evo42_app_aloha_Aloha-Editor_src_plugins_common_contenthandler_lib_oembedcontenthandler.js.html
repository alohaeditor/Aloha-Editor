 21 	getOEmbedProvider = function(url) {
 22 		for (var i = 0; i < providers.length; i++) {
 23 			if (providers[i].matches(url))
 24 				return providers[i];
 25 		}
 26 		return null;
 27 	};

 29 	getNormalizedParams = function (params) {
 30 		var normalizedParams = {}, key;
 31 		if (params == null)
 32 			return null;
 33 		for ( key in params) {
 34 			if (key != null)
 35 				normalizedParams[key.toLowerCase()] = params[key];
 36 		}
 37 		return normalizedParams;
 38 	};

 40 	OEmbedProvider = function(name, urlPattern, oEmbedUrl, callbackparameter) {
 41 		this.name = name;
 42 		this.urlPattern = urlPattern;
 43 		this.oEmbedUrl = (oEmbedUrl != null) ? oEmbedUrl : "http://oohembed.com/oohembed/";
 44 		this.callbackparameter = (callbackparameter != null) ? callbackparameter : "callback";
 45 		this.maxWidth = 500;
 46 		this.maxHeight = 400;
 47 
 48 		this.matches = function(externalUrl) {
 49 			// TODO: Convert to Regex
 50 			return externalUrl.indexOf(this.urlPattern) >= 0;
 51 		};
 52 
 53 		this.getRequestUrl = function(externalUrl) {
 54 
 55 			var url = this.oEmbedUrl, qs = "", i;
 56 
 57 			if (url.indexOf("?") <= 0)
 58 				url = url + "?";
 59 			else
 60 				url = url + "&";
 61 
 62 			if (this.maxWidth != null && this.params.maxwidth == null)
 63 				this.params.maxwidth = this.maxWidth;				
 64 				
 65 			if (this.maxHeight != null && this.params.maxheight == null)
 66 				this.params.maxheight = this.maxHeight;
 67 
 68 			for ( i in this.params) {
 69 				// We don't want them to jack everything up by changing the callback parameter
 70 				if (i == this.callbackparameter)
 71 				  continue;
 72 				
 73 				// allows the options to be set to null, don't send null values to the server as parameters
 74 				if (this.params[i] != null)
 75 					qs += "&" + escape(i) + "=" + this.params[i];
 76 			}		   
 77 			
 78 				
 79 			url += "format=json&url=" + escape(externalUrl) +		   
 80 					qs + 
 81 					"&" + this.callbackparameter + "=?";
 82 					
 83 			return url;
 84 		};
 85 
 86 		this.embedCode = function(container, externalUrl, callback) {
 87 
 88 			var request = this.getRequestUrl(externalUrl);
 89 
 90 			jQuery.getJSON(request, function(data) {
 91 
 92 				var oembed = data,
 93                     code, type = data.type;
 94 
 95 				switch (type) {
 96 					case "photo":
 97 						oembed.code = jQuery.fn.oembed.getPhotoCode(externalUrl, data);
 98 						break;
 99 					case "video":
100 						oembed.code = jQuery.fn.oembed.getVideoCode(externalUrl, data);
101 						break;
102 					case "rich":
103 						oembed.code = jQuery.fn.oembed.getRichCode(externalUrl, data);
104 						break;
105 					default:
106 						oembed.code = jQuery.fn.oembed.getGenericCode(externalUrl, data);
107 						break;
108 				}
109 
110 				callback(container, oembed);
111 			});
112 		};
113 	};

137 	jQuery.fn.oembed = function(url, options, callback) {
138 
139 		options = jQuery.extend(true, jQuery.fn.oembed.defaults, options);
140 
141 		return this.each(function() {
142 
143 			var container = jQuery(this),
144 				resourceURL = (url != null) ? url : container.attr("href"),
145 				provider;
146 
147 			if (!callback) callback = function(container, oembed) {			
148 				 jQuery.fn.oembed.insertCode(container, options.embedMethod, oembed);
149 			};
150 
151 			if (resourceURL != null) {
152 				provider = getOEmbedProvider(resourceURL);
153 
154 				if (provider != null) {						
155 					provider.params = getNormalizedParams(options[provider.name]) || {};
156 					provider.maxWidth = options.maxWidth;
157 					provider.maxHeight = options.maxHeight;										
158 					provider.embedCode(container, resourceURL, callback);
159 					return;
160 				}
161 			}
162 
163 			callback(container, null);
164 		});
165 	};

174 	jQuery.fn.oembed.insertCode = function(container, embedMethod, oembed) {
175 		if (oembed == null)
176 			return;
177 		switch(embedMethod)
178 		{
179 			case "auto":				
180 				if (container.attr("href") != null) {
181 					jQuery.fn.oembed.insertCode(container, "append", oembed);
182 				}
183 				else {
184 					jQuery.fn.oembed.insertCode(container, "replace", oembed);
185 				}
186 				break;
187 			case "replace":	
188 				container.replaceWith(oembed.code);
189 				break;
190 			case "fill":
191 				container.html(oembed.code);
192 				break;
193 			case "append":
194 				var oembedContainer = container.next();
195 				if (oembedContainer == null || !oembedContainer.hasClass("oembed-container")) {
196 					oembedContainer = container
197 						.after('<div class="oembed-container"></div>')
198 						.next(".oembed-container");
199 					if (oembed != null && oembed.provider_name != null)
200 						oembedContainer.toggleClass("oembed-container-" + oembed.provider_name);		
201 				}
202 				oembedContainer.html(oembed.code);				
203 				break;			
204 		}
205 	};

207 	jQuery.fn.oembed.getPhotoCode = function(url, data) {
208 		var alt = data.title ? data.title : '',
209             code;
210 		alt += data.author_name ? ' - ' + data.author_name : '';
211 		alt += data.provider_name ? ' - ' +data.provider_name : '';
212 		code = '<div><a href="' + url + '" target="_blank"><img src="' + data.url + '" alt="' + alt + '"/></a></div>';
213 		if (data.html)
214 			code += "<div>" + data.html + "</div>";
215 		return code;
216 	};

228 	jQuery.fn.oembed.getGenericCode = function(url, data) {
229 		var title = (data.title != null) ? data.title : url,

231 		if (data.html)
232 			code += "<div>" + data.html + "</div>";

244 	var OEmbedContentHandler = ContentHandlerManager.createHandler({
245 		/**
246 		 * Handle the pasting. Try to detect content pasted from word and transform to clean html
247 		 * @param content
248 		 */
249 		handleContent: function( content ) {
250 			if ( typeof content === 'string' ){
251 				content = jQuery( '<div>' + content + '</div>' );
252 			} else if ( content instanceof jQuery ) {
253 				content = jQuery( '<div>' ).append(content);
254 			}
255 
256 			this.replaceoEmbedContent( content );
257 
258 			return content.html();
259 		},
260 
261 		/**
262 		 * Check whether the content of the given jQuery object is assumed to be pasted from word.
263 		 * @param content
264 		 * @return true for content pasted from word, false for other content
265 		 */
266 		replaceoEmbedContent: function( content ) {
267 			// check every element which was pasted.
268 			content.contents().each(function() {
269 				var container = jQuery(this);
270 				container.oembed( container.text() );
271 			});
272 		}
273 	});
