 38 /**
 39  * init Delicious repository
 40  */
 41 GENTICS.Aloha.Repositories.delicious.init = function() {
 42 	var that = this;
 43 
 44 	// check weight
 45 	if ( this.settings.weight + 0.15 > 1 ) {
 46 		this.settings.weight = 1 - 0.15;
 47 	}
 48 
 49 	// default delicious URL. Returns most popular links.
 50 	this.deliciousURL = "http://feeds.delicious.com/v2/json/";
 51 
 52 	if ( this.settings.username ) {
 53 
 54 		// if a username is set use public user links
 55 		this.deliciousURL += this.settings.username + '/';
 56 
 57 		// set the repository name
 58 		this.repositoryName = 'deliciuos/' + this.settings.username;
 59 
 60 		// when a user is specified get his tags and store it local
 61 		this.tags = [];
 62 
 63 		jQuery.ajax({ type: "GET",
 64 			dataType: "jsonp",
 65 			url: 'http://feeds.delicious.com/v2/json/tags/'+that.settings.username,
 66 			success: function(data) {
 67 				// convert data
 68 				for (var tag in data) {
 69 					that.tags.push(tag);
 70 				}
 71 			}
 72 		});
 73 	} else {
 74 		// set the repository name
 75 		this.repositoryName = 'deliciuos/' + popular;
 76 
 77 		this.deliciousURL += 'tag/';
 78 	}
 79 };

 82 /**
 83  * Searches a repository for items matching query if objectTypeFilter.
 84  * If none found it returns null.
 85  */
 86 GENTICS.Aloha.Repositories.delicious.query = function( p, callback) {
 87 	var that = this;
 88 
 89 	if ( p.objectTypeFilter && jQuery.inArray('website', p.objectTypeFilter) == -1) {
 90 
 91 		// return if no website type is requested
 92 		callback.call( this, []);
 93 
 94 	} else {
 95 
 96 		// prepare tags
 97 		var tags = [];
 98 		if ( this.settings.username ) {
 99 
100 			// search in user tags
101 			var queryTags = p.queryString ? p.queryString.split(' ') : [];
102 		    for (var i = 0; i < queryTags.length; i++) {
103 				var queryTag = queryTags[i].trim();
104 				if ( jQuery.inArray(queryTag, that.tags) == -1 ) {
105 					var newtags = that.tags.filter(function(e, i, a) {
106 						var r = new RegExp(queryTag, 'i');
107 						return ( e.match(r) );
108 					});
109 					if ( newtags.length > 0 ) {
110 						tags.push(newtags[0]);
111 					}
112 				} else {
113 					tags.push(queryTag);
114 				}
115 			}
116 
117 		} else {
118 
119 			// handle each word as tag
120 			tags = p.queryString.split(' ');
121 
122 		}
123 
124 		// search in tree
125 		var folderTags = p.inFolderId ? p.inFolderId.split('+') : [];
126 		jQuery.extend(tags, folderTags);
127 
128 		// if we have a query and no tag matching return
129 		if ( p.queryString && tags.length == 0 ) {
130 			callback.call( that, []);
131 			return;
132 		}
133 
134 		jQuery.ajax({ type: "GET",
135 			dataType: "jsonp",
136 			url: that.deliciousURL + tags.join('+'),
137 			success: function(data) {
138 				var items = [];
139 				// convert data to Aloha objects
140 				for (var i = 0; i < data.length; i++) {
141 					if (typeof data[i] != 'function' ) {
142 						items.push(new GENTICS.Aloha.Repository.Document ({
143 							id: data[i].u,
144 							name: data[i].d,
145 							repositoryId: that.repositoryId,
146 							type: 'website',
147 							url: data[i].u,
148 							weight: that.settings.weight + (15-1)/100
149 						}));
150 					}
151 			    }
152 				callback.call( that, items);
153 			}
154 		});
155 	}
156 };

158 /**
159  * Returns all tags for username in a tree style way
160  */
161 GENTICS.Aloha.Repositories.delicious.getChildren = function( p, callback) {
162 	var that = this;
163 
164 	// tags are only available when a username is available
165 	if ( this.settings.username ) {
166 
167 		// return all tags
168 		var items = [];
169 		if ( p.inFolderId == this.repositoryId ) {
170 
171 			for (var i = 0; i < this.tags.length; i++) {
172 				if (typeof this.tags[i] != 'function' ) {
173 					items.push(new GENTICS.Aloha.Repository.Folder ({
174 						id: this.tags[i],
175 						name: this.tags[i],
176 						repositoryId: this.repositoryId,
177 						type: 'tag',
178 						url: 'http://feeds.delicious.com/v2/rss/tags/'+that.settings.username+'/'+this.tags[i]
179 					}));
180 				}
181 		    }
182 			callback.call( this, items);
183 
184 		} else {
185 			jQuery.ajax({ type: "GET",
186 				dataType: "jsonp",
187 				url: 'http://feeds.delicious.com/v2/json/tags/'+that.settings.username+'/'+p.inFolderId,
188 				success: function(data) {
189 					var items = [];
190 					// convert data
191 					for (var tag in data) {
192 						// the id is tag[+tag+...+tag]
193 						var id = (p.inFolderId)?p.inFolderId + '+' + tag:tag;
194 						if (typeof data[tag] != 'function' ) {
195 							items.push(new GENTICS.Aloha.Repository.Folder({
196 197 								id: id,
198 								name: tag,
199 								repositoryId: that.repositoryId,
200 								type: 'tag',
201 								url: 'http://feeds.delicious.com/v2/rss/tags/'+that.settings.username+'/'+id,
202 								hasMoreItems: true
203 							}));
204 						}
205 					}
206 					callback.call( that, items);
207 				}
208 			});
209 
210 		}
211 	} else {
212 		callback.call( this, []);
213 	}
214 };

216 /**
217  * Get the repositoryItem with given id
218  * @param itemId {String} id of the repository item to fetch
219  * @param callback {function} callback function
220  * @return {GENTICS.Aloha.Repository.Object} item with given id
221  */
222 GENTICS.Aloha.Repositories.delicious.getObjectById = function ( itemId, callback ) {
223 	var that = this;
224 
225 	jQuery.ajax({
226 		type: 'GET',
227 		dataType: "jsonp",
228 		url: 'http://feeds.delicious.com/v2/json/urlinfo/' + jQuery.md5(itemId),
229 		success: function (data) {
230 			var items = [];
231 			// convert data to Aloha objects
232 			for (var i = 0; i < data.length; i++) {
233 				if (typeof data[i] != 'function' ) {
234 					items.push(new GENTICS.Aloha.Repository.Document ({
235 						id: itemId,
236 						name: data[i].title,
237 						repositoryId: that.repositoryId,
238 						type: 'website',
239 						url: itemId,
240 						weight: that.settings.weight + (15-1)/100
241 					}));
242 				}
243 		    }
244 			callback.call( that, items);
245 		}
246 	});
247 };
