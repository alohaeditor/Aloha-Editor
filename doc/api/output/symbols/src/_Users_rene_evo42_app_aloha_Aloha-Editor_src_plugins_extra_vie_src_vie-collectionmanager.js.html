1 if (typeof VIE === 'undefined') {
  2     VIE = {};

  5 VIE.CollectionManager = {
  6     collections: [],
  7 
  8     loadCollections: function() {
  9         jQuery('[typeof="http://purl.org/dc/dcmitype/Collection"]').each(function() {
 10             VIE.CollectionManager.getCollectionForContainer(this);
 11         });
 12         return VIE.CollectionManager.collections;
 13     },
 14 
 15     getCollectionForContainer: function(element) {
 16         element = jQuery(element);
 17 
 18         var firstChild = element.children(':first-child');
 19         if (firstChild === undefined) {
 20             return null;
 21         }
 22 
 23         var preparedNewElement = VIE.ContainerManager.cloneContainer(firstChild);
 24 
 25         var Collection = Backbone.Collection.extend({
 26             model: VIE.ContainerManager.getModelForContainer(preparedNewElement)
 27         });
 28 
 29         var collectionInstance = new Collection({});
 30 
 31         collectionInstance.view = VIE.CollectionManager._getViewForCollection(preparedNewElement, element, collectionInstance);
 32 
 33         VIE.CollectionManager.collections.push(collectionInstance);
 34 
 35         return collectionInstance;
 36     },
 37 
 38     /**
 39      * @private
 40      */
 41     _getViewForCollection: function(element, collectionElement, collectionInstance) {
 42         var collectionView = Backbone.View.extend({
 43             collection: collectionInstance,
 44             el: collectionElement,
 45 
 46             initialize: function() {
 47                 _.bindAll(this, 'addItem', 'removeItem');
 48                 this.collection.bind('add', this.addItem);
 49                 this.collection.bind('remove', this.removeItem);
 50             },
 51 
 52             addItem: function(itemInstance) {
 53                 itemInstance = VIE.ContainerManager.registerInstance(itemInstance, VIE.ContainerManager.cloneContainer(element));
 54                 var itemViewElement = itemInstance.views[0].render().el;
 55                 this.el.append(itemViewElement);
 56                 itemViewElement.show();
 57             },
 58 
 59             removeItem: function(itemInstance) {
 60                 if (typeof itemInstance.view === 'undefined') {
 61                     return;
 62                 }
 63                 itemInstance.view.el.hide();
 64             }
 65         });
 66 
 67         return new collectionView({});
 68     }
 69 };
