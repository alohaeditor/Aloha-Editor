130 	/**
131 	 * Method to (re-)calculate the common ancestor container and to get it.
132 	 * The common ancestor container is the DOM Object which encloses the
133 	 * whole range and is nearest to the start and end container objects.
134 	 * @return {DOMObject} get the common ancestor container
135 	 * @method
136 	 */
137 	getCommonAncestorContainer: function() {
138 		if (this.commonAncestorContainer) {
139 			// sometimes it's cached (or was set)
140 			return this.commonAncestorContainer;
141 		}
142 
143 		// if it's not cached, calculate and then cache it
144 		this.updateCommonAncestorContainer();
145 
146 		// now return it anyway
147 		return this.commonAncestorContainer;

150 	/**
151 	 * Get the parent elements of the startContainer/endContainer up to the given limit. When the startContainer/endContainer
152 	 * is no text element, but a node, the node itself is returned as first element.
153 	 * @param {jQuery} limit limit object (default: body)
154 	 * @param {boolean} fromStart true to fetch the parents from the startContainer, false for the endContainer
155 	 * @return {jQuery} parent elements of the startContainer/endContainer as jQuery objects
156 	 * @method
157 	 */
158 	getContainerParents: function (limit, fromEnd) {
159 		// TODO cache the calculated parents
160 		var
161 			container = fromEnd ? this.endContainer : this.startContainer,
162 			parents, limitIndex,
163 			i;
164 
165 		if (!container) {
166 			return false;
167 		}
168 
169 		if ( typeof limit === 'undefined' || ! limit ) {
170 			limit = jQuery('body');
171 		}
172 
173 		
174 		if (container.nodeType == 3) {
175 			parents = jQuery(container).parents();
176 		} else {
177 			parents = jQuery(container).parents();
178 			for (i = parents.length; i > 0; --i) {
179 				parents[i] = parents[i - 1];
180 			}
181 			parents[0] = container;
182 		}
183 
184 		// now slice this array
185 		limitIndex = parents.index(limit);
186 
187 		if (limitIndex >= 0) {
188 			parents = parents.slice(0, limitIndex);
189 		}
190 
191 		return parents;
192 	},

216 	/**
217 	 * TODO: the commonAncestorContainer is not calculated correctly, if either the start or
218 	 * the endContainer would be the cac itself (e.g. when the startContainer is a textNode
219 	 * and the endContainer is the startContainer's parent <p>). in this case the cac will be set
220 	 * to the parent div
221 	 * Method to update a range object internally
222 	 * @param commonAncestorContainer (DOM Object); optional Parameter; if set, the parameter
223 	 * will be used instead of the automatically calculated CAC
224 	 * @return void
225 	 * @hide
226 	 */
227 	updateCommonAncestorContainer: function(commonAncestorContainer) {
228 		// this will be needed either right now for finding the CAC or later for the crossing index
229 		var parentsStartContainer = this.getStartContainerParents(),
230 			parentsEndContainer = this.getEndContainerParents(),
231 			i;
232 
233 		// if no parameter was passed, calculate it
234 		if (!commonAncestorContainer) {
235 			// find the crossing between startContainer and endContainer parents (=commonAncestorContainer)
236 			if (!(parentsStartContainer.length > 0 && parentsEndContainer.length > 0)) {
237 				console.warn('could not find commonAncestorContainer');
238 				return false;
239 			}
240 
241 			for (i = 0; i < parentsStartContainer.length; i++) {
242 				if (parentsEndContainer.index( parentsStartContainer[ i ] ) != -1) {
243 					this.commonAncestorContainer = parentsStartContainer[ i ];
244 					break;
245 				}
246 			}
247 		} else {
248 			this.commonAncestorContainer = commonAncestorContainer;
249 		}
250 
251 		// if everything went well, return true :-)
252 		console.debug(commonAncestorContainer? 'commonAncestorContainer was set successfully' : 'commonAncestorContainer was calculated successfully');
253 		return true;
254 	},

256 	/**
257 	 * Helper function for selection in IE. Creates a collapsed text range at the given position
258 	 * @param container container
259 	 * @param offset offset
260 	 * @return collapsed text range at that position
261 	 * @hide
262 	 */
263 	getCollapsedIERange: function(container, offset) {
264 		// create a text range
265 		var
266 			ieRange = document.body.createTextRange(),
267 			tmpRange, right, parent, left;
268 
269 		// search to the left for the next element
270 		left = this.searchElementToLeft(container, offset);
271 		if (left.element) {
272 			// found an element, set the start to the end of that element
273 			tmpRange = document.body.createTextRange();
274 			tmpRange.moveToElementText(left.element);
275 			ieRange.setEndPoint('StartToEnd', tmpRange);
276 
277 			// and correct the start
278 			if (left.characters !== 0) {
279 				ieRange.moveStart('character', left.characters);
280 			} else {
281 				// this is a hack, when we are at the start of a text node, move the range anyway
282 				ieRange.moveStart('character', 1);
283 				ieRange.moveStart('character', -1);
284 			}
285 		} else {
286 			// found nothing to the left, so search right
287 			right = this.searchElementToRight(container, offset);
288 			if (false && right.element) {
289 				// found an element, set the start to the start of that element
290 				tmpRange = document.body.createTextRange();
291 				tmpRange.moveToElementText(right.element);
292 				ieRange.setEndPoint('StartToStart', tmpRange);
293 
294 				// and correct the start
295 				if (right.characters !== 0) {
296 					ieRange.moveStart('character', -right.characters);
297 				} else {
298 					ieRange.moveStart('character', -1);
299 					ieRange.moveStart('character', 1);
300 				}
301 			} else {
302 				// also found no element to the right, use the container itself
303 				parent = container.nodeType == 3 ? container.parentNode : container;
304 				tmpRange = document.body.createTextRange();
305 				tmpRange.moveToElementText(parent);
306 307 				ieRange.setEndPoint('StartToStart', tmpRange);
308 
309 				// and correct the start
310 				if (left.characters !== 0) {
311 					ieRange.moveStart('character', left.characters);
312 				}
313 			}
314 		}
315 		ieRange.collapse();
316 
317 		return ieRange;
318 	},

322 	 * If the selection is collapsed, this will result in a blinking cursor,

341 	/**
342 	 * Starting at the given position, search for the next element to the left and count the number of characters are in between
343 	 * @param container container of the startpoint
344 	 * @param offset offset of the startpoint in the container
345 	 * @return object with 'element' (null if no element found) and 'characters'
346 	 * @hide
347 	 */
348 	searchElementToLeft: function (container, offset) {
349 		var
350 			checkElement,
351 			characters = 0;
352 
353 		if (container.nodeType === 3) {
354 			// start is in a text node
355 			characters = offset;
356 			// begin check at the element to the left (if any)
357 			checkElement = container.previousSibling;
358 		} else {
359 			// start is between nodes, begin check at the element to the left (if any)
360 			if (offset > 0) {
361 				checkElement = container.childNodes[offset - 1];
362 			}
363 		}
364 
365 		// move to the right until we find an element
366 		while (checkElement && checkElement.nodeType === 3) {
367 			characters += checkElement.data.length;
368 			checkElement = checkElement.previousSibling;
369 		}
370 
371 		return {'element' : checkElement, 'characters' : characters};
372 	},
373 
374 	/**
375 	 * Starting at the given position, search for the next element to the right and count the number of characters that are in between
376 	 * @param container container of the startpoint
377 	 * @param offset offset of the startpoint in the container
378 	 * @return object with 'element' (null if no element found) and 'characters'
379 	 * @hide
380 	 */
381 	searchElementToRight: function (container, offset) {
382 		var
383 			checkElement,
384 			characters = 0;
385 
386 		if (container.nodeType === 3) {
387 			// start is in a text node
388 			characters = container.data.length - offset;
389 
390 			// begin check at the element to the right (if any)
391 			checkElement = container.nextSibling;
392 		} else {
393 			// start is between nodes, begin check at the element to the right (if any)
394 			if (offset < container.childNodes.length) {
395 				checkElement = container.childNodes[offset];
396 			}
397 		}
398 
399 		// move to the right until we find an element
400 		while (checkElement && checkElement.nodeType === 3) {
401 			characters += checkElement.data.length;
402 			checkElement = checkElement.nextSibling;
403 		}
404 
405 		return {'element' : checkElement, 'characters' : characters};
406 	},

420 	/**
421 	 * Initialize the current range object from the user selection of the browser.
422 	 * @param event which calls the method
423 	 * @return void
424 	 * @hide
425 	 */
426 	initializeFromUserSelection: function(event) {
427 		var
428 			selection = rangy.getSelection(),
429 			browserRange;
430 
431 		if (!selection) {
432 			return false;
433 		}
434 
435 		// check if a ragne exists
436 		if ( !selection.rangeCount ) {
437 			return false;
438 		}
439 
440 		// getBrowserRange
441 		browserRange = selection.getRangeAt(0);
442 		if (!browserRange) {
443 			return false;
444 		}
445 
446 		// initially set the range to what the browser tells us
447 		this.startContainer = browserRange.startContainer;
448 		this.endContainer = browserRange.endContainer;
449 		this.startOffset = browserRange.startOffset;
450 		this.endOffset = browserRange.endOffset;
451 
452 		// now try to correct the range
453 		this.correctRange();
454 		return;
455 	},
456 
457 	/**
458 	 * Correct the current range. The general goal of the algorithm is to have start
459 	 * and end of the range in text nodes if possible and the end of the range never
460 	 * at the beginning of an element or text node. Details of the algorithm can be
461 	 * found in the code comments
462 	 * @method
463 	 */
464 	correctRange: function() {
465 		var
466 			adjacentTextNode,
467 			textNode,
468 			checkedElement,
469 			parentNode,
470 			offset;
471 
472 		this.clearCaches();
473 		if (this.isCollapsed()) {
474 			// collapsed ranges are treated specially
475 
476 			// first check if the range is not in a text node
477 			if (this.startContainer.nodeType === 1) {
478 				if (this.startOffset > 0 && this.startContainer.childNodes[this.startOffset - 1].nodeType === 3) {
479 					// when the range is between nodes (container is an element
480 					// node) and there is a text node to the left -> move into this text
481 					// node (at the end)
482 					this.startContainer = this.startContainer.childNodes[this.startOffset - 1];
483 					this.startOffset = this.startContainer.data.length;
484 					this.endContainer = this.startContainer;
485 					this.endOffset = this.startOffset;
486 					return;
487 				}
488 
489 				if (this.startOffset > 0 && this.startContainer.childNodes[this.startOffset - 1].nodeType === 1) {
490 					// search for the next text node to the left
491 					adjacentTextNode = GENTICS.Utils.Dom.searchAdjacentTextNode(this.startContainer, this.startOffset, true);
492 					if (adjacentTextNode) {
493 						this.startContainer = this.endContainer = adjacentTextNode;
494 						this.startOffset = this.endOffset = adjacentTextNode.data.length;
495 						return;
496 					}
497 					// search for the next text node to the right
498 					adjacentTextNode = GENTICS.Utils.Dom.searchAdjacentTextNode(this.startContainer, this.startOffset, false);
499 					if (adjacentTextNode) {
500 						this.startContainer = this.endContainer = adjacentTextNode;
501 						this.startOffset = this.endOffset = 0;
502 						return;
503 					}
504 				}
505 
506 				if (this.startOffset < this.startContainer.childNodes.length && this.startContainer.childNodes[this.startOffset].nodeType === 3) {
507 					// when the range is between nodes and there is a text node
508 					// to the right -> move into this text node (at the start)
509 					this.startContainer = this.startContainer.childNodes[this.startOffset];
510 511 					this.startOffset = 0;
512 					this.endContainer = this.startContainer;
513 					this.endOffset = 0;
514 					return;
515 				}
516 			}
517 
518 			// when the selection is in a text node at the start, look for an adjacent text node and if one found, move into that at the end
519 			if (this.startContainer.nodeType === 3 && this.startOffset === 0) {
520 				adjacentTextNode = GENTICS.Utils.Dom.searchAdjacentTextNode(this.startContainer.parentNode, GENTICS.Utils.Dom.getIndexInParent(this.startContainer), true);
521 				if (adjacentTextNode) {
522 					this.startContainer = this.endContainer = adjacentTextNode;
523 					this.startOffset = this.endOffset = adjacentTextNode.data.length;
524 				}
525 			}
526 		} else {
527 			// expanded range found
528 
529 			// correct the start, but only if between nodes
530 			if (this.startContainer.nodeType === 1) {
531 				// if there is a text node to the right, move into this
532 				if (this.startOffset < this.startContainer.childNodes.length && this.startContainer.childNodes[this.startOffset].nodeType === 3) {
533 					this.startContainer = this.startContainer.childNodes[this.startOffset];
534 					this.startOffset = 0;
535 				} else if (this.startOffset < this.startContainer.childNodes.length && this.startContainer.childNodes[this.startOffset].nodeType === 1) {
536 					// there is an element node to the right, so recursively check all first child nodes until we find a text node
537 					textNode = false;
538 					checkedElement = this.startContainer.childNodes[this.startOffset];
539 					while (textNode === false && checkedElement.childNodes && checkedElement.childNodes.length > 0) {
540 						// go to the first child of the checked element
541 						checkedElement = checkedElement.childNodes[0];
542 						// when this element is a text node, we are done
543 						if (checkedElement.nodeType === 3) {
544 							textNode = checkedElement;
545 						}
546 					}
547 
548 					// found a text node, so move into it
549 					if (textNode !== false) {
550 						this.startContainer = textNode;
551 						this.startOffset = 0;
552 					}
553 				}
554 			}
555 
556 			// check whether the start is inside a text node at the end
557 			if (this.startContainer.nodeType === 3 && this.startOffset === this.startContainer.data.length) {
558 				// check whether there is an adjacent text node to the right and if
559 				// yes, move into it
560 				adjacentTextNode = GENTICS.Utils.Dom
561 						.searchAdjacentTextNode(this.startContainer.parentNode, GENTICS.Utils.Dom
562 						.getIndexInParent(this.startContainer) + 1, false);
563 				if (adjacentTextNode) {
564 					this.startContainer = adjacentTextNode;
565 					this.startOffset = 0;
566 				}
567 			}
568 
569 			// now correct the end
570 			if (this.endContainer.nodeType === 3 && this.endOffset === 0) {
571 				// we are in a text node at the start
572 				if (this.endContainer.previousSibling && this.endContainer.previousSibling.nodeType === 3) {
573 					// found a text node to the left -> move into it (at the end)
574 					this.endContainer = this.endContainer.previousSibling;
575 					this.endOffset = this.endContainer.data.length;
576 				} else if (this.endContainer.previousSibling && this.endContainer.previousSibling.nodeType === 1 && this.endContainer.parentNode) {
577 					// found an element node to the left -> move in between
578 					parentNode = this.endContainer.parentNode;
579 					for (offset = 0; offset < parentNode.childNodes.length; ++offset) {
580 						if (parentNode.childNodes[offset] == this.endContainer) {
581 							this.endOffset = offset;
582 							break;
583 						}
584 					}
585 					this.endContainer = parentNode;
586 				}
587 			}
588 
589 			if (this.endContainer.nodeType == 1 && this.endOffset === 0) {
590 				// we are in an element node at the start, possibly move to the previous sibling at the end
591 				if (this.endContainer.previousSibling) {
592 					if (this.endContainer.previousSibling.nodeType === 3) {
593 						// previous sibling is a text node, move end into here (at the end)
594 						this.endContainer = this.endContainer.previousSibling;
595 						this.endOffset = this.endContainer.data.length;
596 					} else if (
597 							this.endContainer.previousSibling.nodeType === 1
598 							&& this.endContainer.previousSibling.childNodes
599 							&& this.endContainer.previousSibling.childNodes.length > 0) {
600 						// previous sibling is another element node with children,
601 						// move end into here (at the end)
602 						this.endContainer = this.endContainer.previousSibling;
603 						this.endOffset = this.endContainer.childNodes.length;
604 					}
605 				}
606 			}
607 
608 			// correct the end, but only if between nodes
609 			if (this.endContainer.nodeType == 1) {
610 				// if there is a text node to the left, move into this
611 				if (this.endOffset > 0 && this.endContainer.childNodes[this.endOffset - 1].nodeType === 3) {
612 					this.endContainer = this.endContainer.childNodes[this.endOffset - 1];
613 					this.endOffset = this.endContainer.data.length;
614 				} else if (this.endOffset > 0 && this.endContainer.childNodes[this.endOffset - 1].nodeType === 1) {
615 					// there is an element node to the left, so recursively check all last child nodes until we find a text node
616 					textNode = false;
617 					checkedElement = this.endContainer.childNodes[this.endOffset - 1];
618 					while (textNode === false && checkedElement.childNodes && checkedElement.childNodes.length > 0) {
619 						// go to the last child of the checked element
620 						checkedElement = checkedElement.childNodes[checkedElement.childNodes.length - 1];
621 						// when this element is a text node, we are done
622 						if (checkedElement.nodeType === 3) {
623 							textNode = checkedElement;
624 						}
625 					}
626 
627 					// found a text node, so move into it
628 					if (textNode !== false) {
629 						this.endContainer = textNode;
630 						this.endOffset = this.endContainer.data.length;
631 					}
632 				}
633 			}
634 		}
635 	},
636 
637 	/**
638 	 * Clear the caches for this range. This method must be called when the range itself (start-/endContainer or start-/endOffset) is modified.
639 	 * @method
640 	 */
641 	clearCaches: function () {
642 		this.commonAncestorContainer = undefined;
643 	},
644 
645 	/**
646 	 * Get the range tree of this range.
647 	 * The range tree will be cached for every root object. When the range itself is modified, the cache should be cleared by calling GENTICS.Utils.RangeObject.clearCaches
648 	 * @param {DOMObject} root root object of the range tree, if non given, the common ancestor container of the start and end containers will be used
649 	 * @return {RangeTree} array of RangeTree object for the given root object
650 	 * @method
651 	 */
652 	getRangeTree: function (root) {
653 		// TODO cache rangeTrees
654 		if (typeof root === 'undefined') {
655 			root = this.getCommonAncestorContainer();
656 		}
657 
658 		this.inselection = false;
659 		return this.recursiveGetRangeTree(root);
660 	},
661 
662 	/**
663 	 * Recursive inner function for generating the range tree.
664 	 * @param currentObject current DOM object for which the range tree shall be generated
665 	 * @return array of Tree objects for the children of the current DOM object
666 	 * @hide
667 	 */
668 	recursiveGetRangeTree: function (currentObject) {
669 		// get all direct children of the given object
670 		var jQueryCurrentObject = jQuery(currentObject),
671 			childCount = 0,
672 			that = this,
673 			currentElements = [];
674 
675 		jQueryCurrentObject.contents().each(function(index) {
676 			var type = 'none',
677 				startOffset = false,
678 				endOffset = false,
679 				collapsedFound = false,
680 				noneFound = false,
681 				partialFound = false,
682 				fullFound = false,
683 				i;
684 
685 			// check for collapsed selections between nodes
686 			if (that.isCollapsed() && currentObject === that.startContainer && that.startOffset === index) {
687 				// insert an extra rangetree object for the collapsed range here
688 				currentElements[childCount] = new GENTICS.Utils.RangeTree();
689 				currentElements[childCount].type = 'collapsed';
690 				currentElements[childCount].domobj = undefined;
691 				that.inselection = false;
692 				collapsedFound = true;
693 				childCount++;
694 			}
695 
696 			if (!that.inselection && !collapsedFound) {
697 				// the start of the selection was not yet found, so look for it now
698 				// check whether the start of the selection is found here
699 
700 				// check is dependent on the node type
701 				switch(this.nodeType) {
702 				case 3: // text node
703 					if (this === that.startContainer) {
704 						// the selection starts here
705 						that.inselection = true;
706 
707 						// when the startoffset is > 0, the selection type is only partial
708 						type = that.startOffset > 0 ? 'partial' : 'full';
709 						startOffset = that.startOffset;
710 						endOffset = this.length;
711 712 					}
713 					break;
714 				case 1: // element node
715 					if (this === that.startContainer && that.startOffset === 0) {
716 						// the selection starts here
717 						that.inselection = true;
718 						type = 'full';
719 					}
720 					if (currentObject === that.startContainer && that.startOffset === index) {
721 						// the selection starts here
722 						that.inselection = true;
723 						type = 'full';
724 					}
725 					break;
726 				}
727 			}
728 
729 			if (that.inselection && !collapsedFound) {
730 				if (type == 'none') {
731 					type = 'full';
732 				}
733 				// we already found the start of the selection, so look for the end of the selection now
734 				// check whether the end of the selection is found here
735 
736 				switch(this.nodeType) {
737 				case 3: // text node
738 					if (this === that.endContainer) {
739 						// the selection ends here
740 						that.inselection = false;
741 
742 						// check for partial selection here
743 						if (that.endOffset < this.length) {
744 							type = 'partial';
745 						}
746 						if (startOffset === false) {
747 							startOffset = 0;
748 						}
749 						endOffset = that.endOffset;
750 					}
751 					break;
752 				case 1: // element node
753 					if (this === that.endContainer && that.endOffset === 0) {
754 						that.inselection = false;
755 					}
756 					break;
757 				}
758 				if (currentObject === that.endContainer && that.endOffset <= index) {
759 					that.inselection = false;
760 					type = 'none';
761 				}
762 			}
763 
764 			// create the current selection tree entry
765 			currentElements[childCount] = new GENTICS.Utils.RangeTree();
766 			currentElements[childCount].domobj = this;
767 			currentElements[childCount].type = type;
768 			if (type == 'partial') {
769 				currentElements[childCount].startOffset = startOffset;
770 				currentElements[childCount].endOffset = endOffset;
771 			}
772 
773 			// now do the recursion step into the current object
774 			currentElements[childCount].children = that.recursiveGetRangeTree(this);
775 
776 			// check whether a selection was found within the children
777 			if (currentElements[childCount].children.length > 0) {
778 				for ( i = 0; i < currentElements[childCount].children.length; ++i) {
779 					switch(currentElements[childCount].children[i].type) {
780 					case 'none':
781 						noneFound = true;
782 						break;
783 					case 'full':
784 						fullFound = true;
785 						break;
786 					case 'partial':
787 						partialFound = true;
788 						break;
789 					}
790 				}
791 
792 				if (partialFound || (fullFound && noneFound)) {
793 					// found at least one 'partial' DOM object in the children, or both 'full' and 'none', so this element is also 'partial' contained
794 					currentElements[childCount].type = 'partial';
795 				} else if (fullFound && !partialFound && !noneFound) {
796 					// only found 'full' contained children, so this element is also 'full' contained
797 					currentElements[childCount].type = 'full';
798 				}
799 			}
800 
801 			childCount++;
802 		});
803 
804 		// extra check for collapsed selections at the end of the current element
805 		if (this.isCollapsed()
806 				&& currentObject === this.startContainer
807 				&& this.startOffset == currentObject.childNodes.length) {
808 			currentElements[childCount] = new GENTICS.Utils.RangeTree();
809 			currentElements[childCount].type = 'collapsed';
810 			currentElements[childCount].domobj = undefined;
811 		}
812 
813 		return currentElements;
814 	},
815 
816 	/**
817 	 * Find certain the first occurrence of some markup within the parents of either the start or the end of this range.
818 	 * The markup can be identified by means of a given comparator function. The function will be passed every parent (up to the eventually given limit object, which itself is not considered) to the comparator function as this.
819 	 * When the comparator function returns boolean true, the markup found and finally returned from this function as dom object.<br/>
820 	 * Example for finding an anchor tag at the start of the range up to the active editable object:<br/>
821 	 * <pre>
822 	 * range.findMarkup(
823 	 *   function() {
824 	 *     return this.nodeName.toLowerCase() == 'a';
825 	 *   },
826 	 *   jQuery(Aloha.activeEditable.obj)
827 	 * );
828 	 * </pre>
829 	 * @param {function} comparator comparator function to find certain markup
830 	 * @param {jQuery} limit limit objects for limit the parents taken into consideration
831 	 * @param {boolean} atEnd true for searching at the end of the range, false for the start (default: false)
832 	 * @return {DOMObject} the found dom object or false if nothing found.
833 	 * @method
834 	 */
835 	findMarkup: function (comparator, limit, atEnd) {
836 		var parents = this.getContainerParents(limit, atEnd),
837 			returnValue = false;
838 		jQuery.each(parents, function (index, domObj) {
839 			if (comparator.apply(domObj)) {
840 				returnValue = domObj;
841 				return false;
842 			}
843 		});
844 
845 		return returnValue;
846 	},
847 
848 	/**
849 	 * Get the text enclosed by this range
850 	 * @return {String} the text of the range
851 	 * @method
852 	 */
853 	getText: function() {
854 		if (this.isCollapsed()) {
855 			return '';
856 		} else {
857 			return this.recursiveGetText(this.getRangeTree());
858 		}
859 	},
860 
861 	recursiveGetText: function (tree) {
862 		if (!tree) {
863 			return '';
864 		} else {
865 			var that = this,
866 				text = '';
867 			jQuery.each(tree, function() {
868 				if (this.type == 'full') {
869 					// fully selected element/text node
870 					text += jQuery(this.domobj).text();
871 				} else if (this.type == 'partial' && this.domobj.nodeType === 3) {
872 					// partially selected text node
873 					text += jQuery(this.domobj).text().substring(this.startOffset, this.endOffset);
874 				} else if (this.type == 'partial' && this.domobj.nodeType === 1 && this.children) {
875 					// partially selected element node
876 					text += that.recursiveGetText(this.children);
877 				}
878 			});
879 			return text;
880 		}
881 	}
