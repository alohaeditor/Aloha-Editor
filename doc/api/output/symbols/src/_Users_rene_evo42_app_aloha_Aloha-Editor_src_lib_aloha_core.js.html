114 		/**
115 		 * Initialize the initialization process
116 		 */
117 		init: function () {
118 				
119 			// merge defaults and settings and provide all in settings
120 			Aloha.settings = jQuery.extendObjects( true, {}, Aloha.defaults, Aloha.settings );
121 
122 			// initialize rangy. This is probably necessary here,
123 			// because due to the current loading mechanism, rangy
124 			// doesn't initialize itself in all browsers
125 			if (window.rangy) {
126 				window.rangy.init();
127 			}
128 			
129 			// Load & Initialise
130 			Aloha.stage = 'loadPlugins';
131 			Aloha.loadPlugins(function(){
132 				Aloha.stage = 'initAloha';
133 				Aloha.initAloha(function(){
134 					Aloha.stage = 'initPlugins';
135 					Aloha.initPlugins(function(){
136 						Aloha.stage = 'initGui';
137 						Aloha.initGui(function(){
138 							Aloha.stage = 'alohaReady';
139 							Aloha.trigger('aloha-ready');
140 						});
141 					});
142 				});
143 			});
144 		},

146 		/**
147 		 * Load Plugins
148 		 */
149 		loadPlugins: function (next) {
150 			// contains an array like [common/format, common/block]
151 			var configuredPluginsWithBundle = this.getPluginsToBeLoaded();
152 
153 			if (configuredPluginsWithBundle.length) {
154 				var paths = {},
155 				    pluginNames = [],
156 				    requiredInitializers = [],
157 				    pathsToPlugins = {};
158 
159 				// Background: We do not use CommonJS packages for our Plugins
160 				// as this breaks the loading order when these modules have
161 				// other dependencies.
162 				// We "emulate" the commonjs modules with the path mapping.
163 				/* require(
164 				 *  { paths: {
165 				 *      'format': 'plugins/common/format/lib',
166 				 *      'format/nls': 'plugins/common/format/nls',
167 				 *      ... for every plugin ...
168 				 *    }
169 				 *  },
170 				 *  ['format/format-plugin', ... for every plugin ...],
171 				 *  next <-- when everything is loaded, we continue
172 173 				 */
174 				jQuery.each(configuredPluginsWithBundle, function (i, configuredPluginWithBundle) {
175 					var tmp, bundleName, pluginName, bundlePath = '';
176 
177 					tmp = configuredPluginWithBundle.split('/');
178 					bundleName = tmp[0];
179 					pluginName = tmp[1];
180 
181 					// TODO assertion if pluginName or bundleName NULL _-> ERROR!!
182 
183 					if (Aloha.settings.basePath) {
184 						bundlePath = Aloha.settings.basePath;
185 					}
186 
187 					if (Aloha.settings.bundles && Aloha.settings.bundles[bundleName]) {
188 						bundlePath += Aloha.settings.bundles[bundleName];
189 					} else {
190 						bundlePath += '../plugins/' + bundleName;
191 					}
192 
193 					pluginNames.push(pluginName);
194 					paths[pluginName] = bundlePath + '/' + pluginName + '/lib';
195 
196 					pathsToPlugins[pluginName] = bundlePath + '/' + pluginName;
197 
198 					// As the "nls" path lies NOT inside /lib/, but is a sibling to /lib/, we need
199 					// to register it explicitely. The same goes for the "css" folder.
200 					jQuery.each(['nls', 'css', 'vendor', 'res'], function() {
201 						paths[pluginName + '/' + this] = bundlePath + '/' + pluginName + '/' + this;
202 					});
203 
204 					requiredInitializers.push(pluginName + '/' + pluginName + '-plugin');
205 				});
206 
207 				this.loadedPlugins = pluginNames;
208 				this.requirePaths = paths;
209 				
210 				// Main Require.js loading call, which fetches all the plugins.
211 				require(
212 					{
213 						context: 'aloha',
214 						paths: paths,
215 						locale: this.settings.locale || this.defaults.locale || 'en'
216 					},
217 					requiredInitializers,
218 					next
219 				);
220 
221 				pluginPaths = pathsToPlugins;
222 			} else {
223 				next();
224 			}
225 		},
226 
227 		/**
228 		 * Fetches plugins the user wants to have loaded. Returns all plugins the user
229 		 * has specified with the data-plugins property as array, with the bundle
230 		 * name in front.
231 		 *
232 		 * @return array
233 		 * @internal
234 		 */
235 		getPluginsToBeLoaded: function() {
236 			// look for data-aloha-plugins attributes and load values
237 			var
238 				plugins = jQuery('[data-aloha-plugins]').data('aloha-plugins');
239 
240 			// Determine Plugins
241 			if ( typeof plugins === 'string' && plugins !== "") {
242 				return plugins.replace(/\s+/g, '').split(',');
243 			}
244 			// Return
245 			return [];
246 			
247 		},
248 
249 		/**
250 		 * Returns list of loaded plugins (without Bundle name)
251 		 *
252 		 * @return array
253 		 */
254 		getLoadedPlugins: function() {
255 			return this.loadedPlugins;
256 		},
257 
258 259 		/**
260 		 * Returns true if a certain plugin is loaded, false otherwise.
261 		 */
262 		isPluginLoaded: function(pluginName) {
263 			var found = false;
264 			jQuery.each(this.loadedPlugins, function() {
265 				if (pluginName.toString() === this.toString()) {
266 					found = true;
267 				}
268 			});
269 			return found;
270 		},
271 
272 		/**
273 		 * Initialise Aloha
274 		 */
275 		initAloha: function(next){
276 			// check browser version on init
277 			// this has to be revamped, as
278 			if (jQuery.browser.webkit && parseFloat(jQuery.browser.version) < 532.5 || // Chrome/Safari 4
279 				jQuery.browser.mozilla && parseFloat(jQuery.browser.version) < 1.9 || // FF 3.5
280 				jQuery.browser.msie && jQuery.browser.version < 7 || // IE 7
281 				jQuery.browser.opera && jQuery.browser.version < 11 ) { // right now, Opera needs some work
282 				if (window.console && window.console.log) {
283 					window.console.log( 'Your browser is not supported.' );
284 				}
285 			}
286 
287 			// register the body click event to blur editables
288 			jQuery('html').mousedown(function(e) {
289 				// if an Ext JS modal is visible, we don't want to loose the focus on
290 				// the editable as we assume that the user must have clicked somewhere
291 				// in the modal... where else could he click?
292 				// loosing the editable focus in this case hinders correct table
293 				// column/row deletion, as the table module will clean it's selection
294 				// as soon as the editable is deactivated. Fusubscriberthermore you'd have to
295 				// refocus the editable again, which is just strange UX
296 				if (Aloha.activeEditable && !Aloha.isMessageVisible() && !Aloha.eventHandled) {
297 					Aloha.activeEditable.blur();
298 					Aloha.activeEditable = null;
299 				}
300 			}).mouseup(function(e) {
301 				Aloha.eventHandled = false;
302 			});
303 			
304 			// Initialise the base path to the aloha files
305 			Aloha.settings.base = Aloha.getAlohaUrl();
306 
307 			// initialize the Log
308 			Aloha.Log.init();
309 
310 			// initialize the error handler for general javascript errors
311 			if ( Aloha.settings.errorhandling ) {
312 				window.onerror = function (msg, url, linenumber) {
313 					Aloha.Log.error(Aloha, 'Error message: ' + msg + '\nURL: ' + url + '\nLine Number: ' + linenumber);
314 					// TODO eventually add a message to the message line?
315 					return true;
316 				};
317 			}
318 
319 			// OS detection
320 			if (navigator.appVersion.indexOf('Win') != -1) {
321 				Aloha.OSName = 'Win';
322 			}
323 			if (navigator.appVersion.indexOf('Mac') != -1) {
324 				Aloha.OSName = 'Mac';
325 			}
326 			if (navigator.appVersion.indexOf('X11') != -1) {
327 				Aloha.OSName = 'Unix';
328 			}
329 			if (navigator.appVersion.indexOf('Linux') != -1) {
330 				Aloha.OSName = 'Linux';
331 			}
332 
333 			try {
334 				// this will disable browsers image resizing facilities
335 				// disable resize handles
336 				var supported;
337 				try {
338 					supported = document.queryCommandSupported( 'enableObjectResizing' );
339 				} catch ( e ) {
340 					supported = false;
341 					Aloha.Log.log( 'enableObjectResizing is not supported.' );
342 				}
343 				
344 				if ( supported ) {
345 					document.execCommand( 'enableObjectResizing', false, false);
346 					Aloha.Log.log( 'enableObjectResizing disabled.' );
347 				}
348 			} catch (e) {
349 				Aloha.Log.error( e, 'Could not disable enableObjectResizing' );
350 				// this is just for others, who will not support disabling enableObjectResizing
351 			}
352 			// Forward
353 			next();
354 		},
355 
356 		/**
357 		 * Loads plugins Aloha
358 		 * @return void
359 		 */
360 		initPlugins: function (next) {
361 			PluginManager.init(function(){
362 				next();
363 			}, this.getLoadedPlugins() );
364 		},

366 		/**
367 		 * Loads GUI components
368 		 * @return void
369 		 */
370 		initGui: function (next) {
371 			
372 			Aloha.RepositoryManager.init();
373 
374 			// activate registered editables
375 			for (var i = 0, editablesLength = Aloha.editables.length; i < editablesLength; i++) {
376 				if ( !Aloha.editables[i].ready ) {
377 					Aloha.editables[i].init();
378 				}
379 			}
380 
381 			// Forward
382 			next();
383 		},
384 
385 		/**
386 		 * Activates editable and deactivates all other Editables
387 		 * @param {Editable} editable the Editable to be activated
388 		 * @return void
389 		 */
390 		activateEditable: function (editable) {
391 
392 			// blur all editables, which are currently active
393 			for (var i = 0, editablesLength = Aloha.editables.length; i < editablesLength; i++) {
394 				if (Aloha.editables[i] != editable && Aloha.editables[i].isActive) {
395 					Aloha.editables[i].blur();
396 				}
397 			}
398 
399 			Aloha.activeEditable = editable;
400 		},
401

403 		 * Returns the current Editable
404 		 * @return {Editable} returns the active Editable
405 		 */
406 		getActiveEditable: function() {

408 		},

410 		/**
411 		 * deactivated the current Editable
412 		 * @return void
413 		 */
414 		deactivateEditable: function () {
415 
416 			if ( typeof Aloha.activeEditable === 'undefined' || Aloha.activeEditable === null ) {
417 				return;
418 			}
419 
420 			// blur the editable
421 			Aloha.activeEditable.blur();
422 			Aloha.activeEditable = null;
423 		},
424 
425 		/**
426 		 * Gets an editable by an ID or null if no Editable with that ID registered.
427 		 * @param {string} id the element id to look for.
428 		 * @return {Aloha.Editable} editable
429 		 */
430 		getEditableById: function (id) {
431 
432 			// if the element is a textarea than route to the editable div
433 			if (jQuery('#'+id).get(0).nodeName.toLowerCase() === 'textarea' ) {
434 				id = id + '-aloha';
435 			}
436 
437 			// serach all editables for id
438 			for (var i = 0, editablesLength = Aloha.editables.length; i < editablesLength; i++) {
439 				if (Aloha.editables[i].getId() == id) {
440 					return Aloha.editables[i];
441 				}
442 			}
443 
444 			return null;
445 		},
446 
447 		/**
448 		 * Checks wheater an object is a registered Aloha Editable.
449 		 * @param {jQuery} obj the jQuery object to be checked.
450 		 * @return {boolean}
451 		 */
452 		isEditable: function (obj) {
453 			for (var i=0, editablesLength = Aloha.editables.length; i < editablesLength; i++) {
454 				if ( Aloha.editables[i].originalObj.get(0) === obj ) {
455 					return true;
456 				}
457 			}
458 			return false;
459 		},
460

462 		 * Logs a message to the console
463 		 * @param level Level of the log ("error", "warn" or "info", "debug")
464 		 * @param component Component that calls the log
465 		 * @param message log message
466 		 * @return void
467 		 * @hide
468 		 */
469 		log: function(level, component, message) {
470 			if (typeof Aloha.Log !== "undefined")
471 				Aloha.Log.log(level, component, message);
472 		},

484 		/**
485 		 * Unregister the given editable. It will be deactivated and removed from editables.
486 		 * @param editable editable to unregister
487 		 * @return void
488 		 * @hide
489 		 */
490 		unregisterEditable: function (editable) {
491 
492 			// Find the index
493 			var id = Aloha.editables.indexOf( editable );
494 			// Remove it if really found!
495 			if (id != -1) {
496 				Aloha.editables.splice(id, 1);
497 			}
498 		},

502 		 * @hide

505 			return 'Aloha';
506 		},
507 
508 		/**
509 		 * Check whether at least one editable was modified
510 		 * @method
511 		 * @return {boolean} true when at least one editable was modified, false if not
512 		 */
513 		isModified: function () {
514 			// check if something needs top be saved
515 			for (var i = 0; i < Aloha.editables.length; i++) {
516 				if (Aloha.editables[i].isModified && Aloha.editables[i].isModified()) {
517 					return true;
518 				}
519 			}
520 
521 			return false;
522 		},
523 
524 		/**
525 		 * Determines the Aloha Url
526 		 * @method
527 		 * @return {String} alohaUrl
528 		 */
529 		getAlohaUrl: function( suffix ) {
530 			// aloha base path is defined by a script tag with 2 data attributes
531 			var requireJs = jQuery('[data-aloha-plugins]'),
532 				baseUrl = ( requireJs.length ) ? requireJs[0].src.replace( /\/?aloha.js$/ , '' ) : '';
533 				
534 			return baseUrl;
535 		},
536 
537 		/**
538 		 * Gets the Plugin Url
539 		 * @method
540 		 * @param {String} name
541 		 * @return {String} url
542 		 */
543 		getPluginUrl: function (name) {
544 			var url;
545 
546 			if (name) {
547 				url = pluginPaths[name];
548 				if(url) {
549 					//Check if url is absolute and attach base url if it is not
550 					if(!url.match("^(\/|http[s]?:).*")) {
551 						url = Aloha.getAlohaUrl() + '/' + url;
552 					}
553 				}
554 			}
555 
556 			return url;
557 		}
558 
559 	});

561 	return Aloha;
