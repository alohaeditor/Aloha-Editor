 53 	insertBreak: function () {
 54 		var range = Aloha.Selection.rangeObject,nonWSIndex,nextTextNode,newBreak;
 55 		if (!range.isCollapsed()) {
 56 			this.removeSelectedMarkup();
 57 		}
 58 
 59 		newBreak = jQuery('<br/>');
 60 		GENTICS.Utils.Dom.insertIntoDOM(newBreak, range, Aloha.activeEditable.obj);
 61 
 62 		nextTextNode = GENTICS.Utils.Dom.searchAdjacentTextNode(newBreak.parent().get(0), GENTICS.Utils.Dom.getIndexInParent(newBreak.get(0)) + 1, false);
 63 		if (nextTextNode) {
 64 			// trim leading whitespace
 65 			nonWSIndex = nextTextNode.data.search(/\S/);
 66 			if (nonWSIndex > 0) {
 67 				nextTextNode.data = nextTextNode.data.substring(nonWSIndex);
 68 			}
 69 		}
 70 
 71 		range.startContainer = range.endContainer = newBreak.get(0).parentNode;
 72 		range.startOffset = range.endOffset = GENTICS.Utils.Dom.getIndexInParent(newBreak.get(0)) + 1;
 73 		range.correctRange();
 74 		range.clearCaches();
 75 		range.select();
 76 	},

 78 	/**
 79 	 * first method to handle key strokes
 80 	 * @param event DOM event
 81 	 * @param rangeObject as provided by Aloha.Selection.getRangeObject();
 82 	 * @return "Aloha.Selection"
 83 	 */
 84 	preProcessKeyStrokes: function(event) {
 85 		if (event.type != 'keydown') {
 86 			return false;
 87 		}
 88 
 89 		var rangeObject = Aloha.Selection.rangeObject,
 90 			handlers,
 91 			i;
 92 
 93 		if (this.keyHandlers[event.keyCode]) {
 94 			handlers = this.keyHandlers[event.keyCode];
 95 			for ( i = 0; i < handlers.length; ++i) {
 96 				if (!handlers[i](event)) {
 97 					return false;
 98 				}
 99 			}
100 		}
101 
102 		// handle left (37) and right (39) keys for block detection
103 		if (event.keyCode === 37 || event.keyCode === 39) {
104 			return this.processCursor(rangeObject, event.keyCode);
105 		}
106 
107 		// BACKSPACE
108 		if (event.keyCode === 8) {
109 			event.preventDefault(); // prevent history.back() even on exception
110 			Aloha.execCommand( 'delete', false );
111 			return false;
112 		}
113 
114 		// DELETE
115 		if (event.keyCode === 46) {
116 			Aloha.execCommand( 'forwarddelete', false );
117 			return false;
118 		}
119 		
120 		// ENTER
121 		if  (event.keyCode === 13 ) {
122 			if (event.shiftKey) {
123 				Aloha.execCommand( 'insertlinebreak', false );
124 				return false;
125 			} else {
126 				Aloha.execCommand( 'insertparagraph', false );
127 				return false;
128 			}
129 		}
130 		return true;
131 	},

133 	/**
134 	 * Processing of cursor keys
135 	 * will currently detect blocks (elements with contenteditable=false)
136 	 * and selects them (normally the cursor would jump right past them)
137 	 *
138 	 * For each block an 'aloha-block-selected' event will be triggered.
139 	 *
140 	 * @param range the current range object
141 	 * @param keyCode keyCode of current keypress
142 	 * @return false if a block was found to prevent further events, true otherwise
143 	 */
144 	processCursor: function(range, keyCode) {
145 		var rt = range.getRangeTree(), // RangeTree reference
146 			i = 0,
147 			cursorLeft = keyCode === 37,
148 			cursorRight = keyCode === 39,
149 			nextSiblingIsBlock = false, // check whether the next sibling is a block (contenteditable = false)
150 			cursorIsWithinBlock = false, // check whether the cursor is positioned within a block (contenteditable = false)
151 			cursorAtLastPos = false, // check if the cursor is within the last position of the currently active dom element
152 			obj; // will contain references to dom objects
153 
154 		if ( !range.isCollapsed() ) {
155 			return true;
156 		}
157 
158 		for (;i < rt.length; i++) {
159 			if ( typeof rt[i].domobj !== 'undefined' ) {
160 				cursorAtLastPos = range.startOffset === rt[i].domobj.length;
161 				if (cursorAtLastPos) {
162 					nextSiblingIsBlock = jQuery( rt[i].domobj.nextSibling ).attr('contenteditable') === 'false';
163 					cursorIsWithinBlock = jQuery( rt[i].domobj ).parents('[contenteditable=false]').length > 0;
164 			
165 					if ( cursorRight && nextSiblingIsBlock ) {
166 						obj = rt[i].domobj.nextSibling;
167 						GENTICS.Utils.Dom.selectDomNode( obj );
168 						Aloha.trigger( 'aloha-block-selected', obj );
169 						Aloha.Selection.preventSelectionChanged();
170 						return false;
171 					}
172 
173 					if ( cursorLeft && cursorIsWithinBlock ) {
174 						obj = jQuery( rt[i].domobj ).parents('[contenteditable=false]').get(0);
175 						GENTICS.Utils.Dom.selectDomNode( obj );
176 						Aloha.trigger( 'aloha-block-selected', obj );
177 						Aloha.Selection.preventSelectionChanged();
178 						return false;
179 					}
180 				}
181 			}
182 		}

226 	/**
227 	 * insert an HTML Break <br /> into current selection
228 	 * @param Aloha.Selection.SelectionRange of the current selection
229 	 * @return void
230 	 */
231 	insertHTMLBreak: function(selectionTree, rangeObject, inBetweenMarkup) {
232 		var i, treeLength, el, jqEl, jqElBefore, jqElAfter, tmpObject, offset, checkObj;
233 
234 		inBetweenMarkup = inBetweenMarkup ? inBetweenMarkup: jQuery('<br/>');
235 		for ( i = 0, treeLength = selectionTree.length; i < treeLength; i++) {
236 			el = selectionTree[i];
237 			jqEl = el.domobj ? jQuery(el.domobj) : undefined;
238 			if (el.selection !== 'none') { // before cursor, leave this part inside the splitObject
239 				if (el.selection == 'collapsed') {
240 					// collapsed selection found (between nodes)
241 					if (i > 0) {
242 						// not at the start, so get the element to the left
243 						jqElBefore = jQuery(selectionTree[i-1].domobj);
244 						// and insert the break after it
245 						jqElBefore.after(inBetweenMarkup);
246 					} else {
247 						// at the start, so get the element to the right
248 						jqElAfter = jQuery(selectionTree[1].domobj);
249 						// and insert the break before it
250 						jqElAfter.before(inBetweenMarkup);
251 					}
252 
253 					// now set the range
254 					rangeObject.startContainer = rangeObject.endContainer = inBetweenMarkup[0].parentNode;
255 					rangeObject.startOffset = rangeObject.endOffset = GENTICS.Utils.Dom.getIndexInParent(inBetweenMarkup[0]) + 1;
256 					rangeObject.correctRange();
257 				} else if (el.domobj && el.domobj.nodeType === 3) { // textNode
258 					// when the textnode is immediately followed by a blocklevel element (like p, h1, ...) we need to add an additional br in between
259 					if (el.domobj.nextSibling
260 							&& el.domobj.nextSibling.nodeType == 1
261 							&& Aloha.Selection.replacingElements[el.domobj.nextSibling.nodeName
262 									.toLowerCase()]) {
263 						// TODO check whether this depends on the browser
264 						jqEl.after('<br/>');
265 					}
266 
267 					if (this.needEndingBreak()) {
268 						// when the textnode is the last inside a blocklevel element
269 						// (like p, h1, ...) we need to add an additional br as very
270 						// last object in the blocklevel element
271 						checkObj = el.domobj;
272 						while (checkObj) {
273 							if (checkObj.nextSibling) {
274 								checkObj = false;
275 							} else {
276 								// go to the parent
277 								checkObj = checkObj.parentNode;
278 								// found a blocklevel or list element, we are done
279 								if (GENTICS.Utils.Dom.isBlockLevelElement(checkObj) || GENTICS.Utils.Dom.isListElement(checkObj)) {
280 									break;
281 								}
282 								// reached the limit object, we are done
283 								if (checkObj === rangeObject.limitObject) {
284 									checkObj = false;
285 								}
286 							}
287 						}
288 						// when we found a blocklevel element, insert a break at the
289 						// end. Mark the break so that it is cleaned when the
290 						// content is fetched.
291 						if (checkObj) {
292 							jQuery(checkObj).append('<br class="aloha-cleanme"/>');
293 						}
294 					}
295 
296 					// insert the break
297 					jqEl.between(inBetweenMarkup, el.startOffset);
298 
299 					// correct the range
300 					// count the number of previous siblings
301 					offset = 0;
302 					tmpObject = inBetweenMarkup[0];
303 					while (tmpObject) {
304 						tmpObject = tmpObject.previousSibling;
305 						offset++;
306 					}
307 
308 					rangeObject.startContainer = inBetweenMarkup[0].parentNode;
309 					rangeObject.endContainer = inBetweenMarkup[0].parentNode;
310 					rangeObject.startOffset = offset;
311 					rangeObject.endOffset = offset;
312 					rangeObject.correctRange();
313 				} else if (el.domobj && el.domobj.nodeType === 1) { // other node, normally a break
314 					if (jqEl.parent().find('br.aloha-ephemera').length === 0) {
315 316 						// but before putting it, remove all:
317 						jQuery(rangeObject.limitObject).find('br.aloha-ephemera').remove();
318 						//  now put it:
319 						jQuery(rangeObject.commonAncestorContainer).append(this.getFillUpElement(rangeObject.splitObject));
320 					}
321 					jqEl.after(inBetweenMarkup);
322 
323 					// now set the selection. Since we just added one break do the currect el
324 					// the new position must be el's position + 1. el's position is the index
325 					// of the el in the selection tree, which is i. then we must add
326 					// another +1 because we want to be AFTER the object, not before. therefor +2
327 					rangeObject.startContainer = rangeObject.commonAncestorContainer;
328 					rangeObject.endContainer = rangeObject.startContainer;
329 					rangeObject.startOffset = i+2;
330 					rangeObject.endOffset = i+2;
331 					rangeObject.update();
332 				}
333 			}
334 		}
335 		rangeObject.select();
336 	},

361 	/**
362 	 * Recursive function to get the selected text from the selection tree starting at the given level
363 	 * @param selectionTree array of selectiontree elements
364 	 * @param astext true when the contents shall be fetched as text, false for getting as html markup
365 	 * @return selected text from that level (incluiding all sublevels)
366 	 */
367 	getFromSelectionTree: function (selectionTree, astext) {
368 		var text = '', i, treeLength, el, clone;
369 		for ( i = 0, treeLength = selectionTree.length; i < treeLength; i++) {
370 			el = selectionTree[i];
371 			if (el.selection == 'partial') {
372 				if (el.domobj.nodeType === 3) {
373 					// partial text node selected, get the selected part
374 					text += el.domobj.data.substring(el.startOffset, el.endOffset);
375 				} else if (el.domobj.nodeType === 1 && el.children) {
376 					// partial element node selected, do the recursion into the children
377 					if (astext) {
378 						text += this.getFromSelectionTree(el.children, astext);
379 					} else {
380 						// when the html shall be fetched, we create a clone of the element and remove all the children
381 						clone = jQuery(el.domobj).clone(false).empty();
382 						// then we do the recursion and add the selection into the clone
383 						clone.html(this.getFromSelectionTree(el.children, astext));
384 						// finally we get the html of the clone
385 						text += clone.outerHTML();
386 					}
387 				}
388 			} else if (el.selection == 'full') {
389 				if (el.domobj.nodeType === 3) {
390 					// full text node selected, get the text
391 					text += jQuery(el.domobj).text();
392 				} else if (el.domobj.nodeType === 1 && el.children) {
393 					// full element node selected, get the html of the node and all children
394 					text += astext ? jQuery(el.domobj).text() : jQuery(el.domobj).outerHTML();
395 				}
396 			}
397 		}
398 
399 		return text;
400 	},

443 	/**
444 	 * Recursively remove the selected items, starting with the given level in the selectiontree
445 	 * @param selectionTree current level of the selectiontree
446 	 * @param newRange new collapsed range to be set after the removal
447 	 */
448 	removeFromSelectionTree: function (selectionTree, newRange) {
449 		// remember the first found partially selected element node (in case we need
450 		// to merge it with the last found partially selected element node)
451 		var firstPartialElement, newdata, i, el, adjacentTextNode, treeLength;
452 
453 		// iterate through the selection tree
454 		for (i = 0, treeLength = selectionTree.length; i < treeLength; i++) {
455 			el = selectionTree[i];
456 			// check the type of selection
457 			if (el.selection == 'partial') {
458 				if (el.domobj.nodeType === 3) {
459 					// partial text node selected, so remove the selected portion
460 					newdata = '';
461 					if (el.startOffset > 0) {
462 						newdata += el.domobj.data.substring(0, el.startOffset);
463 					}
464 					if (el.endOffset < el.domobj.data.length) {
465 						newdata += el.domobj.data.substring(el.endOffset, el.domobj.data.length);
466 					}
467 					el.domobj.data = newdata;
468 
469 					// eventually set the new range (if not done before)
470 					if (!newRange.startContainer) {
471 						newRange.startContainer = newRange.endContainer = el.domobj;
472 						newRange.startOffset = newRange.endOffset = el.startOffset;
473 					}
474 				} else if (el.domobj.nodeType === 1 && el.children) {
475 					// partial element node selected, so do the recursion into the children
476 					this.removeFromSelectionTree(el.children, newRange);
477 					if (firstPartialElement) {
478 						// when the first parially selected element is the same type
479 						// of element, we need to merge them
480 						if (firstPartialElement.nodeName == el.domobj.nodeName) {
481 							// merge the nodes
482 							jQuery(firstPartialElement).append(jQuery(el.domobj).contents());
483 							// and remove the latter one
484 							jQuery(el.domobj).remove();
485 						}
486 					} else {
487 						// remember this element as first partially selected element
488 						firstPartialElement = el.domobj;
489 					}
490 				}
491 			} else if (el.selection == 'full') {
492 				// eventually set the new range (if not done before)
493 				if (!newRange.startContainer) {
494 					adjacentTextNode = GENTICS.Utils.Dom.searchAdjacentTextNode(el.domobj.parentNode, GENTICS.Utils.Dom.getIndexInParent(el.domobj) + 1, false, {'blocklevel' : false});
495 					if (adjacentTextNode) {
496 						newRange.startContainer = newRange.endContainer = adjacentTextNode;
497 						newRange.startOffset = newRange.endOffset = 0;
498 					} else {
499 						newRange.startContainer = newRange.endContainer = el.domobj.parentNode;
500 						newRange.startOffset = newRange.endOffset = GENTICS.Utils.Dom.getIndexInParent(el.domobj) + 1;
501 					}
502 				}
503 
504 				// full node selected, so just remove it (will also remove all children)
505 				jQuery(el.domobj).remove();
506 			}
507 		}
508 	},

510 	/**
511 	 * split passed rangeObject without or with optional markup
512 	 * @param Aloha.Selection.SelectionRange of the current selection
513 	 * @param markup object (jQuery) to insert in between the split elements
514 	 * @return void
515 	 */
516 	splitRangeObject: function(rangeObject, markup) {
517 		// UAAAA: first check where the markup can be inserted... *grrrrr*, then decide where to split
518 		// object which is split up
519 		var
520 			splitObject = jQuery(rangeObject.splitObject),
521 			selectionTree, insertAfterObject, followUpContainer;
522 
523 		// update the commonAncestor with the splitObject (so that the selectionTree is correct)
524 		rangeObject.update(rangeObject.splitObject); // set the splitObject as new commonAncestorContainer and update the selectionTree
525 
526 		// calculate the selection tree. NOTE: it is necessary to do this before
527 		// getting the followupcontainer, since getting the selection tree might
528 529 		// possibly merge text nodes, which would lead to differences in the followupcontainer
530 		selectionTree = rangeObject.getSelectionTree();
531 
532 		// object to be inserted after the splitObject
533 		followUpContainer = this.getSplitFollowUpContainer(rangeObject);
534 
535 		// now split up the splitObject into itself AND the followUpContainer
536 		this.splitRangeObjectHelper(selectionTree, rangeObject, followUpContainer); // split the current object into itself and the followUpContainer
537 
538 		// check whether the followupcontainer is still marked for removal
539 		if (followUpContainer.hasClass('preparedForRemoval')) {
540 			// TODO shall we just remove the class or shall we not use the followupcontainer?
541 			followUpContainer.removeClass('preparedForRemoval');
542 		}
543 
544 		// now let's find the place, where the followUp is inserted afterwards. normally that's the splitObject itself, but in
545 		// some cases it might be their parent (e.g. inside a list, a <p> followUp must be inserted outside the list)
546 		insertAfterObject = this.getInsertAfterObject(rangeObject, followUpContainer);
547 
548 		// now insert the followUpContainer
549 		jQuery(followUpContainer).insertAfter(insertAfterObject); // attach the followUpContainer right after the insertAfterObject
550 
551 		// in some cases, we want to remove the "empty" splitObject (e.g. LIs, if enter was hit twice)
552 		if (rangeObject.splitObject.nodeName.toLowerCase() === 'li' && !Aloha.Selection.standardTextLevelSemanticsComparator(rangeObject.splitObject, followUpContainer)) {
553 			jQuery(rangeObject.splitObject).remove();
554 		}
555 
556 		rangeObject.startContainer = null;
557 		// first check whether the followUpContainer starts with a <br/>
558 		// if so, place the cursor right before the <br/>
559 		var followContents = followUpContainer.contents();
560 		if (followContents.length > 0
561 				&& followContents.get(0).nodeType == 1
562 				&& followContents.get(0).nodeName.toLowerCase() === 'br') {
563 			rangeObject.startContainer = followUpContainer.get(0);
564 		}
565 
566 		if (!rangeObject.startContainer) {
567 			// find a possible text node in the followUpContainer and set the selection to it
568 			// if no textnode is available, set the selection to the followup container itself
569 			rangeObject.startContainer = followUpContainer.textNodes(true, true).first().get(0);
570 		}
571 		if (!rangeObject.startContainer) { // if no text node was found, select the parent object of <br class="aloha-ephemera" />
572 			rangeObject.startContainer = followUpContainer.textNodes(false).first().parent().get(0);
573 		}
574 		if (rangeObject.startContainer) {
575 			// the cursor is always at the beginning of the followUp
576 			rangeObject.endContainer = rangeObject.startContainer;
577 			rangeObject.startOffset = 0;
578 			rangeObject.endOffset = 0;
579 		} else {
580 			rangeObject.startContainer = rangeObject.endContainer = followUpContainer.parent().get(0);
581 			rangeObject.startOffset = rangeObject.endOffset = GENTICS.Utils.Dom.getIndexInParent(followUpContainer.get(0));
582 		}
583

658 	/**
659 	 * recursive method to parallelly walk through two dom subtrees, leave elements before startContainer in first subtree and move rest to other
660 	 * @param selectionTree tree to iterate over as contained in rangeObject. must be passed separately to allow recursion in the selection tree, but not in the rangeObject
661 	 * @param rangeObject Aloha.Selection.SelectionRange of the current selection
662 	 * @param followUpContainer optional jQuery object; if provided the rangeObject will be split and the second part will be insert inside of this object
663 	 * @param inBetweenMarkup jQuery object to be inserted between the two split parts. will be either a <br> (if no followUpContainer is passed) OR e.g. a table, which must be inserted between the splitobject AND the follow up
664 	 * @return void
665 	 */
666 	splitRangeObjectHelper: function(selectionTree, rangeObject, followUpContainer, inBetweenMarkup) {
667 		if (!followUpContainer) {
668 			Aloha.Log.warn(this, 'no followUpContainer, no inBetweenMarkup, nothing to do...');
669 		}
670 
671 		var fillUpElement = this.getFillUpElement(rangeObject.splitObject),
672 			splitObject = jQuery(rangeObject.splitObject),
673 			startMoving = false,
674 			el, i, completeText, jqObj, mirrorLevel, parent, treeLength;
675 		
676 		if (selectionTree.length > 0) {
677 			mirrorLevel = followUpContainer.contents();
678 
679 			// if length of mirrorLevel and selectionTree are not equal, the mirrorLevel must be corrected. this happens, when the mirrorLevel contains whitespace textNodes
680 			if (mirrorLevel.length !== selectionTree.length) {
681 				this.removeElementContentWhitespaceObj(mirrorLevel);
682 			}
683 
684 			for (i = 0, treeLength = selectionTree.length; i < treeLength; i++) {
685 				el = selectionTree[i];
686 				// remove all objects in the mirrorLevel, which are BEFORE the cursor
687 				// OR if the cursor is at the last position of the last Textnode (causing an empty followUpContainer to be appended)
688 				if (
689 						(el.selection === 'none' && startMoving === false) ||
690 						(el.domobj && el.domobj.nodeType === 3 && el === selectionTree[ (selectionTree.length-1) ] && el.startOffset === el.domobj.data.length)
691 				) {
692 					// iteration is before cursor, leave this part inside the splitObject, remove from followUpContainer
693 					// however if the object to remove is the last existing textNode within the followUpContainer, insert a BR instead
694 					// otherwise the followUpContainer is invalid and takes up no vertical space
695 
696 					if (followUpContainer.textNodes().length > 1 || (el.domobj.nodeType === 1 && el.children.length === 0)) {
697 						// note: the second part of the if (el.domobj.nodeType === 1 && el.children.length === 0) covers a very special condition,
698 						// where an empty tag is located right before the cursor when pressing enter. In this case the empty tag would not be
699 						// removed correctly otherwise
700 						mirrorLevel.eq(i).remove();
701 					} else if (GENTICS.Utils.Dom.isSplitObject(followUpContainer[0])) {
702 						if (fillUpElement) {
703 							followUpContainer.html(fillUpElement); // for your zoological german knowhow: ephemera = Eintagsfliege
704 						} else {
705 							followUpContainer.empty();
706 						}
707 					} else {
708 						followUpContainer.empty();
709 						followUpContainer.addClass('preparedForRemoval');
710 					}
711 					continue;
712 				} else {
713 					// split objects, which are AT the cursor Position or directly above
714 					if (el.selection !== 'none') { // before cursor, leave this part inside the splitObject
715 						// TODO better check for selection == 'partial' here?
716 						if (el.domobj && el.domobj.nodeType === 3 && el.startOffset !== undefined) {
717 							completeText = el.domobj.data;
718 							if (el.startOffset > 0) {// first check, if there will be some text left in the splitObject
719 								el.domobj.data = completeText.substr(0, el.startOffset);
720 							} else if (selectionTree.length > 1) { // if not, check if the splitObject contains more than one node, because then it can be removed. this happens, when ENTER is pressed inside of a textnode, but not at the borders
721 								jQuery(el.domobj).remove();
722 							} else { // if the "empty" textnode is the last node left in the splitObject, replace it with a ephemera break
723 								// if the parent is a blocklevel element, we insert the fillup element
724 								parent = jQuery(el.domobj).parent();
725 								if (GENTICS.Utils.Dom.isSplitObject(parent[0])) {
726 									if (fillUpElement) {
727 										parent.html(fillUpElement);
728 									} else {
729 										parent.empty();
730 									}
731 								} else {
732 									// if the parent is no blocklevel element and would be empty now, we completely remove it
733 									parent.remove();
734 								}
735 							}
736 							if (completeText.length - el.startOffset > 0) {
737 								// first check if there is text left to put in the followUpContainer's textnode. this happens, when ENTER is pressed inside of a textnode, but not at the borders
738 								mirrorLevel[i].data = completeText.substr(el.startOffset, completeText.length);
739 							} else if (mirrorLevel.length > 1) {
740 								// if not, check if the followUpContainer contains more than one node, because if yes, the "empty" textnode can be removed
741 								mirrorLevel.eq( (i) ).remove();
742 							} else if (GENTICS.Utils.Dom.isBlockLevelElement(followUpContainer[0])) {
743 								// if the "empty" textnode is the last node left in the followUpContainer (which is a blocklevel element), replace it with a ephemera break
744 								if (fillUpElement) {
745 									followUpContainer.html(fillUpElement);
746 								} else {
747 									followUpContainer.empty();
748 								}
749 							} else {
750 								// if the "empty" textnode is the last node left in a non-blocklevel element, mark it for removal
751 								followUpContainer.empty();
752 								followUpContainer.addClass('preparedForRemoval');
753 							}
754 						}
755 						startMoving = true;
756 						if (el.children.length > 0) {
757 							this.splitRangeObjectHelper(el.children, rangeObject, mirrorLevel.eq(i), inBetweenMarkup);
758 						}
759 					} else {
760 
761 						// remove all objects in the origin, which are AFTER the cursor
762 						if (el.selection === 'none' && startMoving === true) {
763 							// iteration is after cursor, remove from splitObject and leave this part inside the followUpContainer
764 							jqObj = jQuery(el.domobj).remove();
765 						}
766 					}
767 				}
768 			}
769 		} else {
770 			Aloha.Log.error(this, 'can not split splitObject due to an empty selection tree');
771 		}
772 
773 		// and finally cleanup: remove all fillUps > 1
774 		splitObject.find('br.aloha-ephemera:gt(0)').remove(); // remove all elements greater than (gt) 0, that also means: leave one
775 		followUpContainer.find('br.aloha-ephemera:gt(0)').remove(); // remove all elements greater than (gt) 0, that also means: leave one
776 
777 		// remove objects prepared for removal
778 		splitObject.find('.preparedForRemoval').remove();
779 		followUpContainer.find('.preparedForRemoval').remove();
780 
781 		// if splitObject / followUp are empty, place a fillUp inside
782 		if (splitObject.contents().length === 0 && GENTICS.Utils.Dom.isSplitObject(splitObject[0]) && fillUpElement) {
783 			splitObject.html(fillUpElement);
784 		}
785 		
786 		if (followUpContainer.contents().length === 0 && GENTICS.Utils.Dom.isSplitObject(followUpContainer[0]) && fillUpElement) {
787 			followUpContainer.html(fillUpElement);
788 		}
789 	},

791 	/**
792 	 * returns a jQuery object fitting the passed splitObject as follow up object
793 	 * examples,
794 	 * - when passed a p it will return an empty p (clone of the passed p)
795 	 * - when passed an h1, it will return either an h1 (clone of the passed one) or a new p (if the collapsed selection was at the end)
796 	 * @param rangeObject Aloha.RangeObject
797 	 * @return void
798 	 */
799 	getSplitFollowUpContainer: function(rangeObject) {
800 		var
801 			tagName = rangeObject.splitObject.nodeName.toLowerCase(),
802 			returnObj, inside, lastObj;
803 
804 		switch(tagName) {
805 			case 'h1':
806 			case 'h2':
807 			case 'h3':
808 			case 'h4':
809 			case 'h5':
810 			case 'h6':
811 				// get the last textnode in the splitobject, but don't consider aloha-cleanme elements
812 			 	lastObj = jQuery(rangeObject.splitObject).textNodes(':not(.aloha-cleanme)').last()[0];
813 				// special case: when enter is hit at the end of a heading, the followUp should be a <p>
814 				if (lastObj && rangeObject.startContainer === lastObj && rangeObject.startOffset === lastObj.length) {
815 					returnObj = jQuery('<p></p>');
816 					inside = jQuery(rangeObject.splitObject).clone().contents();
817 					returnObj.append(inside);
818 					return returnObj;
819 				}
820 				break;
821 
822 			case 'li':
823 				// TODO check whether the li is the last one
824 				// special case: if enter is hit twice inside a list, the next item should be a <p> (and inserted outside the list)
825 				if (rangeObject.startContainer.nodeName.toLowerCase() === 'br' && jQuery(rangeObject.startContainer).hasClass('aloha-ephemera')) {
826 					returnObj = jQuery('<p></p>');
827 					inside = jQuery(rangeObject.splitObject).clone().contents();
828 					returnObj.append(inside);
829 					return returnObj;
830 				}
831 				// when the li is the last one and empty, we also just return a <p>
832 				if (!rangeObject.splitObject.nextSibling && jQuery.trim(jQuery(rangeObject.splitObject).text()).length === 0) {
833 					returnObj = jQuery('<p></p>');
834 					return returnObj;
835 				}
836 		}
837 
838 		return jQuery(rangeObject.splitObject).clone();
839 	},

841 	/**
842 	 * Transform the given domobj into an object with the given new nodeName.
843 	 * Preserves the content and all attributes. If a range object is given, also the range will be preserved
844 	 * @param domobj dom object to transform
845 	 * @param nodeName new node name
846 	 * @param range range object
847 	 * @api
848 	 * @return new object as jQuery object
849 	 */
850 	transformDomObject: function (domobj, nodeName, range) {
851 		// first create the new element
852 		var
853 			jqOldObj = jQuery(domobj),
854 			jqNewObj = jQuery('<' + nodeName + '></' + nodeName + '>'),
855 			i;
856 
857 		// TODO what about events? css properties?
858 
859 		// copy attributes
860 		if (jqOldObj[0].attributes) {
861 			for (i = 0; i < jqOldObj[0].attributes.length; i++) {
862 				jqNewObj.attr(jqOldObj[0].attributes[i].nodeName, jqOldObj[0].attributes[i].nodeValue);
863 			}
864 		}
865 
866 		// copy inline CSS
867 		if (jqOldObj[0].style && jqOldObj[0].style.cssText) {
868 			jqNewObj[0].style.cssText = jqOldObj[0].style.cssText;
869 		}
870 
871 		// now move the contents of the old dom object into the new dom object
872 		jqOldObj.contents().appendTo(jqNewObj);
873 
874 		// finally replace the old object with the new one
875 		jqOldObj.replaceWith(jqNewObj);
876 
877 		// preserve the range
878 		if (range) {
879 			if (range.startContainer == domobj) {
880 				range.startContainer = jqNewObj.get(0);
881 			}
882 			if (range.endContainer == domobj) {
883 				range.endContainer = jqNewObj.get(0);
884 			}
885 		}
886 		return jqNewObj;
887 	},
