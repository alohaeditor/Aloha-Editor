 26 /**
 27  * Class containing the diff, match and patch methods.
 28  * @constructor
 29  */
 30 function diff_match_patch() {
 31 
 32   // Defaults.
 33   // Redefine these in your program to override the defaults.
 34 
 35   // Number of seconds to map a diff before giving up (0 for infinity).
 36   this.Diff_Timeout = 1.0;
 37   // Cost of an empty edit operation in terms of edit characters.
 38   this.Diff_EditCost = 4;
 39   // At what point is no match declared (0.0 = perfection, 1.0 = very loose).
 40   this.Match_Threshold = 0.5;
 41   // How far to search for a match (0 = exact location, 1000+ = broad match).
 42   // A match this many characters away from the expected location will add
 43   // 1.0 to the score (0.0 is a perfect match).
 44   this.Match_Distance = 1000;
 45   // When deleting a large block of text (over ~64 characters), how close does
 46   // the contents have to match the expected contents. (0.0 = perfection,
 47   // 1.0 = very loose).  Note that Match_Threshold controls how closely the
 48   // end points of a delete need to match.
 49   this.Patch_DeleteThreshold = 0.5;
 50   // Chunk size for context length.
 51   this.Patch_Margin = 4;
 52 
 53   // The number of bits in an int.
 54   this.Match_MaxBits = 32;
 55 }

 61 /**
 62  * The data structure representing a diff is an array of tuples:
 63  * [[DIFF_DELETE, 'Hello'], [DIFF_INSERT, 'Goodbye'], [DIFF_EQUAL, ' world.']]
 64  * which means: delete 'Hello', add 'Goodbye' and keep ' world.'
 65  */
 66 var DIFF_DELETE = -1;
 67 var DIFF_INSERT = 1;
 68 var DIFF_EQUAL = 0;
 69 
 70 /** @typedef {!Array.<number|string>} */
 71 diff_match_patch.Diff;
 72 
 73 
 74 /**
 75  * Find the differences between two texts.  Simplifies the problem by stripping
 76  * any common prefix or suffix off the texts before diffing.
 77  * @param {string} text1 Old string to be diffed.
 78  * @param {string} text2 New string to be diffed.
 79  * @param {boolean=} opt_checklines Optional speedup flag. If present and false,
 80  *     then don't run a line-level diff first to identify the changed areas.
 81  *     Defaults to true, which does a faster, slightly less optimal diff.
 82  * @param {number} opt_deadline Optional time when the diff should be complete
 83  *     by.  Used internally for recursive calls.  Users should set DiffTimeout
 84  *     instead.
 85  * @return {!Array.<!diff_match_patch.Diff>} Array of diff tuples.
 86  */
 87 diff_match_patch.prototype.diff_main = function(text1, text2, opt_checklines,
 88     opt_deadline) {
 89   // Set a deadline by which time the diff must be complete.
 90   if (typeof opt_deadline == 'undefined') {
 91     if (this.Diff_Timeout <= 0) {
 92       opt_deadline = Number.MAX_VALUE;
 93     } else {
 94       opt_deadline = (new Date).getTime() + this.Diff_Timeout * 1000;
 95     }
 96   }
 97   var deadline = opt_deadline;
 98 
 99   // Check for null inputs.
100   if (text1 == null || text2 == null) {
101     throw new Error('Null input. (diff_main)');
102   }
103 
104   // Check for equality (speedup).
105   if (text1 == text2) {
106     if (text1) {
107       return [[DIFF_EQUAL, text1]];
108     }
109     return [];
110   }
111 
112   if (typeof opt_checklines == 'undefined') {
113     opt_checklines = true;
114   }
115   var checklines = opt_checklines;
116 
117   // Trim off common prefix (speedup).
118   var commonlength = this.diff_commonPrefix(text1, text2);
119   var commonprefix = text1.substring(0, commonlength);
120   text1 = text1.substring(commonlength);
121   text2 = text2.substring(commonlength);
122 
123   // Trim off common suffix (speedup).
124   commonlength = this.diff_commonSuffix(text1, text2);
125   var commonsuffix = text1.substring(text1.length - commonlength);
126   text1 = text1.substring(0, text1.length - commonlength);
127   text2 = text2.substring(0, text2.length - commonlength);
128 
129   // Compute the diff on the middle block.
130   var diffs = this.diff_compute_(text1, text2, checklines, deadline);
131 
132   // Restore the prefix and suffix.
133   if (commonprefix) {
134     diffs.unshift([DIFF_EQUAL, commonprefix]);
135   }
136   if (commonsuffix) {
137     diffs.push([DIFF_EQUAL, commonsuffix]);
138   }
139   this.diff_cleanupMerge(diffs);
140   return diffs;
141 };

144 /**
145  * Find the differences between two texts.  Assumes that the texts do not
146  * have any common prefix or suffix.
147  * @param {string} text1 Old string to be diffed.
148  * @param {string} text2 New string to be diffed.
149  * @param {boolean} checklines Speedup flag.  If false, then don't run a
150  *     line-level diff first to identify the changed areas.
151  *     If true, then run a faster, slightly less optimal diff.
152  * @param {number} deadline Time when the diff should be complete by.
153  * @return {!Array.<!diff_match_patch.Diff>} Array of diff tuples.
154  * @private
155  */
156 diff_match_patch.prototype.diff_compute_ = function(text1, text2, checklines,
157     deadline) {
158   var diffs;
159 
160   if (!text1) {
161     // Just add some text (speedup).
162 163     return [[DIFF_INSERT, text2]];
164   }
165 
166   if (!text2) {
167     // Just delete some text (speedup).
168     return [[DIFF_DELETE, text1]];
169   }
170 
171   var longtext = text1.length > text2.length ? text1 : text2;
172   var shorttext = text1.length > text2.length ? text2 : text1;
173   var i = longtext.indexOf(shorttext);
174   if (i != -1) {
175     // Shorter text is inside the longer text (speedup).
176     diffs = [[DIFF_INSERT, longtext.substring(0, i)],
177              [DIFF_EQUAL, shorttext],
178              [DIFF_INSERT, longtext.substring(i + shorttext.length)]];
179     // Swap insertions for deletions if diff is reversed.
180     if (text1.length > text2.length) {
181       diffs[0][0] = diffs[2][0] = DIFF_DELETE;
182     }
183     return diffs;
184   }
185 
186   if (shorttext.length == 1) {
187     // Single character string.
188     // After the previous speedup, the character can't be an equality.
189     return [[DIFF_DELETE, text1], [DIFF_INSERT, text2]];
190   }
191   longtext = shorttext = null;  // Garbage collect.
192 
193   // Check to see if the problem can be split in two.
194   var hm = this.diff_halfMatch_(text1, text2);
195   if (hm) {
196     // A half-match was found, sort out the return data.
197     var text1_a = hm[0];
198     var text1_b = hm[1];
199     var text2_a = hm[2];
200     var text2_b = hm[3];
201     var mid_common = hm[4];
202     // Send both pairs off for separate processing.
203     var diffs_a = this.diff_main(text1_a, text2_a, checklines, deadline);
204     var diffs_b = this.diff_main(text1_b, text2_b, checklines, deadline);
205     // Merge the results.
206     return diffs_a.concat([[DIFF_EQUAL, mid_common]], diffs_b);
207   }
208 
209   if (checklines && text1.length > 100 && text2.length > 100) {
210     return this.diff_lineMode_(text1, text2, deadline);
211   }
212 
213   return this.diff_bisect_(text1, text2, deadline);
214 };

217 /**
218  * Do a quick line-level diff on both strings, then rediff the parts for
219  * greater accuracy.
220  * This speedup can produce non-minimal diffs.
221  * @param {string} text1 Old string to be diffed.
222  * @param {string} text2 New string to be diffed.
223  * @param {number} deadline Time when the diff should be complete by.
224  * @return {!Array.<!diff_match_patch.Diff>} Array of diff tuples.
225  * @private
226  */
227 diff_match_patch.prototype.diff_lineMode_ = function(text1, text2, deadline) {
228   // Scan the text on a line-by-line basis first.
229   var a = this.diff_linesToChars_(text1, text2);
230   text1 = /** @type {string} */(a[0]);
231   text2 = /** @type {string} */(a[1]);
232   var linearray = /** @type {!Array.<string>} */(a[2]);
233 
234   var diffs = this.diff_bisect_(text1, text2, deadline);
235 
236   // Convert the diff back to original text.
237   this.diff_charsToLines_(diffs, linearray);
238   // Eliminate freak matches (e.g. blank lines)
239   this.diff_cleanupSemantic(diffs);
240 
241   // Rediff any replacement blocks, this time character-by-character.
242   // Add a dummy entry at the end.
243   diffs.push([DIFF_EQUAL, '']);
244   var pointer = 0;
245   var count_delete = 0;
246   var count_insert = 0;
247   var text_delete = '';
248   var text_insert = '';
249   while (pointer < diffs.length) {
250     switch (diffs[pointer][0]) {
251       case DIFF_INSERT:
252         count_insert++;
253         text_insert += diffs[pointer][1];
254         break;
255       case DIFF_DELETE:
256         count_delete++;
257         text_delete += diffs[pointer][1];
258         break;
259       case DIFF_EQUAL:
260         // Upon reaching an equality, check for prior redundancies.
261         if (count_delete >= 1 && count_insert >= 1) {
262           // Delete the offending records and add the merged ones.
263           var a = this.diff_main(text_delete, text_insert, false, deadline);
264           diffs.splice(pointer - count_delete - count_insert,
265                        count_delete + count_insert);
266           pointer = pointer - count_delete - count_insert;
267           for (var j = a.length - 1; j >= 0; j--) {
268             diffs.splice(pointer, 0, a[j]);
269           }
270           pointer = pointer + a.length;
271         }
272         count_insert = 0;
273         count_delete = 0;
274         text_delete = '';
275         text_insert = '';
276         break;
277     }
278     pointer++;
279   }
280   diffs.pop();  // Remove the dummy entry at the end.
281 
282   return diffs;
283 };

286 /**
287  * Find the 'middle snake' of a diff, split the problem in two
288  * and return the recursively constructed diff.
289  * See Myers 1986 paper: An O(ND) Difference Algorithm and Its Variations.
290  * @param {string} text1 Old string to be diffed.
291  * @param {string} text2 New string to be diffed.
292  * @param {number} deadline Time at which to bail if not yet complete.
293  * @return {!Array.<!diff_match_patch.Diff>} Array of diff tuples.
294  * @private
295  */
296 diff_match_patch.prototype.diff_bisect_ = function(text1, text2, deadline) {
297   // Cache the text lengths to prevent multiple calls.
298   var text1_length = text1.length;
299   var text2_length = text2.length;
300   var max_d = Math.ceil((text1_length + text2_length) / 2);
301   var v_offset = max_d;
302   var v_length = 2 * max_d;
303   var v1 = new Array(v_length);
304   var v2 = new Array(v_length);
305   // Setting all elements to -1 is faster in Chrome & Firefox than mixing
306   // integers and undefined.
307   for (var x = 0; x < v_length; x++) {
308     v1[x] = -1;
309     v2[x] = -1;
310   }
311   v1[v_offset + 1] = 0;
312   v2[v_offset + 1] = 0;
313   var delta = text1_length - text2_length;
314   // If the total number of characters is odd, then the front path will collide
315   // with the reverse path.
316   var front = (delta % 2 != 0);
317   // Offsets for start and end of k loop.
318   // Prevents mapping of space beyond the grid.
319   var k1start = 0;
320   var k1end = 0;
321   var k2start = 0;
322   var k2end = 0;
323   for (var d = 0; d < max_d; d++) {
324     // Bail out if deadline is reached.
325     if ((new Date()).getTime() > deadline) {
326       break;
327     }
328 
329     // Walk the front path one step.
330     for (var k1 = -d + k1start; k1 <= d - k1end; k1 += 2) {
331       var k1_offset = v_offset + k1;
332       var x1;
333       if (k1 == -d || k1 != d && v1[k1_offset - 1] < v1[k1_offset + 1]) {
334         x1 = v1[k1_offset + 1];
335       } else {
336         x1 = v1[k1_offset - 1] + 1;
337       }
338       var y1 = x1 - k1;
339       while (x1 < text1_length && y1 < text2_length &&
340              text1.charAt(x1) == text2.charAt(y1)) {
341         x1++;
342         y1++;
343       }
344       v1[k1_offset] = x1;
345       if (x1 > text1_length) {
346         // Ran off the right of the graph.
347         k1end += 2;
348       } else if (y1 > text2_length) {
349         // Ran off the bottom of the graph.
350         k1start += 2;
351       } else if (front) {
352         var k2_offset = v_offset + delta - k1;
353         if (k2_offset >= 0 && k2_offset < v_length && v2[k2_offset] != -1) {
354           // Mirror x2 onto top-left coordinate system.
355           var x2 = text1_length - v2[k2_offset];
356           if (x1 >= x2) {
357             // Overlap detected.
358             return this.diff_bisectSplit_(text1, text2, x1, y1, deadline);
359           }
360         }
361       }
362     }
363 
364     // Walk the reverse path one step.
365     for (var k2 = -d + k2start; k2 <= d - k2end; k2 += 2) {
366       var k2_offset = v_offset + k2;
367       var x2;
368       if (k2 == -d || k2 != d && v2[k2_offset - 1] < v2[k2_offset + 1]) {
369         x2 = v2[k2_offset + 1];
370       } else {
371         x2 = v2[k2_offset - 1] + 1;
372       }
373       var y2 = x2 - k2;
374       while (x2 < text1_length && y2 < text2_length &&
375              text1.charAt(text1_length - x2 - 1) ==
376              text2.charAt(text2_length - y2 - 1)) {
377         x2++;
378         y2++;
379       }
380       v2[k2_offset] = x2;
381       if (x2 > text1_length) {
382         // Ran off the left of the graph.
383         k2end += 2;
384       } else if (y2 > text2_length) {
385         // Ran off the top of the graph.
386         k2start += 2;
387       } else if (!front) {
388         var k1_offset = v_offset + delta - k2;
389         if (k1_offset >= 0 && k1_offset < v_length && v1[k1_offset] != -1) {
390           var x1 = v1[k1_offset];
391           var y1 = v_offset + x1 - k1_offset;
392           // Mirror x2 onto top-left coordinate system.
393           x2 = text1_length - x2;
394           if (x1 >= x2) {
395             // Overlap detected.
396             return this.diff_bisectSplit_(text1, text2, x1, y1, deadline);
397           }
398         }
399       }
400     }
401   }
402   // Diff took too long and hit the deadline or
403   // number of diffs equals number of characters, no commonality at all.
404   return [[DIFF_DELETE, text1], [DIFF_INSERT, text2]];
405 };

408 /**
409  * Given the location of the 'middle snake', split the diff in two parts
410  * and recurse.
411  * @param {string} text1 Old string to be diffed.
412  * @param {string} text2 New string to be diffed.
413  * @param {number} x Index of split point in text1.
414  * @param {number} y Index of split point in text2.
415  * @param {number} deadline Time at which to bail if not yet complete.
416  * @return {!Array.<!diff_match_patch.Diff>} Array of diff tuples.
417  * @private
418  */
419 diff_match_patch.prototype.diff_bisectSplit_ = function(text1, text2, x, y,
420     deadline) {
421   var text1a = text1.substring(0, x);
422   var text2a = text2.substring(0, y);
423   var text1b = text1.substring(x);
424 425   var text2b = text2.substring(y);
426 
427   // Compute both diffs serially.
428   var diffs = this.diff_main(text1a, text2a, false, deadline);
429   var diffsb = this.diff_main(text1b, text2b, false, deadline);
430 
431   return diffs.concat(diffsb);
432 };

435 /**
436  * Split two texts into an array of strings.  Reduce the texts to a string of
437  * hashes where each Unicode character represents one line.
438  * @param {string} text1 First string.
439  * @param {string} text2 Second string.
440  * @return {!Array.<string|!Array.<string>>} Three element Array, containing the
441  *     encoded text1, the encoded text2 and the array of unique strings.  The
442  *     zeroth element of the array of unique strings is intentionally blank.
443  * @private
444  */
445 diff_match_patch.prototype.diff_linesToChars_ = function(text1, text2) {
446   var lineArray = [];  // e.g. lineArray[4] == 'Hello\n'
447   var lineHash = {};   // e.g. lineHash['Hello\n'] == 4
448 
449   // '\x00' is a valid character, but various debuggers don't like it.
450   // So we'll insert a junk entry to avoid generating a null character.
451   lineArray[0] = '';
452 
453   /**
454    * Split a text into an array of strings.  Reduce the texts to a string of
455    * hashes where each Unicode character represents one line.
456    * Modifies linearray and linehash through being a closure.
457    * @param {string} text String to encode.
458    * @return {string} Encoded string.
459    * @private
460    */
461   function diff_linesToCharsMunge_(text) {
462     var chars = '';
463     // Walk the text, pulling out a substring for each line.
464     // text.split('\n') would would temporarily double our memory footprint.
465     // Modifying text would create many large strings to garbage collect.
466     var lineStart = 0;
467     var lineEnd = -1;
468     // Keeping our own length variable is faster than looking it up.
469     var lineArrayLength = lineArray.length;
470     while (lineEnd < text.length - 1) {
471       lineEnd = text.indexOf('\n', lineStart);
472       if (lineEnd == -1) {
473         lineEnd = text.length - 1;
474       }
475       var line = text.substring(lineStart, lineEnd + 1);
476       lineStart = lineEnd + 1;
477 
478       if (lineHash.hasOwnProperty ? lineHash.hasOwnProperty(line) :
479           (lineHash[line] !== undefined)) {
480         chars += String.fromCharCode(lineHash[line]);
481       } else {
482         chars += String.fromCharCode(lineArrayLength);
483         lineHash[line] = lineArrayLength;
484         lineArray[lineArrayLength++] = line;
485       }
486     }
487     return chars;
488   }
489 
490   var chars1 = diff_linesToCharsMunge_(text1);
491   var chars2 = diff_linesToCharsMunge_(text2);
492   return [chars1, chars2, lineArray];
493 };

496 /**
497  * Rehydrate the text in a diff from a string of line hashes to real lines of
498  * text.
499  * @param {!Array.<!diff_match_patch.Diff>} diffs Array of diff tuples.
500  * @param {!Array.<string>} lineArray Array of unique strings.
501  * @private
502  */
503 diff_match_patch.prototype.diff_charsToLines_ = function(diffs, lineArray) {
504   for (var x = 0; x < diffs.length; x++) {
505     var chars = diffs[x][1];
506     var text = [];
507     for (var y = 0; y < chars.length; y++) {
508       text[y] = lineArray[chars.charCodeAt(y)];
509     }
510     diffs[x][1] = text.join('');
511   }
512 };

515 /**
516  * Determine the common prefix of two strings.
517  * @param {string} text1 First string.
518  * @param {string} text2 Second string.
519  * @return {number} The number of characters common to the start of each
520  *     string.
521  */
522 diff_match_patch.prototype.diff_commonPrefix = function(text1, text2) {
523   // Quick check for common null cases.
524   if (!text1 || !text2 || text1.charAt(0) != text2.charAt(0)) {
525     return 0;
526   }
527   // Binary search.
528   // Performance analysis: http://neil.fraser.name/news/2007/10/09/
529   var pointermin = 0;
530   var pointermax = Math.min(text1.length, text2.length);
531   var pointermid = pointermax;
532   var pointerstart = 0;
533   while (pointermin < pointermid) {
534     if (text1.substring(pointerstart, pointermid) ==
535         text2.substring(pointerstart, pointermid)) {
536       pointermin = pointermid;
537       pointerstart = pointermin;
538     } else {
539       pointermax = pointermid;
540     }
541     pointermid = Math.floor((pointermax - pointermin) / 2 + pointermin);
542   }
543   return pointermid;
544 };

547 /**
548  * Determine the common suffix of two strings.
549  * @param {string} text1 First string.
550  * @param {string} text2 Second string.
551  * @return {number} The number of characters common to the end of each string.
552  */
553 diff_match_patch.prototype.diff_commonSuffix = function(text1, text2) {
554   // Quick check for common null cases.
555   if (!text1 || !text2 ||
556       text1.charAt(text1.length - 1) != text2.charAt(text2.length - 1)) {
557     return 0;
558   }
559   // Binary search.
560   // Performance analysis: http://neil.fraser.name/news/2007/10/09/
561   var pointermin = 0;
562   var pointermax = Math.min(text1.length, text2.length);
563   var pointermid = pointermax;
564   var pointerend = 0;
565   while (pointermin < pointermid) {
566     if (text1.substring(text1.length - pointermid, text1.length - pointerend) ==
567         text2.substring(text2.length - pointermid, text2.length - pointerend)) {
568       pointermin = pointermid;
569       pointerend = pointermin;
570     } else {
571       pointermax = pointermid;
572     }
573     pointermid = Math.floor((pointermax - pointermin) / 2 + pointermin);
574   }
575   return pointermid;
576 };

579 /**
580  * Determine if the suffix of one string is the prefix of another.
581  * @param {string} text1 First string.
582  * @param {string} text2 Second string.
583  * @return {number} The number of characters common to the end of the first
584  *     string and the start of the second string.
585  * @private
586  */
587 diff_match_patch.prototype.diff_commonOverlap_ = function(text1, text2) {
588   // Cache the text lengths to prevent multiple calls.
589   var text1_length = text1.length;
590   var text2_length = text2.length;
591   // Eliminate the null case.
592   if (text1_length == 0 || text2_length == 0) {
593     return 0;
594   }
595   // Truncate the longer string.
596   if (text1_length > text2_length) {
597     text1 = text1.substring(text1_length - text2_length);
598   } else if (text1_length < text2_length) {
599     text2 = text2.substring(0, text1_length);
600   }
601   var text_length = Math.min(text1_length, text2_length);
602   // Quick check for the worst case.
603   if (text1 == text2) {
604     return text_length;
605   }
606 
607   // Start by looking for a single character match
608   // and increase length until no match is found.
609   // Performance analysis: http://neil.fraser.name/news/2010/11/04/
610   var best = 0;
611   var length = 1;
612   while (true) {
613     var pattern = text1.substring(text_length - length);
614     var found = text2.indexOf(pattern);
615     if (found == -1) {
616       return best;
617     }
618     length += found;
619     if (found == 0 || text1.substring(text_length - length) ==
620         text2.substring(0, length)) {
621       best = length;
622       length++;
623     }

628 /**
629  * Do the two texts share a substring which is at least half the length of the
630  * longer text?
631  * This speedup can produce non-minimal diffs.
632  * @param {string} text1 First string.
633  * @param {string} text2 Second string.
634  * @return {Array.<string>} Five element Array, containing the prefix of
635  *     text1, the suffix of text1, the prefix of text2, the suffix of
636  *     text2 and the common middle.  Or null if there was no match.
637  * @private
638  */
639 diff_match_patch.prototype.diff_halfMatch_ = function(text1, text2) {
640   if (this.Diff_Timeout <= 0) {
641     // Don't risk returning a non-optimal diff if we have unlimited time.
642     return null;
643   }
644   var longtext = text1.length > text2.length ? text1 : text2;
645   var shorttext = text1.length > text2.length ? text2 : text1;
646   if (longtext.length < 4 || shorttext.length * 2 < longtext.length) {
647     return null;  // Pointless.
648   }
649   var dmp = this;  // 'this' becomes 'window' in a closure.
650 
651   /**
652    * Does a substring of shorttext exist within longtext such that the substring
653    * is at least half the length of longtext?
654    * Closure, but does not reference any external variables.
655    * @param {string} longtext Longer string.
656    * @param {string} shorttext Shorter string.
657    * @param {number} i Start index of quarter length substring within longtext.
658    * @return {Array.<string>} Five element Array, containing the prefix of
659    *     longtext, the suffix of longtext, the prefix of shorttext, the suffix
660    *     of shorttext and the common middle.  Or null if there was no match.
661    * @private
662    */
663   function diff_halfMatchI_(longtext, shorttext, i) {
664     // Start with a 1/4 length substring at position i as a seed.
665     var seed = longtext.substring(i, i + Math.floor(longtext.length / 4));
666     var j = -1;
667     var best_common = '';
668     var best_longtext_a, best_longtext_b, best_shorttext_a, best_shorttext_b;
669     while ((j = shorttext.indexOf(seed, j + 1)) != -1) {
670       var prefixLength = dmp.diff_commonPrefix(longtext.substring(i),
671                                                shorttext.substring(j));
672       var suffixLength = dmp.diff_commonSuffix(longtext.substring(0, i),
673                                                shorttext.substring(0, j));
674       if (best_common.length < suffixLength + prefixLength) {
675         best_common = shorttext.substring(j - suffixLength, j) +
676             shorttext.substring(j, j + prefixLength);
677         best_longtext_a = longtext.substring(0, i - suffixLength);
678         best_longtext_b = longtext.substring(i + prefixLength);
679         best_shorttext_a = shorttext.substring(0, j - suffixLength);
680         best_shorttext_b = shorttext.substring(j + prefixLength);
681       }
682     }
683     if (best_common.length * 2 >= longtext.length) {
684       return [best_longtext_a, best_longtext_b,
685               best_shorttext_a, best_shorttext_b, best_common];
686     } else {
687       return null;
688     }
689   }
690 
691   // First check if the second quarter is the seed for a half-match.
692   var hm1 = diff_halfMatchI_(longtext, shorttext,
693                              Math.ceil(longtext.length / 4));
694   // Check again based on the third quarter.
695   var hm2 = diff_halfMatchI_(longtext, shorttext,
696                              Math.ceil(longtext.length / 2));
697   var hm;
698   if (!hm1 && !hm2) {
699     return null;
700   } else if (!hm2) {
701     hm = hm1;
702   } else if (!hm1) {
703     hm = hm2;
704   } else {
705     // Both matched.  Select the longest.
706     hm = hm1[4].length > hm2[4].length ? hm1 : hm2;
707   }
708 
709   // A half-match was found, sort out the return data.
710   var text1_a, text1_b, text2_a, text2_b;
711   if (text1.length > text2.length) {
712     text1_a = hm[0];
713     text1_b = hm[1];
714     text2_a = hm[2];
715     text2_b = hm[3];
716   } else {
717     text2_a = hm[0];
718     text2_b = hm[1];
719     text1_a = hm[2];
720     text1_b = hm[3];
721   }
722   var mid_common = hm[4];
723   return [text1_a, text1_b, text2_a, text2_b, mid_common];
724 };

727 /**
728  * Reduce the number of edits by eliminating semantically trivial equalities.
729  * @param {!Array.<!diff_match_patch.Diff>} diffs Array of diff tuples.
730  */
731 diff_match_patch.prototype.diff_cleanupSemantic = function(diffs) {
732   var changes = false;
733   var equalities = [];  // Stack of indices where equalities are found.
734   var equalitiesLength = 0;  // Keeping our own length var is faster in JS.
735   /** @type {?string} */
736   var lastequality = null;  // Always equal to equalities[equalitiesLength-1][1]
737   var pointer = 0;  // Index of current position.
738   // Number of characters that changed prior to the equality.
739   var length_insertions1 = 0;
740 741   var length_deletions1 = 0;
742   // Number of characters that changed after the equality.
743   var length_insertions2 = 0;
744   var length_deletions2 = 0;
745   while (pointer < diffs.length) {
746     if (diffs[pointer][0] == DIFF_EQUAL) {  // Equality found.
747       equalities[equalitiesLength++] = pointer;
748       length_insertions1 = length_insertions2;
749       length_deletions1 = length_deletions2;
750       length_insertions2 = 0;
751       length_deletions2 = 0;
752       lastequality = /** @type {string} */(diffs[pointer][1]);
753     } else {  // An insertion or deletion.
754       if (diffs[pointer][0] == DIFF_INSERT) {
755         length_insertions2 += diffs[pointer][1].length;
756       } else {
757         length_deletions2 += diffs[pointer][1].length;
758       }
759       if (lastequality !== null && (lastequality.length <=
760           Math.max(length_insertions1, length_deletions1)) &&
761           (lastequality.length <= Math.max(length_insertions2,
762                                            length_deletions2))) {
763         // Duplicate record.
764         diffs.splice(equalities[equalitiesLength - 1], 0,
765                      [DIFF_DELETE, lastequality]);
766         // Change second copy to insert.
767 768         diffs[equalities[equalitiesLength - 1] + 1][0] = DIFF_INSERT;
769         // Throw away the equality we just deleted.
770         equalitiesLength--;
771         // Throw away the previous equality (it needs to be reevaluated).
772         equalitiesLength--;
773         pointer = equalitiesLength > 0 ? equalities[equalitiesLength - 1] : -1;
774         length_insertions1 = 0;  // Reset the counters.
775         length_deletions1 = 0;
776         length_insertions2 = 0;
777         length_deletions2 = 0;
778         lastequality = null;
779         changes = true;
780       }
781     }
782     pointer++;
783   }
784 
785   // Normalize the diff.
786   if (changes) {
787     this.diff_cleanupMerge(diffs);
788   }
789   this.diff_cleanupSemanticLossless(diffs);
790 
791   // Find any overlaps between deletions and insertions.
792   // e.g: <del>abcxx</del><ins>xxdef</ins>
793   //   -> <del>abc</del>xx<ins>def</ins>
794   pointer = 1;
795   while (pointer < diffs.length) {
796     if (diffs[pointer - 1][0] == DIFF_DELETE &&
797         diffs[pointer][0] == DIFF_INSERT) {
798       var deletion = /** @type {string} */(diffs[pointer - 1][1]);
799       var insertion = /** @type {string} */(diffs[pointer][1]);
800       var overlap_length = this.diff_commonOverlap_(deletion, insertion);
801       if (overlap_length) {
802 803         // Overlap found.  Insert an equality and trim the surrounding edits.
804         diffs.splice(pointer, 0,
805             [DIFF_EQUAL, insertion.substring(0, overlap_length)]);
806         diffs[pointer - 1][1] =
807             deletion.substring(0, deletion.length - overlap_length);
808         diffs[pointer + 1][1] = insertion.substring(overlap_length);
809         pointer++;
810       }
811       pointer++;
812     }
813     pointer++;
814   }
815 };

818 /**
819  * Look for single edits surrounded on both sides by equalities
820  * which can be shifted sideways to align the edit to a word boundary.
821  * e.g: The c<ins>at c</ins>ame. -> The <ins>cat </ins>came.
822  * @param {!Array.<!diff_match_patch.Diff>} diffs Array of diff tuples.
823  */
824 diff_match_patch.prototype.diff_cleanupSemanticLossless = function(diffs) {
825   // Define some regex patterns for matching boundaries.
826   var punctuation = /[^a-zA-Z0-9]/;
827   var whitespace = /\s/;
828   var linebreak = /[\r\n]/;
829   var blanklineEnd = /\n\r?\n$/;
830   var blanklineStart = /^\r?\n\r?\n/;
831 
832   /**
833    * Given two strings, compute a score representing whether the internal
834    * boundary falls on logical boundaries.
835    * Scores range from 5 (best) to 0 (worst).
836    * Closure, makes reference to regex patterns defined above.
837    * @param {string} one First string.
838    * @param {string} two Second string.
839    * @return {number} The score.
840    * @private
841    */
842   function diff_cleanupSemanticScore_(one, two) {
843     if (!one || !two) {
844       // Edges are the best.
845       return 5;
846     }
847 
848     // Each port of this function behaves slightly differently due to
849     // subtle differences in each language's definition of things like
850     // 'whitespace'.  Since this function's purpose is largely cosmetic,
851     // the choice has been made to use each language's native features
852     // rather than force total conformity.
853     var score = 0;
854     // One point for non-alphanumeric.
855     if (one.charAt(one.length - 1).match(punctuation) ||
856         two.charAt(0).match(punctuation)) {
857       score++;
858       // Two points for whitespace.
859       if (one.charAt(one.length - 1).match(whitespace) ||
860           two.charAt(0).match(whitespace)) {
861         score++;
862         // Three points for line breaks.
863         if (one.charAt(one.length - 1).match(linebreak) ||
864             two.charAt(0).match(linebreak)) {
865           score++;
866           // Four points for blank lines.
867           if (one.match(blanklineEnd) || two.match(blanklineStart)) {
868             score++;
869           }
870         }
871       }
872     }
873     return score;
874   }
875 
876   var pointer = 1;
877   // Intentionally ignore the first and last element (don't need checking).
878   while (pointer < diffs.length - 1) {
879     if (diffs[pointer - 1][0] == DIFF_EQUAL &&
880         diffs[pointer + 1][0] == DIFF_EQUAL) {
881       // This is a single edit surrounded by equalities.
882       var equality1 = /** @type {string} */(diffs[pointer - 1][1]);
883       var edit = /** @type {string} */(diffs[pointer][1]);
884       var equality2 = /** @type {string} */(diffs[pointer + 1][1]);
885 
886       // First, shift the edit as far left as possible.
887       var commonOffset = this.diff_commonSuffix(equality1, edit);
888       if (commonOffset) {
889         var commonString = edit.substring(edit.length - commonOffset);
890         equality1 = equality1.substring(0, equality1.length - commonOffset);
891         edit = commonString + edit.substring(0, edit.length - commonOffset);
892         equality2 = commonString + equality2;
893       }
894 
895       // Second, step character by character right, looking for the best fit.
896       var bestEquality1 = equality1;
897       var bestEdit = edit;
898       var bestEquality2 = equality2;
899       var bestScore = diff_cleanupSemanticScore_(equality1, edit) +
900           diff_cleanupSemanticScore_(edit, equality2);
901       while (edit.charAt(0) === equality2.charAt(0)) {
902         equality1 += edit.charAt(0);
903         edit = edit.substring(1) + equality2.charAt(0);
904         equality2 = equality2.substring(1);
905         var score = diff_cleanupSemanticScore_(equality1, edit) +
906             diff_cleanupSemanticScore_(edit, equality2);
907         // The >= encourages trailing rather than leading whitespace on edits.
908         if (score >= bestScore) {
909           bestScore = score;
910           bestEquality1 = equality1;
911           bestEdit = edit;
912           bestEquality2 = equality2;
913         }
914       }
915 
916       if (diffs[pointer - 1][1] != bestEquality1) {
917         // We have an improvement, save it back to the diff.
918         if (bestEquality1) {
919           diffs[pointer - 1][1] = bestEquality1;
920         } else {
921           diffs.splice(pointer - 1, 1);
922           pointer--;
923         }
924         diffs[pointer][1] = bestEdit;
925         if (bestEquality2) {
926           diffs[pointer + 1][1] = bestEquality2;
927         } else {
928           diffs.splice(pointer + 1, 1);
929           pointer--;
930         }
931       }
932     }
933     pointer++;
934   }
935 };

938 /**
939  * Reduce the number of edits by eliminating operationally trivial equalities.
940  * @param {!Array.<!diff_match_patch.Diff>} diffs Array of diff tuples.
941  */
942 diff_match_patch.prototype.diff_cleanupEfficiency = function(diffs) {
943   var changes = false;
944   var equalities = [];  // Stack of indices where equalities are found.
945   var equalitiesLength = 0;  // Keeping our own length var is faster in JS.
946   var lastequality = '';  // Always equal to equalities[equalitiesLength-1][1]
947   var pointer = 0;  // Index of current position.
948   // Is there an insertion operation before the last equality.
949   var pre_ins = false;
950   // Is there a deletion operation before the last equality.
951   var pre_del = false;
952   // Is there an insertion operation after the last equality.
953   var post_ins = false;
954   // Is there a deletion operation after the last equality.
955   var post_del = false;
956   while (pointer < diffs.length) {
957     if (diffs[pointer][0] == DIFF_EQUAL) {  // Equality found.
958       if (diffs[pointer][1].length < this.Diff_EditCost &&
959           (post_ins || post_del)) {
960         // Candidate found.
961         equalities[equalitiesLength++] = pointer;
962         pre_ins = post_ins;
963         pre_del = post_del;
964         lastequality = diffs[pointer][1];
965       } else {
966         // Not a candidate, and can never become one.
967         equalitiesLength = 0;
968         lastequality = '';
969       }
970       post_ins = post_del = false;
971     } else {  // An insertion or deletion.
972       if (diffs[pointer][0] == DIFF_DELETE) {
973         post_del = true;
974       } else {
975         post_ins = true;
976       }
977       /*
978        * Five types to be split:
979        * <ins>A</ins><del>B</del>XY<ins>C</ins><del>D</del>
980        * <ins>A</ins>X<ins>C</ins><del>D</del>
981        * <ins>A</ins><del>B</del>X<ins>C</ins>
982        * <ins>A</del>X<ins>C</ins><del>D</del>
983        * <ins>A</ins><del>B</del>X<del>C</del>
984        */
985       if (lastequality && ((pre_ins && pre_del && post_ins && post_del) ||
986                            ((lastequality.length < this.Diff_EditCost / 2) &&
987                             (pre_ins + pre_del + post_ins + post_del) == 3))) {
988         // Duplicate record.
989         diffs.splice(equalities[equalitiesLength - 1], 0,
990                      [DIFF_DELETE, lastequality]);
991         // Change second copy to insert.
992         diffs[equalities[equalitiesLength - 1] + 1][0] = DIFF_INSERT;
993         equalitiesLength--;  // Throw away the equality we just deleted;
994         lastequality = '';
995         if (pre_ins && pre_del) {
996           // No changes made which could affect previous entry, keep going.
997           post_ins = post_del = true;
998           equalitiesLength = 0;
999         } else {
1000           equalitiesLength--;  // Throw away the previous equality.
1001           pointer = equalitiesLength > 0 ?
1002               equalities[equalitiesLength - 1] : -1;
1003           post_ins = post_del = false;
1004         }
1005         changes = true;
1006       }
1007     }
1008     pointer++;
1009   }
1010 
1011   if (changes) {
1012     this.diff_cleanupMerge(diffs);
1013   }
1014 };

1017 /**
1018  * Reorder and merge like edit sections.  Merge equalities.
1019  * Any edit section can move as long as it doesn't cross an equality.
1020  * @param {!Array.<!diff_match_patch.Diff>} diffs Array of diff tuples.
1021  */
1022 diff_match_patch.prototype.diff_cleanupMerge = function(diffs) {
1023   diffs.push([DIFF_EQUAL, '']);  // Add a dummy entry at the end.
1024   var pointer = 0;
1025   var count_delete = 0;
1026   var count_insert = 0;
1027   var text_delete = '';
1028   var text_insert = '';
1029   var commonlength;
1030   while (pointer < diffs.length) {
1031     switch (diffs[pointer][0]) {
1032       case DIFF_INSERT:
1033         count_insert++;
1034         text_insert += diffs[pointer][1];
1035         pointer++;
1036         break;
1037       case DIFF_DELETE:
1038         count_delete++;
1039         text_delete += diffs[pointer][1];
1040         pointer++;
1041         break;
1042       case DIFF_EQUAL:
1043         // Upon reaching an equality, check for prior redundancies.
1044         if (count_delete + count_insert > 1) {
1045           if (count_delete !== 0 && count_insert !== 0) {
1046 1047             // Factor out any common prefixies.
1048             commonlength = this.diff_commonPrefix(text_insert, text_delete);
1049             if (commonlength !== 0) {
1050               if ((pointer - count_delete - count_insert) > 0 &&
1051                   diffs[pointer - count_delete - count_insert - 1][0] ==
1052                   DIFF_EQUAL) {
1053                 diffs[pointer - count_delete - count_insert - 1][1] +=
1054                     text_insert.substring(0, commonlength);
1055               } else {
1056                 diffs.splice(0, 0, [DIFF_EQUAL,
1057                                     text_insert.substring(0, commonlength)]);
1058                 pointer++;
1059               }
1060               text_insert = text_insert.substring(commonlength);
1061               text_delete = text_delete.substring(commonlength);
1062             }
1063             // Factor out any common suffixies.
1064             commonlength = this.diff_commonSuffix(text_insert, text_delete);
1065             if (commonlength !== 0) {
1066               diffs[pointer][1] = text_insert.substring(text_insert.length -
1067                   commonlength) + diffs[pointer][1];
1068               text_insert = text_insert.substring(0, text_insert.length -
1069                   commonlength);
1070               text_delete = text_delete.substring(0, text_delete.length -
1071                   commonlength);
1072             }
1073           }
1074           // Delete the offending records and add the merged ones.
1075           if (count_delete === 0) {
1076             diffs.splice(pointer - count_delete - count_insert,
1077                 count_delete + count_insert, [DIFF_INSERT, text_insert]);
1078           } else if (count_insert === 0) {
1079             diffs.splice(pointer - count_delete - count_insert,
1080                 count_delete + count_insert, [DIFF_DELETE, text_delete]);
1081           } else {
1082             diffs.splice(pointer - count_delete - count_insert,
1083                 count_delete + count_insert, [DIFF_DELETE, text_delete],
1084                 [DIFF_INSERT, text_insert]);
1085           }
1086           pointer = pointer - count_delete - count_insert +
1087                     (count_delete ? 1 : 0) + (count_insert ? 1 : 0) + 1;
1088         } else if (pointer !== 0 && diffs[pointer - 1][0] == DIFF_EQUAL) {
1089           // Merge this equality with the previous one.
1090           diffs[pointer - 1][1] += diffs[pointer][1];
1091           diffs.splice(pointer, 1);
1092         } else {
1093           pointer++;
1094         }
1095         count_insert = 0;
1096         count_delete = 0;
1097         text_delete = '';
1098         text_insert = '';
1099         break;
1100     }
1101   }
1102   if (diffs[diffs.length - 1][1] === '') {
1103     diffs.pop();  // Remove the dummy entry at the end.
1104   }
1105 
1106   // Second pass: look for single edits surrounded on both sides by equalities
1107   // which can be shifted sideways to eliminate an equality.
1108   // e.g: A<ins>BA</ins>C -> <ins>AB</ins>AC
1109   var changes = false;
1110   pointer = 1;
1111   // Intentionally ignore the first and last element (don't need checking).
1112   while (pointer < diffs.length - 1) {
1113     if (diffs[pointer - 1][0] == DIFF_EQUAL &&
1114         diffs[pointer + 1][0] == DIFF_EQUAL) {
1115       // This is a single edit surrounded by equalities.
1116       if (diffs[pointer][1].substring(diffs[pointer][1].length -
1117           diffs[pointer - 1][1].length) == diffs[pointer - 1][1]) {
1118         // Shift the edit over the previous equality.
1119         diffs[pointer][1] = diffs[pointer - 1][1] +
1120             diffs[pointer][1].substring(0, diffs[pointer][1].length -
1121                                         diffs[pointer - 1][1].length);
1122         diffs[pointer + 1][1] = diffs[pointer - 1][1] + diffs[pointer + 1][1];
1123         diffs.splice(pointer - 1, 1);
1124         changes = true;
1125       } else if (diffs[pointer][1].substring(0, diffs[pointer + 1][1].length) ==
1126           diffs[pointer + 1][1]) {
1127         // Shift the edit over the next equality.
1128         diffs[pointer - 1][1] += diffs[pointer + 1][1];
1129         diffs[pointer][1] =
1130             diffs[pointer][1].substring(diffs[pointer + 1][1].length) +
1131             diffs[pointer + 1][1];
1132         diffs.splice(pointer + 1, 1);
1133         changes = true;
1134       }
1135     }
1136     pointer++;
1137   }
1138   // If shifts were made, the diff needs reordering and another shift sweep.
1139   if (changes) {
1140     this.diff_cleanupMerge(diffs);
1141   }
1142 };

1145 /**
1146  * loc is a location in text1, compute and return the equivalent location in
1147  * text2.
1148  * e.g. 'The cat' vs 'The big cat', 1->1, 5->8
1149  * @param {!Array.<!diff_match_patch.Diff>} diffs Array of diff tuples.
1150  * @param {number} loc Location within text1.
1151  * @return {number} Location within text2.
1152  */
1153 diff_match_patch.prototype.diff_xIndex = function(diffs, loc) {
1154   var chars1 = 0;
1155   var chars2 = 0;
1156   var last_chars1 = 0;
1157   var last_chars2 = 0;
1158   var x;
1159   for (x = 0; x < diffs.length; x++) {
1160     if (diffs[x][0] !== DIFF_INSERT) {  // Equality or deletion.
1161       chars1 += diffs[x][1].length;
1162     }
1163     if (diffs[x][0] !== DIFF_DELETE) {  // Equality or insertion.
1164       chars2 += diffs[x][1].length;
1165     }
1166     if (chars1 > loc) {  // Overshot the location.
1167       break;
1168     }
1169     last_chars1 = chars1;
1170     last_chars2 = chars2;
1171   }
1172   // Was the location was deleted?
1173   if (diffs.length != x && diffs[x][0] === DIFF_DELETE) {
1174     return last_chars2;
1175   }
1176   // Add the remaining character length.
1177   return last_chars2 + (loc - last_chars1);
1178 };

1181 /**
1182  * Convert a diff array into a pretty HTML report.
1183  * @param {!Array.<!diff_match_patch.Diff>} diffs Array of diff tuples.
1184  * @return {string} HTML representation.
1185  */
1186 diff_match_patch.prototype.diff_prettyHtml = function(diffs) {
1187   var html = [];
1188   var i = 0;
1189   var pattern_amp = /&/g;
1190   var pattern_lt = /</g;
1191   var pattern_gt = />/g;
1192   var pattern_para = /\n/g;
1193   for (var x = 0; x < diffs.length; x++) {
1194     var op = diffs[x][0];    // Operation (insert, delete, equal)
1195     var data = diffs[x][1];  // Text of change.
1196     var text = data.replace(pattern_amp, '&').replace(pattern_lt, '<')
1197         .replace(pattern_gt, '>').replace(pattern_para, '¶<br>');

1199       case DIFF_INSERT:

1251  * substituted characters.
1252  * @param {!Array.<!diff_match_patch.Diff>} diffs Array of diff tuples.

1256   var levenshtein = 0;
1257   var insertions = 0;
1258   var deletions = 0;
1259   for (var x = 0; x < diffs.length; x++) {
1260     var op = diffs[x][0];
1261     var data = diffs[x][1];

1265         break;

1273         deletions = 0;

1309 /**
1310  * Given the original text1, and an encoded string which describes the
1311  * operations required to transform text1 into text2, compute the full diff.
1312  * @param {string} text1 Source string for the diff.
1313  * @param {string} delta Delta text.
1314  * @return {!Array.<!diff_match_patch.Diff>} Array of diff tuples.
1315  * @throws {!Error} If invalid input.
1316  */
1317 diff_match_patch.prototype.diff_fromDelta = function(text1, delta) {
1318   var diffs = [];
1319   var diffsLength = 0;  // Keeping our own length var is faster in JS.
1320   var pointer = 0;  // Cursor in text1
1321   var tokens = delta.split(/\t/g);
1322   for (var x = 0; x < tokens.length; x++) {
1323     // Each token begins with a one character parameter which specifies the
1324     // operation of this token (delete, insert, equality).
1325     var param = tokens[x].substring(1);
1326     switch (tokens[x].charAt(0)) {
1327       case '+':
1328         try {
1329           diffs[diffsLength++] = [DIFF_INSERT, decodeURI(param)];
1330         } catch (ex) {
1331           // Malformed URI sequence.
1332           throw new Error('Illegal escape in diff_fromDelta: ' + param);
1333         }
1334         break;
1335       case '-':
1336         // Fall through.
1337       case '=':
1338         var n = parseInt(param, 10);
1339         if (isNaN(n) || n < 0) {
1340           throw new Error('Invalid number in diff_fromDelta: ' + param);
1341         }
1342         var text = text1.substring(pointer, pointer += n);
1343         if (tokens[x].charAt(0) == '=') {
1344           diffs[diffsLength++] = [DIFF_EQUAL, text];
1345         } else {
1346           diffs[diffsLength++] = [DIFF_DELETE, text];
1347         }
1348         break;
1349       default:
1350         // Blank tokens are ok (from a trailing \t).
1351         // Anything else is an error.
1352         if (tokens[x]) {
1353           throw new Error('Invalid diff operation in diff_fromDelta: ' +
1354                           tokens[x]);
1355         }
1356     }
1357   }
1358   if (pointer != text1.length) {
1359     throw new Error('Delta length (' + pointer +
1360         ') does not equal source text length (' + text1.length + ').');
1361   }
1362   return diffs;
1363 };

1369 /**
1370  * Locate the best instance of 'pattern' in 'text' near 'loc'.
1371  * @param {string} text The text to search.
1372  * @param {string} pattern The pattern to search for.
1373  * @param {number} loc The location to search around.
1374  * @return {number} Best match index or -1.
1375  */
1376 diff_match_patch.prototype.match_main = function(text, pattern, loc) {
1377   // Check for null inputs.
1378   if (text == null || pattern == null || loc == null) {
1379     throw new Error('Null input. (match_main)');
1380   }
1381 
1382   loc = Math.max(0, Math.min(loc, text.length));
1383   if (text == pattern) {
1384     // Shortcut (potentially not guaranteed by the algorithm)
1385     return 0;
1386   } else if (!text.length) {
1387     // Nothing to match.
1388     return -1;
1389   } else if (text.substring(loc, loc + pattern.length) == pattern) {
1390     // Perfect match at the perfect spot!  (Includes case of null pattern)
1391     return loc;
1392   } else {
1393     // Do a fuzzy compare.
1394     return this.match_bitap_(text, pattern, loc);
1395   }
1396 };

1399 /**
1400  * Locate the best instance of 'pattern' in 'text' near 'loc' using the
1401  * Bitap algorithm.
1402  * @param {string} text The text to search.
1403  * @param {string} pattern The pattern to search for.
1404  * @param {number} loc The location to search around.
1405  * @return {number} Best match index or -1.
1406  * @private
1407  */
1408 diff_match_patch.prototype.match_bitap_ = function(text, pattern, loc) {
1409   if (pattern.length > this.Match_MaxBits) {
1410     throw new Error('Pattern too long for this browser.');
1411   }
1412 
1413   // Initialise the alphabet.
1414   var s = this.match_alphabet_(pattern);
1415 
1416   var dmp = this;  // 'this' becomes 'window' in a closure.
1417 
1418   /**
1419    * Compute and return the score for a match with e errors and x location.
1420    * Accesses loc and pattern through being a closure.
1421    * @param {number} e Number of errors in match.
1422    * @param {number} x Location of match.
1423    * @return {number} Overall score for match (0.0 = good, 1.0 = bad).
1424    * @private
1425    */
1426   function match_bitapScore_(e, x) {
1427     var accuracy = e / pattern.length;
1428     var proximity = Math.abs(loc - x);
1429     if (!dmp.Match_Distance) {
1430       // Dodge divide by zero error.
1431       return proximity ? 1.0 : accuracy;
1432     }
1433     return accuracy + (proximity / dmp.Match_Distance);
1434   }
1435 
1436   // Highest score beyond which we give up.
1437   var score_threshold = this.Match_Threshold;
1438   // Is there a nearby exact match? (speedup)
1439   var best_loc = text.indexOf(pattern, loc);
1440   if (best_loc != -1) {
1441     score_threshold = Math.min(match_bitapScore_(0, best_loc), score_threshold);
1442     // What about in the other direction? (speedup)
1443     best_loc = text.lastIndexOf(pattern, loc + pattern.length);
1444     if (best_loc != -1) {
1445       score_threshold =
1446           Math.min(match_bitapScore_(0, best_loc), score_threshold);
1447     }
1448   }
1449 
1450   // Initialise the bit arrays.
1451   var matchmask = 1 << (pattern.length - 1);
1452   best_loc = -1;
1453 
1454   var bin_min, bin_mid;
1455   var bin_max = pattern.length + text.length;
1456   var last_rd;
1457   for (var d = 0; d < pattern.length; d++) {
1458     // Scan for the best match; each iteration allows for one more error.
1459     // Run a binary search to determine how far from 'loc' we can stray at this
1460     // error level.
1461     bin_min = 0;
1462     bin_mid = bin_max;
1463     while (bin_min < bin_mid) {
1464       if (match_bitapScore_(d, loc + bin_mid) <= score_threshold) {
1465         bin_min = bin_mid;
1466       } else {
1467         bin_max = bin_mid;
1468       }
1469       bin_mid = Math.floor((bin_max - bin_min) / 2 + bin_min);
1470     }
1471     // Use the result from this iteration as the maximum for the next.
1472     bin_max = bin_mid;
1473     var start = Math.max(1, loc - bin_mid + 1);
1474     var finish = Math.min(loc + bin_mid, text.length) + pattern.length;
1475 
1476     var rd = Array(finish + 2);
1477     rd[finish + 1] = (1 << d) - 1;
1478     for (var j = finish; j >= start; j--) {
1479       // The alphabet (s) is a sparse hash, so the following line generates
1480       // warnings.
1481       var charMatch = s[text.charAt(j - 1)];
1482       if (d === 0) {  // First pass: exact match.
1483         rd[j] = ((rd[j + 1] << 1) | 1) & charMatch;
1484       } else {  // Subsequent passes: fuzzy match.
1485         rd[j] = ((rd[j + 1] << 1) | 1) & charMatch |
1486                 (((last_rd[j + 1] | last_rd[j]) << 1) | 1) |
1487                 last_rd[j + 1];
1488       }
1489       if (rd[j] & matchmask) {
1490         var score = match_bitapScore_(d, j - 1);
1491         // This match will almost certainly be better than any existing match.
1492         // But check anyway.
1493         if (score <= score_threshold) {
1494           // Told you so.
1495           score_threshold = score;
1496           best_loc = j - 1;
1497           if (best_loc > loc) {
1498             // When passing loc, don't exceed our current distance from loc.
1499             start = Math.max(1, 2 * loc - best_loc);
1500           } else {
1501             // Already passed loc, downhill from here on in.
1502             break;
1503           }
1504         }
1505       }
1506     }
1507     // No hope for a (better) match at greater error levels.
1508     if (match_bitapScore_(d + 1, loc) > score_threshold) {
1509       break;
1510     }
1511     last_rd = rd;
1512   }
1513   return best_loc;
1514 };

1538 /**
1539  * Increase the context until it is unique,
1540  * but don't let the pattern expand beyond Match_MaxBits.
1541  * @param {!patch_obj} patch The patch to grow.
1542  * @param {string} text Source text.
1543  * @private
1544  */
1545 diff_match_patch.prototype.patch_addContext_ = function(patch, text) {
1546   if (text.length == 0) {
1547     return;
1548   }
1549   var pattern = text.substring(patch.start2, patch.start2 + patch.length1);
1550   var padding = 0;
1551 
1552   // Look for the first and last matches of pattern in text.  If two different
1553   // matches are found, increase the pattern length.
1554   while (text.indexOf(pattern) != text.lastIndexOf(pattern) &&
1555          pattern.length < this.Match_MaxBits - this.Patch_Margin -
1556          this.Patch_Margin) {
1557     padding += this.Patch_Margin;
1558     pattern = text.substring(patch.start2 - padding,
1559                              patch.start2 + patch.length1 + padding);
1560   }
1561   // Add one chunk for good luck.
1562   padding += this.Patch_Margin;
1563 
1564   // Add the prefix.
1565   var prefix = text.substring(patch.start2 - padding, patch.start2);
1566   if (prefix) {
1567     patch.diffs.unshift([DIFF_EQUAL, prefix]);
1568   }
1569   // Add the suffix.
1570   var suffix = text.substring(patch.start2 + patch.length1,
1571                               patch.start2 + patch.length1 + padding);
1572   if (suffix) {
1573     patch.diffs.push([DIFF_EQUAL, suffix]);
1574   }
1575 1576 
  // Roll back the start points.
1577   patch.start1 -= prefix.length;
1578   patch.start2 -= prefix.length;
1579   // Extend the lengths.
1580   patch.length1 += prefix.length + suffix.length;
1581   patch.length2 += prefix.length + suffix.length;
1582 };

1585 /**
1586  * Compute a list of patches to turn text1 into text2.
1587  * Use diffs if provided, otherwise compute it ourselves.
1588  * There are four ways to call this function, depending on what data is
1589  * available to the caller:
1590  * Method 1:
1591  * a = text1, b = text2
1592  * Method 2:
1593  * a = diffs
1594  * Method 3 (optimal):
1595  * a = text1, b = diffs
1596  * Method 4 (deprecated, use method 3):
1597  * a = text1, b = text2, c = diffs
1598  *
1599  * @param {string|!Array.<!diff_match_patch.Diff>} a text1 (methods 1,3,4) or
1600  * Array of diff tuples for text1 to text2 (method 2).
1601  * @param {string|!Array.<!diff_match_patch.Diff>} opt_b text2 (methods 1,4) or
1602  * Array of diff tuples for text1 to text2 (method 3) or undefined (method 2).
1603  * @param {string|!Array.<!diff_match_patch.Diff>} opt_c Array of diff tuples
1604  * for text1 to text2 (method 4) or undefined (methods 1,2,3).
1605  * @return {!Array.<!patch_obj>} Array of patch objects.
1606  */
1607 diff_match_patch.prototype.patch_make = function(a, opt_b, opt_c) {
1608   var text1, diffs;
1609   if (typeof a == 'string' && typeof opt_b == 'string' &&
1610       typeof opt_c == 'undefined') {
1611     // Method 1: text1, text2
1612     // Compute diffs from text1 and text2.
1613     text1 = /** @type {string} */(a);
1614     diffs = this.diff_main(text1, /** @type {string} */(opt_b), true);
1615     if (diffs.length > 2) {
1616       this.diff_cleanupSemantic(diffs);
1617       this.diff_cleanupEfficiency(diffs);
1618     }
1619   } else if (a && typeof a == 'object' && typeof opt_b == 'undefined' &&
1620       typeof opt_c == 'undefined') {
1621     // Method 2: diffs
1622     // Compute text1 from diffs.
1623     diffs = /** @type {!Array.<!diff_match_patch.Diff>} */(a);
1624     text1 = this.diff_text1(diffs);
1625   } else if (typeof a == 'string' && opt_b && typeof opt_b == 'object' &&
1626       typeof opt_c == 'undefined') {
1627     // Method 3: text1, diffs
1628     text1 = /** @type {string} */(a);
1629     diffs = /** @type {!Array.<!diff_match_patch.Diff>} */(opt_b);
1630   } else if (typeof a == 'string' && typeof opt_b == 'string' &&
1631       opt_c && typeof opt_c == 'object') {
1632     // Method 4: text1, text2, diffs
1633     // text2 is not used.
1634     text1 = /** @type {string} */(a);
1635     diffs = /** @type {!Array.<!diff_match_patch.Diff>} */(opt_c);
1636   } else {
1637     throw new Error('Unknown call format to patch_make.');
1638   }
1639 
1640   if (diffs.length === 0) {
1641     return [];  // Get rid of the null case.
1642   }
1643   var patches = [];
1644   var patch = new patch_obj();
1645   var patchDiffLength = 0;  // Keeping our own length var is faster in JS.
1646   var char_count1 = 0;  // Number of characters into the text1 string.
1647   var char_count2 = 0;  // Number of characters into the text2 string.
1648   // Start with text1 (prepatch_text) and apply the diffs until we arrive at
1649   // text2 (postpatch_text).  We recreate the patches one by one to determine
1650   // context info.
1651   var prepatch_text = text1;
1652   var postpatch_text = text1;
1653   for (var x = 0; x < diffs.length; x++) {
1654     var diff_type = diffs[x][0];
1655     var diff_text = diffs[x][1];
1656 
1657     if (!patchDiffLength && diff_type !== DIFF_EQUAL) {
1658       // A new patch starts here.
1659       patch.start1 = char_count1;
1660       patch.start2 = char_count2;
1661     }
1662 
1663     switch (diff_type) {
1664       case DIFF_INSERT:
1665         patch.diffs[patchDiffLength++] = diffs[x];
1666         patch.length2 += diff_text.length;
1667         postpatch_text = postpatch_text.substring(0, char_count2) + diff_text +
1668                          postpatch_text.substring(char_count2);
1669         break;
1670       case DIFF_DELETE:
1671         patch.length1 += diff_text.length;
1672         patch.diffs[patchDiffLength++] = diffs[x];
1673         postpatch_text = postpatch_text.substring(0, char_count2) +
1674                          postpatch_text.substring(char_count2 +
1675                              diff_text.length);
1676         break;
1677       case DIFF_EQUAL:
1678         if (diff_text.length <= 2 * this.Patch_Margin &&
1679             patchDiffLength && diffs.length != x + 1) {
1680           // Small equality inside a patch.
1681           patch.diffs[patchDiffLength++] = diffs[x];
1682           patch.length1 += diff_text.length;
1683           patch.length2 += diff_text.length;
1684         } else if (diff_text.length >= 2 * this.Patch_Margin) {
1685           // Time for a new patch.
1686           if (patchDiffLength) {
1687             this.patch_addContext_(patch, prepatch_text);
1688             patches.push(patch);
1689             patch = new patch_obj();
1690             patchDiffLength = 0;
1691             // Unlike Unidiff, our patch lists have a rolling context.
1692             // http://code.google.com/p/google-diff-match-patch/wiki/Unidiff
1693             // Update prepatch text & pos to reflect the application of the
1694             // just completed patch.
1695             prepatch_text = postpatch_text;
1696             char_count1 = char_count2;
1697           }
1698         }
1699         break;
1700     }
1701 
1702     // Update the current character count.
1703     if (diff_type !== DIFF_INSERT) {
1704       char_count1 += diff_text.length;
1705     }
1706     if (diff_type !== DIFF_DELETE) {
1707       char_count2 += diff_text.length;
1708     }
1709   }
1710   // Pick up the leftover patch if not empty.
1711   if (patchDiffLength) {
1712     this.patch_addContext_(patch, prepatch_text);
1713     patches.push(patch);
1714   }
1715 
1716   return patches;
1717 };

1745 /**
1746  * Merge a set of patches onto the text.  Return a patched text, as well
1747 1748  * as a list of true/false values indicating which patches were applied.
1749  * @param {!Array.<!patch_obj>} patches Array of patch objects.
1750  * @param {string} text Old text.
1751  * @return {!Array.<string|!Array.<boolean>>} Two element Array, containing the
1752  *      new text and an array of boolean values.
1753  */
1754 diff_match_patch.prototype.patch_apply = function(patches, text) {
1755   if (patches.length == 0) {
1756     return [text, []];
1757   }
1758 
1759   // Deep copy the patches so that no changes are made to originals.
1760   patches = this.patch_deepCopy(patches);
1761 
1762   var nullPadding = this.patch_addPadding(patches);
1763   text = nullPadding + text + nullPadding;
1764 
1765   this.patch_splitMax(patches);
1766   // delta keeps track of the offset between the expected and actual location
1767   // of the previous patch.  If there are patches expected at positions 10 and
1768   // 20, but the first patch was found at 12, delta is 2 and the second patch
1769   // has an effective expected position of 22.
1770   var delta = 0;
1771   var results = [];
1772   for (var x = 0; x < patches.length; x++) {
1773     var expected_loc = patches[x].start2 + delta;
1774     var text1 = this.diff_text1(patches[x].diffs);
1775     var start_loc;
1776     var end_loc = -1;
1777     if (text1.length > this.Match_MaxBits) {
1778       // patch_splitMax will only provide an oversized pattern in the case of
1779       // a monster delete.
1780       start_loc = this.match_main(text, text1.substring(0, this.Match_MaxBits),
1781                                   expected_loc);
1782       if (start_loc != -1) {
1783         end_loc = this.match_main(text,
1784             text1.substring(text1.length - this.Match_MaxBits),
1785             expected_loc + text1.length - this.Match_MaxBits);
1786         if (end_loc == -1 || start_loc >= end_loc) {
1787           // Can't find valid trailing context.  Drop this patch.
1788           start_loc = -1;
1789         }
1790       }
1791     } else {
1792       start_loc = this.match_main(text, text1, expected_loc);
1793     }
1794     if (start_loc == -1) {
1795       // No match found.  :(
1796       results[x] = false;
1797       // Subtract the delta for this failed patch from subsequent patches.
1798       delta -= patches[x].length2 - patches[x].length1;
1799     } else {
1800       // Found a match.  :)
1801       results[x] = true;
1802       delta = start_loc - expected_loc;
1803       var text2;
1804       if (end_loc == -1) {
1805         text2 = text.substring(start_loc, start_loc + text1.length);
1806       } else {
1807         text2 = text.substring(start_loc, end_loc + this.Match_MaxBits);
1808       }
1809       if (text1 == text2) {
1810         // Perfect match, just shove the replacement text in.
1811         text = text.substring(0, start_loc) +
1812                this.diff_text2(patches[x].diffs) +
1813                text.substring(start_loc + text1.length);
1814       } else {
1815         // Imperfect match.  Run a diff to get a framework of equivalent
1816         // indices.
1817         var diffs = this.diff_main(text1, text2, false);
1818         if (text1.length > this.Match_MaxBits &&
1819             this.diff_levenshtein(diffs) / text1.length >
1820             this.Patch_DeleteThreshold) {
1821           // The end points match, but the content is unacceptably bad.
1822           results[x] = false;
1823         } else {
1824           this.diff_cleanupSemanticLossless(diffs);
1825           var index1 = 0;
1826           var index2;
1827           for (var y = 0; y < patches[x].diffs.length; y++) {
1828             var mod = patches[x].diffs[y];
1829             if (mod[0] !== DIFF_EQUAL) {
1830               index2 = this.diff_xIndex(diffs, index1);
1831             }
1832             if (mod[0] === DIFF_INSERT) {  // Insertion
1833               text = text.substring(0, start_loc + index2) + mod[1] +
1834                      text.substring(start_loc + index2);
1835             } else if (mod[0] === DIFF_DELETE) {  // Deletion
1836               text = text.substring(0, start_loc + index2) +
1837                      text.substring(start_loc + this.diff_xIndex(diffs,
1838                          index1 + mod[1].length));
1839             }
1840             if (mod[0] !== DIFF_DELETE) {
1841               index1 += mod[1].length;
1842             }
1843           }
1844         }
1845       }
1846     }
1847   }
1848   // Strip the padding off.
1849   text = text.substring(nullPadding.length, text.length - nullPadding.length);
1850   return [text, results];
1851 };

1854 /**
1855  * Add some padding on text start and end so that edges can match something.
1856  * Intended to be called only from within patch_apply.
1857  * @param {!Array.<!patch_obj>} patches Array of patch objects.
1858  * @return {string} The padding string added to each side.
1859  */
1860 diff_match_patch.prototype.patch_addPadding = function(patches) {
1861   var paddingLength = this.Patch_Margin;
1862   var nullPadding = '';
1863   for (var x = 1; x <= paddingLength; x++) {
1864     nullPadding += String.fromCharCode(x);
1865   }
1866 
1867   // Bump all the patches forward.
1868   for (var x = 0; x < patches.length; x++) {
1869     patches[x].start1 += paddingLength;
1870     patches[x].start2 += paddingLength;
1871   }
1872 
1873   // Add some padding on start of first diff.
1874   var patch = patches[0];
1875   var diffs = patch.diffs;
1876   if (diffs.length == 0 || diffs[0][0] != DIFF_EQUAL) {
1877     // Add nullPadding equality.
1878     diffs.unshift([DIFF_EQUAL, nullPadding]);
1879     patch.start1 -= paddingLength;  // Should be 0.
1880     patch.start2 -= paddingLength;  // Should be 0.
1881     patch.length1 += paddingLength;
1882     patch.length2 += paddingLength;
1883   } else if (paddingLength > diffs[0][1].length) {
1884     // Grow first equality.
1885     var extraLength = paddingLength - diffs[0][1].length;
1886     diffs[0][1] = nullPadding.substring(diffs[0][1].length) + diffs[0][1];
1887     patch.start1 -= extraLength;
1888     patch.start2 -= extraLength;
1889     patch.length1 += extraLength;
1890     patch.length2 += extraLength;
1891   }
1892 
1893   // Add some padding on end of last diff.
1894   patch = patches[patches.length - 1];
1895   diffs = patch.diffs;
1896   if (diffs.length == 0 || diffs[diffs.length - 1][0] != DIFF_EQUAL) {
1897     // Add nullPadding equality.
1898     diffs.push([DIFF_EQUAL, nullPadding]);
1899     patch.length1 += paddingLength;
1900     patch.length2 += paddingLength;
1901   } else if (paddingLength > diffs[diffs.length - 1][1].length) {
1902     // Grow last equality.
1903     var extraLength = paddingLength - diffs[diffs.length - 1][1].length;
1904     diffs[diffs.length - 1][1] += nullPadding.substring(0, extraLength);
1905     patch.length1 += extraLength;
1906     patch.length2 += extraLength;
1907   }
1908 
1909   return nullPadding;
1910 };

1913 /**
1914  * Look through the patches and break up any which are longer than the maximum
1915  * limit of the match algorithm.
1916  * Intended to be called only from within patch_apply.
1917  * @param {!Array.<!patch_obj>} patches Array of patch objects.
1918  */
1919 diff_match_patch.prototype.patch_splitMax = function(patches) {
1920   var patch_size = this.Match_MaxBits;
1921   for (var x = 0; x < patches.length; x++) {
1922     if (patches[x].length1 > patch_size) {
1923       var bigpatch = patches[x];
1924       // Remove the big old patch.
1925       patches.splice(x--, 1);
1926       var start1 = bigpatch.start1;
1927       var start2 = bigpatch.start2;
1928       var precontext = '';
1929       while (bigpatch.diffs.length !== 0) {
1930         // Create one of several smaller patches.
1931         var patch = new patch_obj();
1932         var empty = true;
1933         patch.start1 = start1 - precontext.length;
1934         patch.start2 = start2 - precontext.length;
1935         if (precontext !== '') {
1936           patch.length1 = patch.length2 = precontext.length;
1937           patch.diffs.push([DIFF_EQUAL, precontext]);
1938         }
1939         while (bigpatch.diffs.length !== 0 &&
1940                patch.length1 < patch_size - this.Patch_Margin) {
1941           var diff_type = bigpatch.diffs[0][0];
1942           var diff_text = bigpatch.diffs[0][1];
1943           if (diff_type === DIFF_INSERT) {
1944             // Insertions are harmless.
1945             patch.length2 += diff_text.length;
1946             start2 += diff_text.length;
1947             patch.diffs.push(bigpatch.diffs.shift());
1948             empty = false;
1949           } else if (diff_type === DIFF_DELETE && patch.diffs.length == 1 &&
1950                      patch.diffs[0][0] == DIFF_EQUAL &&
1951                      diff_text.length > 2 * patch_size) {
1952             // This is a large deletion.  Let it pass in one chunk.
1953             patch.length1 += diff_text.length;
1954             start1 += diff_text.length;
1955             empty = false;
1956             patch.diffs.push([diff_type, diff_text]);
1957             bigpatch.diffs.shift();
1958           } else {
1959             // Deletion or equality.  Only take as much as we can stomach.
1960             diff_text = diff_text.substring(0,
1961                 patch_size - patch.length1 - this.Patch_Margin);
1962             patch.length1 += diff_text.length;
1963             start1 += diff_text.length;
1964             if (diff_type === DIFF_EQUAL) {
1965               patch.length2 += diff_text.length;
1966               start2 += diff_text.length;
1967             } else {
1968               empty = false;
1969             }
1970             patch.diffs.push([diff_type, diff_text]);
1971             if (diff_text == bigpatch.diffs[0][1]) {
1972               bigpatch.diffs.shift();
1973             } else {
1974               bigpatch.diffs[0][1] =
1975                   bigpatch.diffs[0][1].substring(diff_text.length);
1976             }
1977           }
1978         }
1979         // Compute the head context for the next patch.
1980         precontext = this.diff_text2(patch.diffs);
1981         precontext =
1982             precontext.substring(precontext.length - this.Patch_Margin);
1983         // Append the end context for this patch.
1984         var postcontext = this.diff_text1(bigpatch.diffs)
1985                               .substring(0, this.Patch_Margin);
1986         if (postcontext !== '') {
1987           patch.length1 += postcontext.length;
1988           patch.length2 += postcontext.length;
1989           if (patch.diffs.length !== 0 &&
1990               patch.diffs[patch.diffs.length - 1][0] === DIFF_EQUAL) {
1991             patch.diffs[patch.diffs.length - 1][1] += postcontext;
1992           } else {
1993             patch.diffs.push([DIFF_EQUAL, postcontext]);
1994           }
1995         }
1996         if (!empty) {
1997           patches.splice(++x, 0, patch);
1998         }
1999       }
2000     }
2001   }
2002 };

2019 /**
2020  * Parse a textual representation of patches and return a list of patch objects.
2021  * @param {string} textline Text representation of patches.
2022  * @return {!Array.<!patch_obj>} Array of patch objects.
2023  * @throws {!Error} If invalid input.
2024  */
2025 diff_match_patch.prototype.patch_fromText = function(textline) {
2026   var patches = [];
2027   if (!textline) {
2028     return patches;
2029   }
2030   var text = textline.split('\n');
2031   var textPointer = 0;
2032   var patchHeader = /^@@ -(\d+),?(\d*) \+(\d+),?(\d*) @@$/;
2033   while (textPointer < text.length) {
2034     var m = text[textPointer].match(patchHeader);
2035     if (!m) {
2036       throw new Error('Invalid patch string: ' + text[textPointer]);
2037     }
2038     var patch = new patch_obj();
2039     patches.push(patch);
2040     patch.start1 = parseInt(m[1], 10);
2041     if (m[2] === '') {
2042       patch.start1--;
2043       patch.length1 = 1;
2044     } else if (m[2] == '0') {
2045       patch.length1 = 0;
2046     } else {
2047       patch.start1--;
2048       patch.length1 = parseInt(m[2], 10);
2049     }
2050 
2051     patch.start2 = parseInt(m[3], 10);
2052     if (m[4] === '') {
2053       patch.start2--;
2054       patch.length2 = 1;
2055     } else if (m[4] == '0') {
2056       patch.length2 = 0;
2057     } else {
2058       patch.start2--;
2059       patch.length2 = parseInt(m[4], 10);
2060     }
2061     textPointer++;
2062 
2063     while (textPointer < text.length) {
2064       var sign = text[textPointer].charAt(0);
2065       try {
2066         var line = decodeURI(text[textPointer].substring(1));
2067       } catch (ex) {
2068         // Malformed URI sequence.
2069         throw new Error('Illegal escape in patch_fromText: ' + line);
2070       }
2071       if (sign == '-') {
2072         // Deletion.
2073         patch.diffs.push([DIFF_DELETE, line]);
2074       } else if (sign == '+') {
2075         // Insertion.
2076         patch.diffs.push([DIFF_INSERT, line]);
2077       } else if (sign == ' ') {
2078         // Minor equality.
2079         patch.diffs.push([DIFF_EQUAL, line]);
2080       } else if (sign == '@') {
2081         // Start of next patch.
2082         break;
2083       } else if (sign === '') {
2084         // Blank line?  Whatever.
2085       } else {
2086         // WTF?
2087         throw new Error('Invalid patch mode "' + sign + '" in: ' + line);
2088       }
2089       textPointer++;
2090     }
2091   }
2092   return patches;
2093 };

2115  * Emmulate GNU diff's format.
2116 2117  * Header: @@ -382,8 +481,9 @@
2118  * Indicies are printed as 1-based, not 0-based.
2119  * @return {string} The GNU diff string.
2120  */
2121 patch_obj.prototype.toString = function() {
2122   var coords1, coords2;
2123   if (this.length1 === 0) {
2124     coords1 = this.start1 + ',0';
2125   } else if (this.length1 == 1) {
2126     coords1 = this.start1 + 1;
2127   } else {
2128     coords1 = (this.start1 + 1) + ',' + this.length1;
2129   }
2130   if (this.length2 === 0) {
2131     coords2 = this.start2 + ',0';
2132   } else if (this.length2 == 1) {
2133     coords2 = this.start2 + 1;
2134   } else {
2135     coords2 = (this.start2 + 1) + ',' + this.length2;
2136   }
2137   var text = ['@@ -' + coords1 + ' +' + coords2 + ' @@\n'];
2138   var op;
2139   // Escape the body of the patch with %xx notation.
2140   for (var x = 0; x < this.diffs.length; x++) {
2141     switch (this.diffs[x][0]) {
2142       case DIFF_INSERT:
2143         op = '+';
2144         break;
2145       case DIFF_DELETE:
2146         op = '-';
2147         break;
2148       case DIFF_EQUAL:
2149         op = ' ';
2150         break;
2151     }
2152     text[x + 1] = op + encodeURI(this.diffs[x][1]) + '\n';
2153   }
2154   return text.join('').replace(/%20/g, ' ');
