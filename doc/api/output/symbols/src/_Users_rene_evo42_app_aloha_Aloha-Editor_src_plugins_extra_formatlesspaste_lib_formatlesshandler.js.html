  7 define(
  8 ['aloha', 'aloha/jquery', 'aloha/contenthandlermanager'],
  9 function(Aloha, jQuery, ContentHandlerManager) {
 10 	"use strict";
 11 
 12 	var
 13 		GENTICS = window.GENTICS;
 14 
 15 	/**
 16 	 * Register the generic content handler
 17 	 */
 18 	var FormatlessPasteHandler = ContentHandlerManager.createHandler({
 19 		
 20 		/**
 21 		 * Enable/Disable formatless pasting option 
 22 		 */
 23 		enabled: false,
 24 
 25 		
 26 		/**
 27 		 * Handle the pasting. Remove all unwanted stuff.
 28 		 * @param content
 29 		 */
 30 		handleContent: function( content ) {
 31 			if ( typeof content === 'string' ){
 32 				content = jQuery( '<div>' + content + '</div>' );
 33 			} else if ( content instanceof jQuery ) {
 34 				content = jQuery( '<div>' ).append(content);
 35 			}
 36 			// If we find an aloha-block inside the pasted content,
 37 			// we do not modify the pasted stuff, as it most probably
 38 			// comes from Aloha and not from other sources, and does
 39 			// not need to be cleaned up.
 40 			if ( content.find('.aloha-block').length > 0 ) {
 41 				return;
 42 			};
 43 
 44 			// remove formattings
 45 			if ( this.enabled ) {
 46 				this.removeFormatting( content );
 47 			}
 48 				
 49 			return content.html();
 50 		},
 51 
 52 		/**
 53 		 * Remove formatting
 54 		 * @param jqPasteDiv
 55 		 */
 56 		removeFormatting: function( jqPasteDiv ) {
 57 			var formatting_elements = this.strippedElements;
 58 
 59 			// find all formattings we will transform
 60 			jqPasteDiv.find( formatting_elements.join(",")).each(function() {
 61 				jQuery(this).contents().unwrap();
 62 			});
 63 		}
 64   });
 65 
 66 	return FormatlessPasteHandler;
 67 });
