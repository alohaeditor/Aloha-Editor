 41 		/**
 42 		 * Get the group with given index. If it does not yet exist, create a new one
 43 		 * @method
 44 		 * @param {int} group group index of the group to get
 45 		 * @return group object
 46 		 */
 47 		getGroup: function(group) {
 48 			var groupObject = this.groupMap[group];
 49 			if (typeof groupObject === 'undefined') {
 50 				groupObject = new Group();
 51 				this.groupMap[group] = groupObject;
 52 				this.groups.push(groupObject);
 53 				// TODO resort the groups
 54 			}
 55 
 56 			return groupObject;

 59 		/**
 60 		 * Get the EXT component representing the tab
 61 		 * @return EXT component (EXT.Panel)
 62 		 * @hide
 63 		 */
 64 		getExtComponent: function () {
 65 			var that = this;
 66 
 67 			if (typeof this.extPanel === 'undefined') {
 68 				// generate the panel here
 69 				this.extPanel = new Ext.Panel({
 70 					'tbar' : [],
 71 					'title' : this.label,
 72 					'style': 'margin-top:0px',
 73 					'bodyStyle': 'display:none',
 74 					'autoScroll': true
 75 				});
 76 
 77 				// add the groups
 78 				jQuery.each(this.groups, function(index, group) {
 79 					// let each group generate its ext component and add them to the panel
 80 					that.extPanel.getTopToolbar().add(group.getExtComponent());
 81 				});
 82 			}
 83 
 84 			return this.extPanel;
 85 		},

 87 		/**
 88 		 * Recalculate the visibility of all groups within the tab
 89 		 * @hide
 90 		 */
 91 		doLayout: function() {
 92 			var that = this;
 93 
 94 			if (Aloha.Log.isDebugEnabled()) {
 95 				Aloha.Log.debug(this, 'doLayout called for tab ' + this.label);

 97 			this.visible = false;
 98 
 99 			// check all groups in this tab
100 			jQuery.each(this.groups, function(index, group) {
101 				that.visible |= group.doLayout();
102 			});
103 
104 			if (Aloha.Log.isDebugEnabled()) {
105 				Aloha.Log.debug(this, 'tab ' + this.label + (this.visible ? ' is ' : ' is not ') + 'visible now');
106 			}
107 
108 			return this.visible;
109 		}

124 		/**
125 		 * Add a button to this group
126 		 * @param {Button} buttonInfo to add to the group
127 		 */
128 		addButton: function(buttonInfo) {
129 			if (buttonInfo.button instanceof Aloha.ui.AttributeField) {
130 				if (this.fields.length < 2) {
131 					this.fields.push(buttonInfo);
132 				} else {
133 					throw new Error("Too much fields in this group");
134 				}
135 			} else {
136 				// Every plugin API entryPoint (method) should be securised enough
137 				// to avoid Aloha to block at startup even
138 				// if a plugin is badly designed
139 				if (typeof buttonInfo.button !== "undefined"){
140 					this.buttons.push(buttonInfo);
141 				}
142 			}
143 		},

145 		/**
146 		 * Get the EXT component representing the group (Ext.ButtonGroup)
147 		 * @return the Ext.ButtonGroup
148 		 * @hide
149 		 */
150 		getExtComponent: function () {
151 			var that = this, l,
152 				items = [],
153 				buttonCount = 0,
154 				columnCount = 0,
155 				len, idx, half;
156 
157 
158 			if (typeof this.extButtonGroup === 'undefined') {
159 			
160 				if (this.fields.length > 1) {
161 					columnCount = 1;
162 				}
163 
164 				jQuery.each(this.buttons, function(index, button) {
165 					// count the number of buttons (large buttons count as 2)
166 					buttonCount += button.button.size == 'small' ? 1 : 2;
167 				});
168 				columnCount = columnCount + Math.ceil(buttonCount / 2);
169 
170 				len = this.buttons.length;
171 				idx = 0;
172 				half =  Math.ceil(this.buttons.length / 2) - this.buttons.length % 2 ;
173 
174 				if (this.fields.length > 0) {
175 					that.buttons.push(this.fields[0]);
176 					items.push(this.fields[0].button.getExtConfigProperties());
177 				}
178 
179 				while (--len >= half) {
180 					items.push(this.buttons[idx++].button.getExtConfigProperties());
181 				}
182 				++len;
183 				if (this.fields.length > 1) {
184 					that.buttons.push(this.fields[1]);
185 					items.push(this.fields[1].button.getExtConfigProperties());
186 				}
187 				while (--len >=0) {
188 					items.push(this.buttons[idx++].button.getExtConfigProperties());
189 				}
190 
191 				this.extButtonGroup = new Ext.ButtonGroup({
192 					'columns' : columnCount,
193 					'items': items
194 				});
195 	//			jQuery.each(this.fields, function(id, field){
196 	//				that.buttons.push(field);
197 	//			});
198 				// now find the Ext.Buttons and set to the GENTICS buttons
199 				jQuery.each(this.buttons, function(index, buttonInfo) {
200 					buttonInfo.button.extButton = that.extButtonGroup.findById(buttonInfo.button.id);
201 					// the following code is a work arround because ExtJS initializes later.
202 					// The ui wrapper store the information and here we use it... ugly.
203 					// if there are any listeners added before initializing the extButtons
204 					if ( buttonInfo.button.listenerQueue && buttonInfo.button.listenerQueue.length > 0 ) {
205 						while ( true ) {
206 							l = buttonInfo.button.listenerQueue.shift();
207 							if ( !l ) {break;}
208 							buttonInfo.button.extButton.addListener(l.eventName, l.handler, l.scope, l.options);
209 						}
210 					}
211 					if (buttonInfo.button.extButton.setObjectTypeFilter) {
212 						if (buttonInfo.button.objectTypeFilter) {
213 							buttonInfo.button.extButton.noQuery = false;
214 						}
215 						if ( buttonInfo.button.objectTypeFilter == 'all' ) {
216 							buttonInfo.button.objectTypeFilter = null;
217 						}
218 						buttonInfo.button.extButton.setObjectTypeFilter(buttonInfo.button.objectTypeFilter);
219 						if ( buttonInfo.button.displayField) {
220 							buttonInfo.button.extButton.displayField = buttonInfo.button.displayField;
221 						}
222 						if ( buttonInfo.button.tpl ) {
223 							buttonInfo.button.extButton.tpl = buttonInfo.button.tpl;
224 						}
225 					}
226 				});
227 			}
228 
229 			return this.extButtonGroup;
230 		},
231 
232 		/**
233 		 * Recalculate the visibility of the buttons and the group
234 		 * @hide
235 		 */
236 		doLayout: function () {
237 			var groupVisible = false,
238 				that = this;
239 			jQuery.each(this.buttons, function(index, button) {
240 				if (typeof button.button !== "undefined") {
241 					var extButton = that.extButtonGroup.findById(button.button.id),
242 					buttonVisible = button.button.isVisible() && button.scopeVisible;
243 					
244 					if (!extButton) {
245 						return;
246 					}
247 				
248 					if (buttonVisible && extButton.hidden) {
249 						extButton.show();
250 					} else if (!buttonVisible && extButton && !extButton.hidden) {
251 						extButton.hide();
252 					}
253 				
254 					groupVisible |= buttonVisible;
255 				}
256 			});
257 			if (groupVisible && this.extButtonGroup.hidden) {
258 				this.extButtonGroup.show();
259 			} else if (!groupVisible && !this.extButtonGroup.hidden) {
260 				this.extButtonGroup.hide();
261 			}
262 		
263 			return groupVisible;
264 
265 		}
266 	});

387 		/**
388 		 * Initialize the floatingmenu
389 		 * @hide
390 		 */
391 		init: function() {
392 
393 			// check for behaviour setting of the floating menu
394 		    if (Aloha.settings.floatingmenu) {
395 		    	
396 		    	if (typeof Aloha.settings.floatingmenu.draggable === 'boolean') {
397 		    		this.draggable = Aloha.settings.floatingmenu.draggable;
398 		    	}
399 				if (typeof Aloha.settings.floatingmenu.behaviour === 'string') {
400 					this.behaviour = Aloha.settings.floatingmenu.behaviour;
401 				}
402 				if (typeof Aloha.settings.floatingmenu.topalignOffset !== 'undefined') {
403 					this.topalignOffset = Aloha.settings.floatingmenu.topalignOffset;
404 				}
405 				if (typeof Aloha.settings.floatingmenu.horizontalOffset !== 'undefined') {
406 					this.horizontalOffset = Aloha.settings.floatingmenu.horizontalOffset;
407 				}
408 				if (typeof Aloha.settings.floatingmenu.marginTop === 'number') {
409 				    this.marginTop = Aloha.settings.floatingmenu.marginTop;
410 				}
411 				//We just check for undefined
412 				if (typeof Aloha.settings.floatingmenu.width !== 'undefined') {
413 					//Try to pars it
414 					try {
415 						var parsed = parseInt(Aloha.settings.floatingmenu.width);
416 						this.width = Aloha.settings.floatingmenu.width;
417 					} catch(e) {
418 						//do nothing.
419 					}
420 				}
421 		    }
422 
423 			jQuery.storage = new jQuery.store();
424 			this.currentScope = 'Aloha.global';
425 			var that = this;
426 			this.window.unload(function () {
427 				// store fm position if the panel is pinned to be able to restore it next time
428 				if (that.pinned) {
429 					jQuery.storage.set('Aloha.FloatingMenu.pinned', 'true');
430 					jQuery.storage.set('Aloha.FloatingMenu.top', that.top);
431 					jQuery.storage.set('Aloha.FloatingMenu.left', that.left);
432 					if (Aloha.Log.isInfoEnabled()) {
433 						Aloha.Log.info(this, 'stored FloatingMenu pinned position {' + that.left
434 								+ ', ' + that.top + '}');
435 					}
436 				} else {
437 					// delete old localStorages
438 					jQuery.storage.del('Aloha.FloatingMenu.pinned');
439 					jQuery.storage.del('Aloha.FloatingMenu.top');
440 					jQuery.storage.del('Aloha.FloatingMenu.left');
441 				}
442 				if (that.userActivatedTab) {
443 					jQuery.storage.set('Aloha.FloatingMenu.activeTab', that.userActivatedTab);
444 				}
445 			}).resize(function () {
446 				if (that.behaviour === 'float') {
447 					if (that.pinned) {
448 						that.fixPinnedPosition();
449 						that.refreshShadow();
450 						that.extTabPanel.setPosition(that.left, that.top);
451 					} else {
452 						var target = that.calcFloatTarget(Aloha.Selection.getRangeObject());
453 						if (target) {
454 							that.floatTo(target);
455 						}
456 					}
457 				}
458 			});
459 			Aloha.bind('aloha-ready', function() {
460 				that.generateComponent();
461 				that.initialized = true;
462 			});
463 			
464 			if (typeof Aloha.settings.toolbar === 'object') {
465 				this.fromConfig = true;				
466 			}
467 		},

470 		 * jQuery reference to the extjs tabpanel
471 		 * @hide
472 		 */
473 		obj: null,

477 		 * @hide
478 		 */
479 		shadow: null,

493 		/**
494 		 * initialize tabs and groups according to the current configuration
495 		 */
496 		initTabsAndGroups: function () {
497 			var that = this;
498 			
499 			// if there is no toolbar config tabs and groups have been initialized before
500 			if (!this.fromConfig) {
501 				return;
502 			}
503 			
504 			jQuery.each(Aloha.settings.toolbar.tabs, function (tab, groups) {
505 				// generate or retrieve tab
506 				var tabObject = that.tabMap[tab];
507 				if (typeof tabObject === 'undefined') {
508 					// the tab object does not yet exist, so create a new tab and add it to the list
509 					tabObject = new Tab(tab);
510 					that.tabs.push(tabObject);
511 					that.tabMap[tab] = tabObject;
512 				}
513 				
514 				// generate groups for current tab
515 				jQuery.each(groups, function (group, buttons) {
516 					var groupObject = tabObject.getGroup(group),
517 						i;
518 
519 					// now get all the buttons for that group
520 					jQuery.each(buttons, function (j, button) {
521 						if (jQuery.inArray(button, that.buttonsAdded) !== -1) {
522 							// buttons must not be added twice
523 							console.warn('Skipping button {' + button + '}. A button can\'t be added ' + 
524 								'to the floating menu twice. Config key: {Aloha.settings.toolbar.' + 
525 									tab + '.' + group + '}');
526 									return;
527 						}
528 						
529 						// now add the button to the group
530 						for (i = 0; i < that.allButtons.length; i++) {
531 							if (button === that.allButtons[i].button.name) {
532 								groupObject.addButton(that.allButtons[i]);
533 								// remember that we've added the button
534 								that.buttonsAdded.push(that.allButtons[i].button.name);
535 								break;
536 							}
537 						}
538 					});
539 				});
540 			});
541 		},

543 		/**
544 		 * Generate the rendered component for the floatingmenu
545 		 * @hide
546 		 */
547 		generateComponent: function () {
548 			var that = this, pinTab;
549 
550 			// initialize tabs and groups first
551 			this.initTabsAndGroups();
552 
553 			// Initialize and configure the tooltips
554 			Ext.QuickTips.init();
555 			Ext.apply(Ext.QuickTips.getQuickTip(), {
556 				minWidth : 10
557 			});
558 
559 			
560 			
561 			if (this.extTabPanel) {
562 				// TODO dispose of the ext component
563 			} else {
564 				
565 				// Enable or disable the drag functionality
566 				var dragConfiguration = false;
567 
568 				if ( that.draggable ) {
569 					dragConfiguration = {
570 						insertProxy: false,
571 						onDrag : function(e) {
572 							var pel = this.proxy.getEl();
573 							this.x = pel.getLeft(true);
574 							this.y = pel.getTop(true);
575 							this.panel.shadow.hide();
576 						},
577 						endDrag : function(e) {
578 							var top = (that.pinned) ? this.y - jQuery(document).scrollTop() : this.y;
579 					
580 							that.left = this.x;
581 							that.top = top;
582 
583 							this.panel.setPosition(this.x, top);
584 							that.refreshShadow();
585 							this.panel.shadow.show();
586 						}
587 					};
588 				}
589 				// generate the tabpanel object
590 				this.extTabPanel = new Ext.TabPanel({
591 					activeTab: 0,
592 					width: that.width, // 336px this fits the multisplit button and 6 small buttons placed in 3 cols
593 594 					plain: false,
595 					draggable: dragConfiguration,
596 					floating: {shadow: false},
597 					defaults: {
598 						autoScroll: true
599 					},
600 					layoutOnTabChange : true,
601 					shadow: false,
602 					cls: 'aloha-floatingmenu ext-root',
603 					listeners : {
604 						'tabchange' : {
605 							'fn' : function(tabPanel, tab) {
606 								if (tab.title != that.autoActivatedTab) {
607 									if (Aloha.Log.isDebugEnabled()) {
608 										Aloha.Log.debug(that, 'User selected tab ' + tab.title);
609 									}
610 									// remember the last user-selected tab
611 									that.userActivatedTab = tab.title;
612 								} else {
613 									if (Aloha.Log.isDebugEnabled()) {
614 										Aloha.Log.debug(that, 'Tab ' + tab.title + ' was activated automatically');
615 									}
616 								}
617 								that.autoActivatedTab = undefined;
618 						
619 								// ok, this is kind of a hack: when the tab changes, we check all buttons for multisplitbuttons (which have the method setActiveDOMElement).
620 								// if a DOM Element is queued to be set active, we try to do this now.
621 								// the reason for this is that the active DOM element can only be set when the multisplit button is currently visible.
622 								jQuery.each(that.allButtons, function(index, buttonInfo) {
623 									if (typeof buttonInfo.button !== 'undefined'
624 										&& typeof buttonInfo.button.extButton !== 'undefined'
625 										&& buttonInfo.button.extButton !== null
626 										&& typeof buttonInfo.button.extButton.setActiveDOMElement === 'function') {
627 										if (typeof buttonInfo.button.extButton.activeDOMElement !== 'undefined') {
628 											buttonInfo.button.extButton.setActiveDOMElement(buttonInfo.button.extButton.activeDOMElement);
629 										}
630 									}
631 								});
632 						
633 								// adapt the shadow
634 								that.extTabPanel.shadow.show();
635 								that.refreshShadow();
636 							}
637 						}
638 					},
639 					enableTabScroll : true
640 				});
641 		
642 		
643 			}
644 			// add the tabs
645 			jQuery.each(this.tabs, function(index, tab) {
646 				// let each tab generate its ext component and add them to the panel
647 				try {
648 					that.extTabPanel.add(tab.getExtComponent());
649 				} catch(e) {
650 					Aloha.Log.error(that,"Error while inserting tab: " + e);
651 				}
652 			});
653 
654 			// add the dropshadow
655 			this.extTabPanel.shadow = jQuery('<div id="aloha-floatingmenu-shadow" class="aloha-shadow"> </div>');
656 			jQuery('body').append(this.extTabPanel.shadow);
657 
658 			// add an empty pin tab item, store reference
659 			pinTab = this.extTabPanel.add({
660 				title : ' '
661 			});
662 
663 			// finally render the panel to the body
664 			this.extTabPanel.render(document.body);
665 
666 			// finish the pin element after the FM has rendered (before there are noe html contents to be manipulated
667 			jQuery(pinTab.tabEl)

1106 		/**
1107 		 * returns the next possible float target dom obj
1108 		 * the floating menu should only float to h1-h6, p, div, td and pre elements
1109 		 * if the current object is not valid, it's parentNode will be considered, until
1110 		 * the limit object is hit
1111 		 * @param obj the dom object to start from (commonly this would be the commonAncestorContainer)
1112 		 * @param limitObj the object that limits the range (this would be the editable)
1113 		 * @return dom object which qualifies as a float target
1114 		 * @hide
1115 		 */
1116 		nextFloatTargetObj: function (obj, limitObj) {
1117 			// if we've hit the limit object we don't care for it's type
1118 			if (!obj || obj == limitObj) {
1119 				return obj;
1120 			}
1121 
1122 			// fm will only float to h1-h6, p, div, td
1123 			switch (obj.nodeName.toLowerCase()) {
1124 				case 'h1':
1125 				case 'h2':
1126 				case 'h3':
1127 				case 'h4':
1128 				case 'h5':
1129 				case 'h6':
1130 				case 'p':
1131 				case 'div':
1132 				case 'td':
1133 				case 'pre':
1134 				case 'ul':
1135 				case 'ol':
1136 					return obj;
1137 				default:
1138 					return this.nextFloatTargetObj(obj.parentNode, limitObj);
1139 			}

1142 		/**
1143 		 * calculates the float target coordinates for a range
1144 		 * @param range the fm should float to
1145 		 * @return object containing left and top coordinates, like { left : 20, top : 43 }
1146 		 * @hide
1147 		 */
1148 		calcFloatTarget: function(range) {
1149 			var
1150 				i, editableLength, target,
1151 				targetObj, scrollTop, top;
1152 
1153 			// TODO in IE8 somteimes a broken range is handed to this function - investigate this
1154 			if (!Aloha.activeEditable || typeof range.getCommonAncestorContainer === 'undefined') {
1155 				return false;
1156 			}
1157 
1158 			// check if the designated editable is disabled
1159 			for ( i = 0, editableLength = Aloha.editables.length; i < editableLength; i++) {
1160 				if (Aloha.editables[i].obj.get(0) == range.limitObject &&
1161 						Aloha.editables[i].isDisabled()) {
1162 					return false;
1163 				}
1164 			}
1165 
1166 			target = this.nextFloatTargetObj(range.getCommonAncestorContainer(), range.limitObject);
1167 			if ( ! target ) {
1168 				return false;
1169 			}
1170 
1171 			targetObj = jQuery(target);
1172 			scrollTop = GENTICS.Utils.Position.Scroll.top;
1173 			if (!targetObj || !targetObj.offset()) {
1174 				return false;
1175 			}
1176 			top = targetObj.offset().top - this.obj.height() - 50; // 50px offset above the current obj to have some space above
1177 
1178 			// if the floating menu would be placed higher than the top of the screen...
1179 			if ( top < scrollTop) {
1180 				top += 50 + GENTICS.Utils.Position.ScrollCorrection.top;
1181 			}
1182 
1183 			// if the floating menu would float off the bottom of the screen
1184 			// we don't want it to move, so we'll return false
1185 			if (top > this.window.height() + this.window.scrollTop()) {
1186 				return false;
1187 			}
1188 
1189 			return {
1190 				left : Aloha.activeEditable.obj.offset().left,
1191 				top : top
1192 			};
1193 		},
