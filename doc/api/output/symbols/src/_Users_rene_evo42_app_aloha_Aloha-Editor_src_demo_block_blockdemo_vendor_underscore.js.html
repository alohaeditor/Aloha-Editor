 67   // The cornerstone, an `each` implementation, aka `forEach`.
 68   // Handles objects implementing `forEach`, arrays, and raw objects.
 69   // Delegates to **ECMAScript 5**'s native `forEach` if available.
 70   var each = _.each = _.forEach = function(obj, iterator, context) {
 71     if (obj == null) return;
 72     if (nativeForEach && obj.forEach === nativeForEach) {
 73       obj.forEach(iterator, context);
 74     } else if (_.isNumber(obj.length)) {
 75       for (var i = 0, l = obj.length; i < l; i++) {
 76         if (iterator.call(context, obj[i], i, obj) === breaker) return;
 77       }
 78     } else {
 79       for (var key in obj) {
 80         if (hasOwnProperty.call(obj, key)) {
 81           if (iterator.call(context, obj[key], key, obj) === breaker) return;
 82         }
 83       }
 84     }
 85   };

 99   // **Reduce** builds up a single result from a list of values, aka `inject`,
100   // or `foldl`. Delegates to **ECMAScript 5**'s native `reduce` if available.
101   _.reduce = _.foldl = _.inject = function(obj, iterator, memo, context) {
102     var initial = memo !== void 0;
103     if (obj == null) obj = [];
104     if (nativeReduce && obj.reduce === nativeReduce) {
105       if (context) iterator = _.bind(iterator, context);
106       return initial ? obj.reduce(iterator, memo) : obj.reduce(iterator);
107     }
108 109     each(obj, function(value, index, list) {
110       if (!initial && index === 0) {
111         memo = value;
112         initial = true;
113       } else {
114         memo = iterator.call(context, memo, value, index, list);
115       }
116     });
117     if (!initial) throw new TypeError("Reduce of empty array with no initial value");
118     return memo;
119   };

256   // Use a comparator function to figure out at what index an object should
257   // be inserted so as to maintain order. Uses binary search.
258   _.sortedIndex = function(array, obj, iterator) {
259     iterator || (iterator = _.identity);
260     var low = 0, high = array.length;
261     while (low < high) {
262       var mid = (low + high) >> 1;
263       iterator(array[mid]) < iterator(obj) ? low = mid + 1 : high = mid;
264     }
265     return low;

356   // If the browser doesn't supply us with indexOf (I'm looking at you, **MSIE**),
357   // we need this function. Return the position of the first occurrence of an
358   // item in an array, or -1 if the item is not included in the array.
359   // Delegates to **ECMAScript 5**'s native `indexOf` if available.
360   // If the array is large and already in sort order, pass `true`

362   _.indexOf = function(array, item, isSorted) {

368     }
369     if (nativeIndexOf && array.indexOf === nativeIndexOf) return array.indexOf(item);
370     for (i = 0, l = array.length; i < l; i++) if (array[i] === item) return i;
371     return -1;

384   // Generate an integer Array containing an arithmetic progression. A port of
385   // the native Python `range()` function. See
386   // [the Python documentation](http://docs.python.org/library/functions.html#range).
387   _.range = function(start, stop, step) {
388     if (arguments.length <= 1) {
389       stop = start || 0;
390       start = 0;
391     }
392     step = arguments[2] || 1;
393 
394     var len = Math.max(Math.ceil((stop - start) / step), 0);
395     var idx = 0;
396     var range = new Array(len);
397 
398     while(idx < len) {
399       range[idx++] = start;
400       start += step;
401     }
402 
403     return range;
404   };

579   // Perform a deep comparison to check if two objects are equal.
580   _.isEqual = function(a, b) {
581     // Check object identity.
582     if (a === b) return true;
583     // Different types?
584     var atype = typeof(a), btype = typeof(b);
585     if (atype != btype) return false;
586     // Basic equality test (watch out for coercions).
587     if (a == b) return true;
588     // One is falsy and the other truthy.
589     if ((!a && b) || (a && !b)) return false;
590     // Unwrap any wrapped objects.
591     if (a._chain) a = a._wrapped;
592     if (b._chain) b = b._wrapped;
593     // One of them implements an isEqual()?
594     if (a.isEqual) return a.isEqual(b);
595     // Check dates' integer values.
596     if (_.isDate(a) && _.isDate(b)) return a.getTime() === b.getTime();
597     // Both are NaN?
598     if (_.isNaN(a) && _.isNaN(b)) return false;
599     // Compare regular expressions.
600     if (_.isRegExp(a) && _.isRegExp(b))
601       return a.source     === b.source &&
602              a.global     === b.global &&
603              a.ignoreCase === b.ignoreCase &&
604              a.multiline  === b.multiline;
605     // If a is not an object by this point, we can't handle it.
606     if (atype !== 'object') return false;
607     // Check for different array lengths before comparing contents.
608     if (a.length && (a.length !== b.length)) return false;
609     // Nothing else worked, deep compare the contents.
610     var aKeys = _.keys(a), bKeys = _.keys(b);
611     // Different object sizes?
612     if (aKeys.length != bKeys.length) return false;
613     // Recursive comparison of contents.
614     for (var key in a) if (!(key in b) || !_.isEqual(a[key], b[key])) return false;
615     return true;
616   };

730   // JavaScript micro-templating, similar to John Resig's implementation.
731   // Underscore templating handles arbitrary delimiters, preserves whitespace,
732   // and correctly escapes quotes within interpolated code.
733   _.template = function(str, data) {
734     var c  = _.templateSettings;
735     var tmpl = 'var __p=[],print=function(){__p.push.apply(__p,arguments);};' +
736       'with(obj||{}){__p.push(\'' +
737       str.replace(/\\/g, '\\\\')
738          .replace(/'/g, "\\'")
739          .replace(c.interpolate, function(match, code) {
740            return "'," + code.replace(/\\'/g, "'") + ",'";
741          })
742          .replace(c.evaluate || null, function(match, code) {
743            return "');" + code.replace(/\\'/g, "'")
744                               .replace(/[\r\n\t]/g, ' ') + "__p.push('";
745          })
746          .replace(/\r/g, '\\r')
747          .replace(/\n/g, '\\n')
748          .replace(/\t/g, '\\t')
749          + "');}return __p.join('');";
750     var func = new Function('obj', tmpl);
751     return data ? func(data) : func;
752   };

754   // The OOP Wrapper
755   // ---------------

757   // If Underscore is called as a function, it returns a wrapped object that
758   // can be used OO-style. This wrapper holds altered versions of all the
759   // underscore functions. Wrapped objects may be chained.
760   var wrapper = function(obj) { this._wrapped = obj; };
761 
762   // Expose `wrapper.prototype` as `_.prototype`
763   _.prototype = wrapper.prototype;
764 
765   // Helper function to continue chaining intermediate results.
766   var result = function(obj, chain) {
767     return chain ? _(obj).chain() : obj;
768   };
769 
770   // A method to easily add functions to the OOP wrapper.
771   var addToWrapper = function(name, func) {
772     wrapper.prototype[name] = function() {
773       var args = slice.call(arguments);
774       unshift.call(args, this._wrapped);
775       return result(func.apply(_, args), this._chain);
776     };
777   };

780   _.mixin(_);

782   // Add all mutator Array functions to the wrapper.
783   each(['pop', 'push', 'reverse', 'shift', 'sort', 'splice', 'unshift'], function(name) {
784     var method = ArrayProto[name];
785     wrapper.prototype[name] = function() {
786       method.apply(this._wrapped, arguments);
787       return result(this._wrapped, this._chain);
788     };
789   });
790 
791   // Add all accessor Array functions to the wrapper.
792   each(['concat', 'join', 'slice'], function(name) {
793     var method = ArrayProto[name];
794     wrapper.prototype[name] = function() {
795       return result(method.apply(this._wrapped, arguments), this._chain);
796     };
797   });

799   // Start chaining a wrapped Underscore object.
800   wrapper.prototype.chain = function() {
801     this._chain = true;
802     return this;
803   };
804 
805   // Extracts the result from a wrapped and chained object.
806   wrapper.prototype.value = function() {
807     return this._wrapped;
808   };
