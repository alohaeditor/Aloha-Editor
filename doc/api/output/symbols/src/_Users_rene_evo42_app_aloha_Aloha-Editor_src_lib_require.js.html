199         /**
200          * Trims the . and .. from an array of path segments.
201          * It will keep a leading path segment if a .. will become
202          * the first path segment, to help with module name lookups,
203          * which act like paths, but can be remapped. But the end result,
204          * all paths that use this function should look normalized.
205          * NOTE: this method MODIFIES the input array.
206          * @param {Array} ary the array of path segments.
207          */
208         function trimDots(ary) {
209             var i, part;
210             for (i = 0; (part = ary[i]); i++) {
211                 if (part === ".") {
212                     ary.splice(i, 1);
213                     i -= 1;
214                 } else if (part === "..") {
215                     if (i === 1 && (ary[2] === '..' || ary[0] === '..')) {
216                         //End of the line. Keep at least one non-dot
217                         //path segment at the front so it can be mapped
218                         //correctly to disk. Otherwise, there is likely
219                         //no path mapping for a path starting with '..'.
220                         //This can still fail, but catches the most reasonable
221                         //uses of ..
222                         break;
223                     } else if (i > 0) {
224                         ary.splice(i - 1, 2);
225                         i -= 2;
226                     }
227                 }
228             }
229         }

231         /**
232          * Given a relative module name, like ./something, normalize it to
233          * a real name that can be mapped to a path.
234          * @param {String} name the relative name
235          * @param {String} baseName a real name that the name arg is relative
236          * to.
237          * @returns {String} normalized name
238          */
239         function normalize(name, baseName) {
240             var pkgName, pkgConfig;
241 
242             //Adjust any relative paths.
243             if (name && name.charAt(0) === ".") {
244                 //If have a base name, try to normalize against it,
245                 //otherwise, assume it is a top-level require that will
246                 //be relative to baseUrl in the end.
247                 if (baseName) {
248                     if (config.pkgs[baseName]) {
249                         //If the baseName is a package name, then just treat it as one
250                         //name to concat the name with.
251                         baseName = [baseName];
252                     } else {
253                         //Convert baseName to array, and lop off the last part,
254                         //so that . matches that "directory" and not name of the baseName's
255                         //module. For instance, baseName of "one/two/three", maps to
256                         //"one/two/three.js", but we want the directory, "one/two" for
257                         //this normalization.
258                         baseName = baseName.split("/");
259                         baseName = baseName.slice(0, baseName.length - 1);
260                     }
261 
262                     name = baseName.concat(name.split("/"));
263                     trimDots(name);
264 
265                     //Some use of packages may use a . path to reference the
266                     //"main" module name, so normalize for that.
267                     pkgConfig = config.pkgs[(pkgName = name[0])];
268                     name = name.join("/");
269                     if (pkgConfig && name === pkgName + '/' + pkgConfig.main) {
270                         name = pkgName;
271                     }
272                 }
273             }
274             return name;
275         }

277         /**
278          * Creates a module mapping that includes plugin prefix, module
279          * name, and path. If parentModuleMap is provided it will
280          * also normalize the name via require.normalize()
281          *
282          * @param {String} name the module name
283          * @param {String} [parentModuleMap] parent module map
284          * for the module name, used to resolve relative names.
285          *
286          * @returns {Object}
287          */
288         function makeModuleMap(name, parentModuleMap) {
289             var index = name ? name.indexOf("!") : -1,
290                 prefix = null,
291                 parentName = parentModuleMap ? parentModuleMap.name : null,
292                 originalName = name,
293                 normalizedName, url, pluginModule;
294 
295             if (index !== -1) {
296                 prefix = name.substring(0, index);
297                 name = name.substring(index + 1, name.length);
298             }
299 
300             if (prefix) {
301                 prefix = normalize(prefix, parentName);
302             }
303 
304             //Account for relative paths if there is a base name.
305             if (name) {
306                 if (prefix) {
307                     pluginModule = defined[prefix];
308                     if (pluginModule) {
309                         //Plugin is loaded, use its normalize method, otherwise,
310                         //normalize name as usual.
311                         if (pluginModule.normalize) {
312                             normalizedName = pluginModule.normalize(name, function (name) {
313                                 return normalize(name, parentName);
314                             });
315                         } else {
316                             normalizedName = normalize(name, parentName);
317                         }
318                     } else {
319                         //Plugin is not loaded yet, so do not normalize
320                         //the name, wait for plugin to load to see if
321                         //it has a normalize method. To avoid possible
322                         //ambiguity with relative names loaded from another
323                         //plugin, use the parent's name as part of this name.
324                         normalizedName = '__$p' + parentName + '@' + (name || '');
325                     }
326                 } else {
327                     normalizedName = normalize(name, parentName);
328                 }
329 
330                 url = urlMap[normalizedName];
331                 if (!url) {
332                     //Calculate url for the module, if it has a name.
333                     if (req.toModuleUrl) {
334                         //Special logic required for a particular engine,
335                         //like Node.
336                         url = req.toModuleUrl(context, normalizedName, parentModuleMap);
337                     } else {
338                         url = context.nameToUrl(normalizedName, null, parentModuleMap);
339                     }
340 
341                     //Store the URL mapping for later.
342                     urlMap[normalizedName] = url;
343                 }
344             }
345 
346             return {
347                 prefix: prefix,
348                 name: normalizedName,
349                 parentMap: parentModuleMap,
350                 url: url,
351                 originalName: originalName,
352                 fullName: prefix ? prefix + "!" + (normalizedName || '') : normalizedName
353             };
354         }

427         /**
428          * Used to update the normalized name for plugin-based dependencies
429          * after a plugin loads, since it can have its own normalization structure.
430          * @param {String} pluginName the normalized plugin module name.
431          */
432         function updateNormalizedNames(pluginName) {
433 
434             var oldFullName, oldModuleMap, moduleMap, fullName, callbacks,
435                 i, j, k, depArray, existingCallbacks,
436                 maps = normalizedWaiting[pluginName];
437 
438             if (maps) {
439                 for (i = 0; (oldModuleMap = maps[i]); i++) {
440                     oldFullName = oldModuleMap.fullName;
441                     moduleMap = makeModuleMap(oldModuleMap.originalName, oldModuleMap.parentMap);
442                     fullName = moduleMap.fullName;
443                     //Callbacks could be undefined if the same plugin!name was
444                     //required twice in a row, so use empty array in that case.
445                     callbacks = managerCallbacks[oldFullName] || [];
446                     existingCallbacks = managerCallbacks[fullName];
447 
448                     if (fullName !== oldFullName) {
449                         //Update the specified object, but only if it is already
450                         //in there. In sync environments, it may not be yet.
451                         if (oldFullName in specified) {
452                             delete specified[oldFullName];
453                             specified[fullName] = true;
454                         }
455 
456                         //Update managerCallbacks to use the correct normalized name.
457                         //If there are already callbacks for the normalized name,
458                         //just add to them.
459                         if (existingCallbacks) {
460                             managerCallbacks[fullName] = existingCallbacks.concat(callbacks);
461                         } else {
462                             managerCallbacks[fullName] = callbacks;
463                         }
464                         delete managerCallbacks[oldFullName];
465 
466                         //In each manager callback, update the normalized name in the depArray.
467                         for (j = 0; j < callbacks.length; j++) {
468                             depArray = callbacks[j].depArray;
469                             for (k = 0; k < depArray.length; k++) {
470                                 if (depArray[k] === oldFullName) {
471                                     depArray[k] = fullName;
472                                 }
473                             }
474                         }
475                     }
476                 }
477             }
478 
479             delete normalizedWaiting[pluginName];
480         }

482         /*
483          * Queues a dependency for checking after the loader is out of a
484          * "paused" state, for example while a script file is being loaded
485          * in the browser, where it may have many modules defined in it.
486          *
487          * depName will be fully qualified, no relative . or .. path.
488          */
489         function queueDependency(dep) {
490             //Make sure to load any plugin and associate the dependency
491             //with that plugin.
492             var prefix = dep.prefix,
493                 fullName = dep.fullName;
494 
495             //Do not bother if the depName is already in transit
496             if (specified[fullName] || fullName in defined) {
497                 return;
498             }
499 
500             if (prefix && !plugins[prefix]) {
501                 //Queue up loading of the dependency, track it
502                 //via context.plugins. Mark it as a plugin so
503                 //that the build system will know to treat it
504                 //special.
505                 plugins[prefix] = undefined;
506 
507                 //Remember this dep that needs to have normaliztion done
508                 //after the plugin loads.
509                 (normalizedWaiting[prefix] || (normalizedWaiting[prefix] = []))
510                     .push(dep);
511 
512                 //Register an action to do once the plugin loads, to update
513                 //all managerCallbacks to use a properly normalized module
514                 //name.
515                 (managerCallbacks[prefix] ||
516                 (managerCallbacks[prefix] = [])).push({
517                     onDep: function (name, value) {
518                         if (name === prefix) {
519                             updateNormalizedNames(prefix);
520                         }
521                     }
522                 });
523 
524                 queueDependency(makeModuleMap(prefix));
525             }
526 
527             context.paused.push(dep);
528         }

530         function execManager(manager) {
531             var i, ret, waitingCallbacks, err, errFile, errModuleTree,
532                 cb = manager.callback,
533                 fullName = manager.fullName,
534                 args = [],
535                 ary = manager.depArray;
536 
537             //Call the callback to define the module, if necessary.
538             if (cb && isFunction(cb)) {
539                 //Pull out the defined dependencies and pass the ordered
540                 //values to the callback.
541                 if (ary) {
542                     for (i = 0; i < ary.length; i++) {
543                         args.push(manager.deps[ary[i]]);
544                     }
545                 }
546 
547                 if (config.catchError.define) {
548                     try {
549                         ret = req.execCb(fullName, manager.callback, args, defined[fullName]);
550                     } catch (e) {
551                         err = e;
552                     }
553                 } else {
554                     ret = req.execCb(fullName, manager.callback, args, defined[fullName]);
555                 }
556 
557                 if (fullName) {
558                     //If setting exports via "module" is in play,
559                     //favor that over return value and exports. After that,
560                     //favor a non-undefined return value over exports use.
561                     if (manager.cjsModule && manager.cjsModule.exports !== undefined) {
562                         ret = defined[fullName] = manager.cjsModule.exports;
563                     } else if (ret === undefined && manager.usingExports) {
564                         //exports already set the defined value.
565                         ret = defined[fullName];
566                     } else {
567                         //Use the return value from the function.
568                         defined[fullName] = ret;
569                     }
570                 }
571             } else if (fullName) {
572                 //May just be an object definition for the module. Only
573                 //worry about defining if have a module name.
574                 ret = defined[fullName] = cb;
575             }
576 
577             //Clean up waiting. Do this before error calls, and before
578             //calling back waitingCallbacks, so that bookkeeping is correct
579             //in the event of an error and error is reported in correct order,
580             //since the waitingCallbacks will likely have errors if the
581             //onError function does not throw.
582             if (waiting[manager.waitId]) {
583                 delete waiting[manager.waitId];
584                 manager.isDone = true;
585                 context.waitCount -= 1;
586                 if (context.waitCount === 0) {
587                     //Clear the wait array used for cycles.
588                     waitAry = [];
589                 }
590             }
591 
592             if (err) {
593                 errFile = (fullName ? makeModuleMap(fullName).url : '') ||
594                            err.fileName || err.sourceURL;
595                 errModuleTree = err.moduleTree;
596                 err = makeError('defineerror', 'Error evaluating ' +
597                                 'module "' + fullName + '" at location "' +
598                                 errFile + '":\n' +
599                                 err + '\nfileName:' + errFile +
600                                 '\nlineNumber: ' + (err.lineNumber || err.line), err);
601                 err.moduleName = fullName;
602                 err.moduleTree = errModuleTree;
603                 return req.onError(err);
604             }
605 
606             if (fullName) {
607                 //If anything was waiting for this module to be defined,
608                 //notify them now.
609                 waitingCallbacks = managerCallbacks[fullName];
610                 if (waitingCallbacks) {
611                     for (i = 0; i < waitingCallbacks.length; i++) {
612                         waitingCallbacks[i].onDep(fullName, ret);
613                     }
614                     delete managerCallbacks[fullName];
615                 }
616             }
617 
618             return undefined;
619         }

621         function main(inName, depArray, callback, relModuleMap) {
622             var moduleMap = makeModuleMap(inName, relModuleMap),
623                 name = moduleMap.name,
624                 fullName = moduleMap.fullName,
625                 uniques = {},
626                 manager = {
627                     //Use a wait ID because some entries are anon
628                     //async require calls.
629                     waitId: name || reqWaitIdPrefix + (waitIdCounter++),
630                     depCount: 0,
631                     depMax: 0,
632                     prefix: moduleMap.prefix,
633                     name: name,
634                     fullName: fullName,
635                     deps: {},
636                     depArray: depArray,
637                     callback: callback,
638                     onDep: function (depName, value) {
639                         if (!(depName in manager.deps)) {
640                             manager.deps[depName] = value;
641                             manager.depCount += 1;
642                             if (manager.depCount === manager.depMax) {
643                                 //All done, execute!
644                                 execManager(manager);
645                             }
646                         }
647                     }
648                 },
649                 i, depArg, depName, cjsMod;
650 
651             if (fullName) {
652                 //If module already defined for context, or already loaded,
653                 //then leave. Also leave if jQuery is registering but it does
654                 //not match the desired version number in the config.
655                 if (fullName in defined || loaded[fullName] === true ||
656                     (fullName === "jquery" && config.jQuery &&
657                      config.jQuery !== callback().fn.jquery)) {
658                     return;
659                 }
660 
661                 //Set specified/loaded here for modules that are also loaded
662                 //as part of a layer, where onScriptLoad is not fired
663                 //for those cases. Do this after the inline define and
664                 //dependency tracing is done.
665                 specified[fullName] = true;
666                 loaded[fullName] = true;
667 
668                 //If module is jQuery set up delaying its dom ready listeners.
669                 if (fullName === "jquery" && callback) {
670                     jQueryCheck(callback());
671                 }
672             }
673 
674             //Add the dependencies to the deps field, and register for callbacks
675             //on the dependencies.
676             for (i = 0; i < depArray.length; i++) {
677                 depArg = depArray[i];
678                 //There could be cases like in IE, where a trailing comma will
679                 //introduce a null dependency, so only treat a real dependency
680                 //value as a dependency.
681                 if (depArg) {
682                     //Split the dependency name into plugin and name parts
683                     depArg = makeModuleMap(depArg, (name ? moduleMap : relModuleMap));
684                     depName = depArg.fullName;
685 
686                     //Fix the name in depArray to be just the name, since
687                     //that is how it will be called back later.
688                     depArray[i] = depName;
689 
690                     //Fast path CommonJS standard dependencies.
691                     if (depName === "require") {
692                         manager.deps[depName] = makeRequire(moduleMap);
693                     } else if (depName === "exports") {
694                         //CommonJS module spec 1.1
695                         manager.deps[depName] = defined[fullName] = {};

819         /**
820          * Checks if all modules for a context are loaded, and if so, evaluates the
821          * new ones in right dependency order.
822          *
823          * @private
824          */
825         function checkLoaded() {
826             var waitInterval = config.waitSeconds * 1000,
827                 //It is possible to disable the wait interval by using waitSeconds of 0.
828                 expired = waitInterval && (context.startTime + waitInterval) < new Date().getTime(),
829                 noLoads = "", hasLoadedProp = false, stillLoading = false, prop,
830                 err, manager;
831 
832             //If there are items still in the paused queue processing wait.
833             //This is particularly important in the sync case where each paused
834             //item is processed right away but there may be more waiting.
835             if (context.pausedCount > 0) {
836                 return undefined;
837             }
838 
839             //Determine if priority loading is done. If so clear the priority. If
840             //not, then do not check
841             if (config.priorityWait) {
842                 if (isPriorityDone()) {
843                     //Call resume, since it could have
844                     //some waiting dependencies to trace.
845                     resume();
846                 } else {
847                     return undefined;
848                 }
849             }
850 
851             //See if anything is still in flight.
852             for (prop in loaded) {
853                 if (!(prop in empty)) {
854                     hasLoadedProp = true;
855                     if (!loaded[prop]) {
856                         if (expired) {
857                             noLoads += prop + " ";
858                         } else {
859                             stillLoading = true;
860                             break;
861                         }
862                     }
863                 }
864             }
865 
866             //Check for exit conditions.
867             if (!hasLoadedProp && !context.waitCount) {
868                 //If the loaded object had no items, then the rest of
869                 //the work below does not need to be done.
870                 return undefined;
871             }
872             if (expired && noLoads) {
873                 //If wait time expired, throw error of unloaded modules.
874                 err = makeError("timeout", "Load timeout for modules: " + noLoads);
875                 err.requireType = "timeout";
876                 err.requireModules = noLoads;
877                 return req.onError(err);
878             }
879             if (stillLoading || context.scriptCount) {
880                 //Something is still waiting to load. Wait for it, but only
881                 //if a timeout is not already in effect.
882                 if ((isBrowser || isWebWorker) && !checkLoadedTimeoutId) {
883                     checkLoadedTimeoutId = setTimeout(function () {
884                         checkLoadedTimeoutId = 0;
885                         checkLoaded();
886                     }, 50);
887                 }
888                 return undefined;
889             }
890 
891             //If still have items in the waiting cue, but all modules have
892             //been loaded, then it means there are some circular dependencies
893             //that need to be broken.
894             //However, as a waiting thing is fired, then it can add items to
895             //the waiting cue, and those items should not be fired yet, so
896             //make sure to redo the checkLoaded call after breaking a single
897             //cycle, if nothing else loaded then this logic will pick it up
898             //again.
899             if (context.waitCount) {
900                 //Cycle through the waitAry, and call items in sequence.
901                 for (i = 0; (manager = waitAry[i]); i++) {
902                     forceExec(manager, {});
903                 }
904 
905                 //Only allow this recursion to a certain depth. Only
906                 //triggered by errors in calling a module in which its
907                 //modules waiting on it cannot finish loading, or some circular
908                 //dependencies that then may add more dependencies.
909                 //The value of 5 is a bit arbitrary. Hopefully just one extra
910                 //pass, or two for the case of circular dependencies generating
911                 //more work that gets resolved in the sync node case.
912                 if (checkLoadedDepth < 5) {
913                     checkLoadedDepth += 1;
914                     checkLoaded();
915                 }
916             }
917 
918             checkLoadedDepth = 0;
919 
920             //Check for DOM ready, and nothing is waiting across contexts.
921             req.checkReadyState();
922 923 
            return undefined;
924         }
