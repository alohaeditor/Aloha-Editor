230 		/**
231 		 * Internal version of updateSelection that adds the range parameter to be
232 		 * able to work around an IE bug that caused the current user selection
233 		 * sometimes to be on the body element.
234 		 * @param range a substitute for the current user selection. if not provided,
235 		 *   the current user selection will be used.
236 		 * @hide
237 		 */
238 		_updateSelection: function(event, range) {
239 			if (event !== undefined && event.originalEvent !== undefined &&
240 					event.originalEvent.stopSelectionUpdate === true) {
241 				return false;
242 			}
243 			
244 			if (typeof range === "undefined") {
245 				return false;
246 			}
247 
248 			this.rangeObject = range || new Aloha.Selection.SelectionRange(true);
249 			
250 			// Only execute the workaround when a valid rangeObject was provided
251 			if ( typeof this.rangeObject !== "undefined" && typeof this.rangeObject.startContainer !== "undefined" && this.rangeObject.endContainer !== "undefined") {
252 				// workaround for a nasty IE bug that allows the user to select text nodes inside areas with contenteditable "false"
253 				if ( (this.rangeObject.startContainer.nodeType === 3 && !jQuery(this.rangeObject.startContainer.parentNode).contentEditable())
254 						|| (this.rangeObject.endContainer.nodeType === 3 && !jQuery(this.rangeObject.endContainer.parentNode).contentEditable())) {
255 					Aloha.getSelection().removeAllRanges();
256 					return true;
257 				}
258 			} 
259 			
260 			// find the CAC (Common Ancestor Container) and update the selection Tree
261 			this.rangeObject.update();
262 
263 			// check if aloha-selection-changed event has been prevented
264 			if (this.isSelectionChangedPrevented()) {
265 				return true;
266 			}
267 
268 			// Only set the specific scope if an event was provided, which means
269 			// that somehow an editable was selected
270 			// TODO Bind code to aloha-selection-changed event to remove coupling to floatingmenu
271 			if (event !== undefined) {
272 				// Initiallly set the scope to 'continuoustext'
273 				FloatingMenu.setScope('Aloha.continuoustext');
274 			}
275 
276 			// throw the event that the selection has changed. Plugins now have the
277 			// chance to react on the chancurrentElements[childCount].children.lengthged selection
278 			Aloha.trigger('aloha-selection-changed', [ this.rangeObject, event ]);
279 
280 			return true;
281 		},

316 		/**
317 		 * Recursive inner function for generating the selection tree.
318 		 * TODO: remove this (was moved to range.js)
319 		 * @param rangeObject range object
320 		 * @param currentObject current DOM object for which the selection tree shall be generated
321 		 * @return array of SelectionTree objects for the children of the current DOM object
322 		 * @hide
323 		 */
324 		recursiveGetSelectionTree: function (rangeObject, currentObject) {
325 			// get all direct children of the given object
326 			var jQueryCurrentObject = jQuery(currentObject),
327 				childCount = 0,
328 				that = this,
329 				currentElements = [];
330 
331 			jQueryCurrentObject.contents().each(function(index) {
332 				var selectionType = 'none',
333 					startOffset = false,
334 					endOffset = false,
335 					collapsedFound = false,
336 					i, elementsLength,
337 					noneFound = false,
338 					partialFound = false,
339 					fullFound = false;
340 
341 				// check for collapsed selections between nodes
342 				if (rangeObject.isCollapsed() && currentObject === rangeObject.startContainer && rangeObject.startOffset == index) {
343 					// insert an extra selectiontree object for the collapsed selection here
344 					currentElements[childCount] = new Aloha.Selection.SelectionTree();
345 					currentElements[childCount].selection = 'collapsed';
346 					currentElements[childCount].domobj = undefined;
347 					that.inselection = false;
348 					collapsedFound = true;
349 					childCount++;
350 				}
351 
352 				if (!that.inselection && !collapsedFound) {
353 					// the start of the selection was not yet found, so look for it now
354 					// check whether the start of the selection is found here
355 
356 					// check is dependent on the node type
357 					switch(this.nodeType) {
358 					case 3: // text node
359 						if (this === rangeObject.startContainer) {
360 							// the selection starts here
361 							that.inselection = true;
362 
363 							// when the startoffset is > 0, the selection type is only partial
364 							selectionType = rangeObject.startOffset > 0 ? 'partial' : 'full';
365 							startOffset = rangeObject.startOffset;
366 							endOffset = this.length;
367 						}
368 						break;
369 					case 1: // element node
370 						if (this === rangeObject.startContainer && rangeObject.startOffset === 0) {
371 							// the selection starts here
372 							that.inselection = true;
373 							selectionType = 'full';
374 						}
375 						if (currentObject === rangeObject.startContainer && rangeObject.startOffset === index) {
376 							// the selection starts here
377 							that.inselection = true;
378 							selectionType = 'full';
379 						}
380 						break;
381 					}
382 				}
383 
384 				if (that.inselection && !collapsedFound) {
385 					if (selectionType == 'none') {
386 						selectionType = 'full';
387 					}
388 					// we already found the start of the selection, so look for the end of the selection now
389 					// check whether the end of the selection is found here
390 
391 					switch(this.nodeType) {
392 					case 3: // text node
393 						if (this === rangeObject.endContainer) {
394 							// the selection ends here
395 							that.inselection = false;
396 
397 							// check for partial selection here
398 							if (rangeObject.endOffset < this.length) {
399 								selectionType = 'partial';
400 							}
401 							if (startOffset === false) {
402 								startOffset = 0;
403 							}
404 							endOffset = rangeObject.endOffset;
405 						}
406 						break;
407 					case 1: // element node
408 						if (this === rangeObject.endContainer && rangeObject.endOffset === 0) {
409 							that.inselection = false;
410 						}
411 						break;
412 					}
413 					if (currentObject === rangeObject.endContainer && rangeObject.endOffset <= index) {
414 						that.inselection = false;
415 						selectionType = 'none';
416 					}
417 				}
418 
419 				// create the current selection tree entry
420 421 				currentElements[childCount] = new Aloha.Selection.SelectionTree();
422 				currentElements[childCount].domobj = this;
423 				currentElements[childCount].selection = selectionType;
424 				if (selectionType == 'partial') {
425 					currentElements[childCount].startOffset = startOffset;
426 					currentElements[childCount].endOffset = endOffset;
427 				}
428 
429 				// now do the recursion step into the current object
430 				currentElements[childCount].children = that.recursiveGetSelectionTree(rangeObject, this);
431 				elementsLength = currentElements[childCount].children.length;
432 
433 				// check whether a selection was found within the children
434 				if (elementsLength > 0) {
435 					for ( i = 0; i < elementsLength; ++i) {
436 						switch(currentElements[childCount].children[i].selection) {
437 						case 'none':
438 							noneFound = true;
439 							break;
440 						case 'full':
441 							fullFound = true;
442 							break;
443 						case 'partial':
444 							partialFound = true;
445 							break;
446 						}
447 					}
448 
449 					if (partialFound || (fullFound && noneFound)) {
450 						// found at least one 'partial' selection in the children, or both 'full' and 'none', so this element is also 'partial' selected
451 						currentElements[childCount].selection = 'partial';
452 					} else if (fullFound && !partialFound && !noneFound) {
453 						// only found 'full' selected children, so this element is also 'full' selected
454 						currentElements[childCount].selection = 'full';
455 					}
456 				}
457 
458 				childCount++;
459 			});
460 461 
462 			// extra check for collapsed selections at the end of the current element
463 			if (rangeObject.isCollapsed()
464 					&& currentObject === rangeObject.startContainer
465 					&& rangeObject.startOffset == currentObject.childNodes.length) {
466 				currentElements[childCount] = new Aloha.Selection.SelectionTree();
467 				currentElements[childCount].selection = 'collapsed';
468 				currentElements[childCount].domobj = undefined;
469 			}
470 
471 			return currentElements;
472 		},

487 		 * @param markupObject jQuery object of the markup to look for
488 		 * @param tagComparator method, which is used to compare the dom object and the jQuery markup object. the method must accept 2 parameters, the first is the domobj, the second is the jquery object. if no method is specified, the method this.standardTextLevelSemanticsComparator is used

494 			var
495 				domObj = !startOrEnd?rangeObject.startContainer:rangeObject.endContainer,
496 				that = this,
497 				parents = jQuery(domObj).parents(),
498 				returnVal = false,
499 				i = -1;
500 			
501 			// check if a comparison method was passed as parameter ...
502 			if (typeof tagComparator !== 'undefined' && typeof tagComparator !== 'function') {
503 				Aloha.Log.error(this,'parameter tagComparator is not a function');
504 			}
505 			// ... if not use this as standard tag comparison method
506 			if (typeof tagComparator === 'undefined') {
507 				tagComparator = function(domobj, markupObject) {
508 					return that.standardTextLevelSemanticsComparator(domobj, markupObject); // TODO should actually be this.getStandardTagComparator(markupObject)
509 				};
510 			}
511 		
512 			if (parents.length > 0) {
513 				parents.each(function() {
514 					// the limit object was reached (normally the Editable Element)
515 					if (this === limitObject) {
516 						Aloha.Log.debug(that,'reached limit dom obj');
517 						return false; // break() of jQuery .each(); THIS IS NOT THE FUNCTION RETURN VALUE
518 					}
519 					if (tagComparator(this, markupObject)) {
520 						if (returnVal === false) {
521 							returnVal = [];
522 						}
523 						Aloha.Log.debug(that,'reached object equal to markup');
524 						i++;
525 						returnVal[i] = this;
526 						return true; // continue() of jQuery .each(); THIS IS NOT THE FUNCTION RETURN VALUE
527 					}
528 				});
529 			}
530 			return returnVal;
