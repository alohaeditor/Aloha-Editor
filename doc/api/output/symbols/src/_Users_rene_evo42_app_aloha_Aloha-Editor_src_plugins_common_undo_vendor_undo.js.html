 11 // based on Backbone.js' inherits	
 12 var ctor = function(){};
 13 var inherits = function(parent, protoProps) {
 14 	var child;
 15 
 16 	if (protoProps && protoProps.hasOwnProperty('constructor')) {
 17 		child = protoProps.constructor;
 18 	} else {
 19 		child = function(){ return parent.apply(this, arguments); };
 20 	}
 21 
 22 	ctor.prototype = parent.prototype;
 23 	child.prototype = new ctor();
 24 	
 25 	if (protoProps) extend(child.prototype, protoProps);
 26 	
 27 	child.prototype.constructor = child;
 28 	child.__super__ = parent.prototype;
 29 	return child;
 30 };
 31 
 32 function extend(target, ref) {
 33 	for ( name in ref ) {
 34 		var value = ref[name];
 35 		if (value !== undefined) {
 36 			target[ name ] = value;
 37 		}
 38 	}
 39 	return target;
 40 };

 49 Undo.Stack = function() {
 50 	this.commands = [];
 51 	this.stackPosition = -1;
 52 	this.savePosition = -1;
 53 };
 54 
 55 extend(Undo.Stack.prototype, {
 56 	execute: function(command) {
 57 		this._clearRedo();
 58 		command.execute();
 59 		this.commands.push(command);
 60 		this.stackPosition++;
 61 		this.changed();
 62 	},
 63 	undo: function() {
 64 		this.commands[this.stackPosition].undo();
 65 		this.stackPosition--;
 66 		this.changed();
 67 	},
 68 	canUndo: function() {
 69 		return this.stackPosition >= 0;
 70 	},
 71 	redo: function() {
 72 		this.stackPosition++;
 73 		this.commands[this.stackPosition].redo();
 74 		this.changed();
 75 	},
 76 	canRedo: function() {
 77 		return this.stackPosition < this.commands.length - 1;
 78 	},
 79 	save: function() {
 80 		this.savePosition = this.stackPosition;
 81 		this.changed();
 82 	},
 83 	dirty: function() {
 84 		return this.stackPosition != this.savePosition;
 85 	},
 86 	_clearRedo: function() {
 87 		// TODO there's probably a more efficient way for this
 88 		this.commands = this.commands.slice(0, this.stackPosition + 1);
 89 	},
 90 	changed: function() {
 91 		// do nothing, override
 92 	}
 93 });
