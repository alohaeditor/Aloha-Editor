292 		/**
293 		 * Check if object can be edited by Aloha Editor
294 		 * @return {boolean } editable true if Aloha Editor can handle else false

297 		check: function() {
298 			/* TODO check those elements
299 			'map', 'meter', 'object', 'output', 'progress', 'samp',

305 			var	me = this,
306 			    obj = this.obj,
307 			    el = obj.get( 0 ),
308 			    nodeName = el.nodeName.toLowerCase(),
309 
310 				// supported elements
311 			    textElements = [ 'a', 'abbr', 'address', 'article', 'aside',
312 						'b', 'bdo', 'blockquote',  'cite', 'code', 'command',
313 						'del', 'details', 'dfn', 'div', 'dl', 'em', 'footer',
314 						'h1', 'h2', 'h3', 'h4', 'h5', 'h6', 'header', 'i',
315 						'ins', 'menu', 'nav', 'p', 'pre', 'q', 'ruby',
316 						'section', 'small', 'span', 'strong', 'sub', 'sup',
317 						'var' ],
318 			    i, div;
319 
320 			for ( i = 0; i < textElements.length; i++ ) {
321 				if ( nodeName === textElements[ i ] ) {
322 					return true;
323 				}
324 			}
325 
326 			// special handled elements
327 			switch ( nodeName ) {
328 				case 'label':
329 				case 'button':
330 					// TODO need some special handling.
331 					break;
332 				case 'textarea':
333 					// Create a div alongside the textarea
334 					div = jQuery( '<div id="' + this.getId() +
335 							'-aloha" class="aloha-textarea" />' )
336 								.insertAfter( obj );
337 
338 					// Resize the div to the textarea and
339 					// Populate the div with the value of the textarea
340 					// Then, hide the textarea
341 					div.height( obj.height() )
342 					   .width( obj.width() )
343 					   .html( obj.val() );
344 
345 					obj.hide();
346 
347 					// Attach a onsubmit to the form to place the HTML of the
348 					// div back into the textarea
349 					obj.parents( 'form:first' ).submit( function() {
350 						obj.val( me.getContents() );
351 					} );
352 
353 					// Swap textarea reference with the new div
354 					this.obj = div;
355 
356 					// Supported
357 					return true;
358 				default:
359 					break;
360 			}
361

363 			/*
364 			'canvas', 'audio', 'br', 'embed', 'fieldset', 'hgroup', 'hr',

439 		/**
440 		 * remove placeholder from contenteditable. If setCursor is true,
441 		 * will also set the cursor to the start of the selection. However,
442 		 * this will be ASYNCHRONOUS, so if you rely on the fact that
443 		 * the placeholder is removed after calling this method, setCursor
444 		 * should be false ( or not set )
445 		 *
446 		 * @return void
447 		 */
448 		removePlaceholder: function( obj, setCursor ) {
449 			var placeholderClass = this.placeholderClass,
450 			    range;
451 
452 	//		// remove browser br
453 	//		jQuery( 'br', obj ).remove();
454 
455 			// set the cursor // remove placeholder
456 			if ( setCursor === true ) {
457 				range = Selection.getRangeObject();
458 				if ( !range.select ) {
459 					return;
460 				}
461 				range.startContainer = range.endContainer = obj.get( 0 );
462 				range.startOffset = range.endOffset = 0;
463 				range.select();
464 
465 				window.setTimeout( function() {
466 					jQuery( '.' + placeholderClass, obj ).remove();
467 				}, 20 );
468 
469 			} else {
470 				jQuery( '.' + placeholderClass, obj ).remove();
471 			}

589 		/**
590 		 * activates an Editable for editing
591 		 * disables all other active items
592 		 * @method
593 		 */
594 		activate: function( e ) {
595 			// get active Editable before setting the new one.
596 			var oldActive = Aloha.getActiveEditable();
597 
598 			// We need to ommit this call when this flag is set to true.
599 			// This flag will only be set to true before the removePlaceholder method
600 			// is called since that method invokes a focus event which will again trigger
601 			// this method. We want to avoid double invokation of this method.
602 			if ( ignoreNextActivateEvent ) {
603 				ignoreNextActivateEvent = false;
604 				return;
605 			}
606 
607 			// handle special case in which a nested editable is focused by a click
608 			// in this case the "focus" event would be triggered on the parent element
609 			// which actually shifts the focus away to it's parent. this if is here to
610 			// prevent this situation
611 			if ( e && e.type === 'focus' && oldActive !== null
612 			     && oldActive.obj.parent().get( 0 ) === e.currentTarget ) {
613 				return;
614 			}
615 
616 			// leave immediately if this is already the active editable
617 			if ( this.isActive || this.isDisabled() ) {
618 				// we don't want parent editables to be triggered as well, so return false
619 				return;
620 			}
621 
622 			this.obj.addClass( 'aloha-editable-active' );
623 
624 			Aloha.activateEditable( this );
625 
626 			ignoreNextActivateEvent = true;
627 			this.removePlaceholder ( this.obj, true );
628 			ignoreNextActivateEvent = false;
629 
630 			this.isActive = true;
631 
632 			/**
633 			 * @event editableActivated fires after the editable has been activated by clicking on it.
634 			 * This event is triggered in Aloha's global scope Aloha
635 			 * @param {Event} e the event object
636 			 * @param {Array} a an array which contains a reference to last active editable on its first position, as well
637 			 * as the currently active editable on it's second position
638 			 */
639 			// trigger a 'general' editableActivated event
640 			Aloha.trigger( 'aloha-editable-activated', {
641 				'oldActive' : oldActive,
642 				'editable' : this
643 			} );
644 		},
645

722 		/**
723 		 * Generates and signals a smartContentChange event.
724 		 *
725 		 * A smart content change occurs when a special editing action, or a
726 		 * combination of interactions are performed by the user during the
727 		 * course of editing within an editable.
728 		 * The smart content change event would therefore signal to any
729 		 * component that is listening to this event, that content has been
730 		 * inserted into the editable that may need to be prococessed in a
731 		 * special way
732 		 * This is used for smart actions within the content/while editing.
733 		 * @param {Event} event
734 		 * @hide
735 		 */
736 		smartContentChange: function( event ) {
737 			var me = this,
738 			    uniChar = null,
739 			    re,
740 			    match;
741 
742 			// ignore meta keys like crtl+v or crtl+l and so on
743 			if ( event && ( event.metaKey || event.crtlKey || event.altKey ) ) {
744 				return false;
745 			}
746 
747 			if ( event && event.originalEvent ) {
748 				// regex to strip unicode
749 				re = new RegExp( "U\\+(\\w{4})" );
750 				match = re.exec( event.originalEvent.keyIdentifier );
751 
752 				// Use keyIdentifier if available
753 				if ( event.originalEvent.keyIdentifier && 1 === 2 ) {
754 					// @fixme: Because of "&& 1 === 2" above, all the below is
755 					// unreachable code
756 					if ( match !== null ) {
757 						uniChar = unescape( '%u' + match[1] );
758 					}
759 					if ( uniChar === null ) {
760 						uniChar = event.originalEvent.keyIdentifier;
761 					}
762 
763 				// FF & Opera don't support keyIdentifier
764 				} else {
765 					// Use among browsers reliable which http://api.jquery.com/keypress
766 					uniChar = ( this.keyCodeMap[ this.keyCode ] ||
767 									String.fromCharCode( event.which ) || 'unknown' );
768 				}
769 			}
770 
771 			// handle "Enter" -- it's not "U+1234" -- when returned via "event.originalEvent.keyIdentifier"
772 			// reference: http://www.w3.org/TR/2007/WD-DOM-Level-3-Events-20071221/keyset.html
773 			if ( jQuery.inArray( uniChar, this.sccDelimiters ) >= 0 ) {
774 				clearTimeout( this.sccTimerIdle );
775 				clearTimeout( this.sccTimerDelay );
776 
777 				this.sccTimerDelay = setTimeout( function() {
778 					Aloha.trigger( 'aloha-smart-content-changed', {
779 						'editable'        : me,
780 						'keyIdentifier'   : event.originalEvent.keyIdentifier,
781 						'keyCode'         : event.keyCode,
782 						'char'            : uniChar,
783 						'triggerType'     : 'keypress', // keypress, timer, blur, paste
784 						'snapshotContent' : me.getSnapshotContent()
785 					} );
786 
787 					console.debug( 'Aloha.Editable',
788 						'smartContentChanged: event type keypress triggered' );
789 	/*
790 					var r = Aloha.Selection.rangeObject;
791 					if ( r.isCollapsed()
792 						&& r.startContainer.nodeType == 3 ) {
793 
794 						var posDummy = jQuery( '<span id="GENTICS-Aloha-PosDummy" />' );
795 
796 						GENTICS.Utils.Dom.insertIntoDOM(
797 							posDummy,
798 							r,
799 							this.obj,
800 							null,
801 							false,
802 							false
803 						);
804 
805 						console.log( posDummy.offset().top, posDummy.offset().left );
806 
807 						GENTICS.Utils.Dom.removeFromDOM(
808 							posDummy,
809 							r,
810 							false
811 						);
812 
813 						r.select();
814 
815 					}
816 	*/
817 				}, this.sccDelay );
818 
819 			} else if ( event && event.type === 'paste' ) {
820 				Aloha.trigger( 'aloha-smart-content-changed', {
821 					'editable'        : me,
822 					'keyIdentifier'   : null,
823 					'keyCode'         : null,
824 					'char'            : null,
825 					'triggerType'     : 'paste',
826 					'snapshotContent' : me.getSnapshotContent()
827 				} );
828 
829 			} else if ( event && event.type === 'blur' ) {
830 				Aloha.trigger( 'aloha-smart-content-changed', {
831 					'editable'        : me,
832 					'keyIdentifier'   : null,
833 					'keyCode'         : null,
834 					'char'            : null,
835 					'triggerType'     : 'blur',
836 					'snapshotContent' : me.getSnapshotContent()
837 				} );
838 
839 			} else if ( uniChar !== null ) {
840 				// in the rare case idle time is lower then delay time
841 				clearTimeout( this.sccTimerDelay );
842 				clearTimeout( this.sccTimerIdle );
843 				this.sccTimerIdle = setTimeout( function() {
844 					Aloha.trigger( 'aloha-smart-content-changed', {
845 						'editable'        : me,
846 						'keyIdentifier'   : null,
847 						'keyCode'         : null,
848 						'char'            : null,
849 						'triggerType'     : 'idle',
850 						'snapshotContent' : me.getSnapshotContent()
851 					} );
852 				}, this.sccIdle );
853 			}
854 		},
