 11 define([
 12     'aloha',
 13 	'aloha/plugin',
 14 	'aloha/floatingmenu',
 15 ], function CiteClosure (Aloha, Plugin, FloatingMenu ) {
 16 	'use strict';
 17 
 18 	var jQuery = Aloha.jQuery;
 19 	
 20 
 21 	return Plugin.create('googletranslate', {
 22 		
 23 		/**
 24 		 * Configure the available languages to translate to. A complete list of supported languages can be found here:
 25 		 * http://code.google.com/apis/language/translate/v2/using_rest.html
 26 		 */
 27 		translateLangs: [ 'en', 'de', 'fr', 'it' ],
 28 		
 29 		/**
 30 		 * Google translate API key
 31 		 */
 32 		apiKey: "AIzaSyBgsTE6JQ5wsgERpi6m2xBY-9pCn2I5zcA",
 33 		
 34 		/**
 35 		 * Initialize the plugin
 36 		 */
 37 		init: function () {
 38 			var that = this;
 39 			
 40 			Aloha.require( ['css!googletranslate/css/googletranslate.css']);
 41 		
 42 			// use configured api key
 43 			if (this.settings.apiKey) {
 44 				this.apiKey = this.settings.apiKey;
 45 			}
 46 		
 47 			// create buttons for all translation langs
 48 			for (var i=0; i<this.translateLangs.length; i++) {
 49 			    FloatingMenu.addButton(
 50 			        'Aloha.continuoustext',
 51 			        new Aloha.ui.Button({
 52 			            'iconClass' : 'GENTICS_button GENTICS_button_googleTranslate_' + that.translateLangs[i],
 53 			            'size' : 'small',
 54 			            'onclick' : function (a,b,c) {
 55 			        		// determine target lang using the icon class
 56 			        		// there should obviously be a better way to
 57 			        		// determine which button has been clicked...
 58 			        		var targetLang = a.iconCls.replace("GENTICS_button GENTICS_button_googleTranslate_", "");
 59 			        		that.translate(targetLang);
 60 			        	},
 61 			            'tooltip' : that.translateLangs[i],
 62 			            'toggle' : false
 63 			        }),
 64 			        'Translate',
 65 			        1
 66 			    );
 67 			}
 68 		},
 69 		
 70 		/**
 71 		 * translate a text using the google translate api
 72 		 * @param target language
 73 		 * @return void
 74 		 */
 75 		translate: function (targetLang) {
 76 			var that = this;
 77 			var tree = Aloha.Selection.getRangeObject().getSelectionTree();
 78 			var tSource = new Array();
 79 			var c; // the current item
 80 			for (var i=0; i<tree.length; i++) {
 81 				c = tree[i];
 82 				if (c.selection != "none") {
 83 					if (c.selection == "full") {
 84 						tSource.push(jQuery(c.domobj).text());
 85 					} else if (c.selection == "partial") {
 86 						tSource.push(
 87 							jQuery(c.domobj).text().substring(c.startOffset, c.endOffset)
 88 						);
 89 					}
 90 				}
 91 			}
 92 		
 93 			if (tSource.length > 0) {
 94 				var qparams = "";
 95 				for (var i=0; i < tSource.length; i++) {
 96 					qparams += "&q=" + tSource[i];
 97 				}
 98 		
 99 				jQuery.ajax({ type: "GET",
100 					dataType: "jsonp",
101 					targetLang: targetLang, // store a reference to the target language to have it available when success function is triggered
102 					url: 'https://www.googleapis.com/language/translate/v2' +
103 						'?key=' + this.apiKey +
104 						'&target=' + targetLang + '&prettyprint=false' +
105 						qparams,
106 					success: function(res) {
107 						// handle errors
108 						if (typeof res.error == "object") {
109 							that.log("ERROR", "Unable to translate. Error: [" + res.error.code + "] " + res.error.message);
110 							return false;
111 						}
112 		
113 						// translation successful
114 						if (res.data && res.data.translations) {
115 							that.applyTranslation(res.data.translations, tree, this.targetLang);
116 						}
117 					}
118 				});
119 			}
120 		},
121 		
122 		/**
123 		 * apply a translation provided by google to the current selection
124 		 * @param translations list of translations provided by google
125 		 * @param tree the selection tree the translations will be applied to
126 		 * @param {String} lang language the content has been translated to
127 		 */
128 		applyTranslation: function (translations, tree, lang) {
129 			var key = 0;
130 			for (var i=0; i<tree.length; i++) {
131 
132 				var c = tree[i];
133 
134 				if (c.selection != "none") {
135 					if (c.selection == "full") {
136 						this.replaceText(c, translations[key].translatedText, lang);
137 					} else if (c.selection == "partial") {
138 						var txt = jQuery(c.domobj).text();
139 						var pre = txt.substring(0, c.startOffset);
140 						var post = txt.substring(c.endOffset, txt.length);
141 						this.replaceText(c, pre + translations[key].translatedText + post, null);
142 					}
143 					key++;
144 				}
145 			}
146 		},
147 		
148 		/**
149 		 * replace text in a selectionTree
150 		 * @param selectionTreeEntry a single selection tree entry where the text should be replaced
151 		 * @param text replacement text
152 		 * @return void
153 		 */
154 		replaceText: function (selectionTreeEntry, text, lang) {
155 			// GoogleTranslate API will trim spaces so we have to check if
156 			// there was a leading or trailing space
157 			// check if the first char of the original string is a space
158 			if (selectionTreeEntry.domobj.textContent.substring(0,1) == ' ') {
159 				text = ' ' + text;
160 			}
161 		
162 			// check if the last character of the original string is a space
163 			if (selectionTreeEntry.domobj.textContent.substring(
164 					selectionTreeEntry.domobj.textContent.length-1,selectionTreeEntry.domobj.textContent.length) == ' ') {
165 				text = text + ' ';
166 			}
167 		
168 			// special treatment for text nodes, which have to be replaced
169 			if (selectionTreeEntry.domobj.nodeType == 3) {
170 				jQuery(selectionTreeEntry.domobj)
171 					.replaceWith(document
172 					.createTextNode(text)
173 				);
174 			} else {
175 				jQuery(selectionTreeEntry.domobj)
176 					.html(text)
177 					// set the language attribute for non-text-nodes
178 					.attr('lang', lang);
179 			}
180 		}
181 	});
182 });
